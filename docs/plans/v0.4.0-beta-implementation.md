```rust
// crates/xlog-logic/src/parser/term.rs (add function)

fn parse_list(pair: Pair<Rule>) -> Result<Term, ParseError> {
    let mut inner = pair.into_inner();

    // Check for empty list
    let first = match inner.next() {
        None => return Ok(Term::List(vec![])),
        Some(p) => p,
    };

    let mut items = vec![parse_term(first)?];
    let mut tail: Option<Term> = None;

    for pair in inner {
        match pair.as_rule() {
            Rule::term if tail.is_none() => {
                // Check if this is the tail after |
                // The grammar handles this with the optional | term
                items.push(parse_term(pair)?);
            }
            _ => {
                // This is the tail
                tail = Some(parse_term(pair)?);
            }
        }
    }

    // If there's a tail, build cons cells
    if let Some(t) = tail {
        // Build from right to left: [a, b | T] -> Cons(a, Cons(b, T))
        let mut result = t;
        for item in items.into_iter().rev() {
            result = Term::ListCons {
                head: Box::new(item),
                tail: Box::new(result),
            };
        }
        Ok(result)
    } else {
        Ok(Term::List(items))
    }
}
```

```rust
// crates/xlog-runtime/src/unify/lists.rs
use xlog_logic::Term;
use crate::unify::{UnifyError, Substitution};

pub fn unify_list_with_cons(
    list: &[Term],
    head: &Term,
    tail: &Term,
    subst: &mut Substitution,
) -> Result<(), UnifyError> {
    if list.is_empty() {
        return Err(UnifyError::EmptyListWithCons);
    }

    // Unify head with first element
    super::unify(&list[0], head, subst)?;

    // Unify tail with rest of list
    let rest = if list.len() == 1 {
        Term::List(vec![])
    } else {
        Term::List(list[1..].to_vec())
    };

    super::unify(&rest, tail, subst)
}

pub fn unify_cons_with_cons(
    head1: &Term,
    tail1: &Term,
    head2: &Term,
    tail2: &Term,
    subst: &mut Substitution,
) -> Result<(), UnifyError> {
    super::unify(head1, head2, subst)?;
    super::unify(tail1, tail2, subst)
}

pub fn unify_lists(
    list1: &[Term],
    list2: &[Term],
    subst: &mut Substitution,
) -> Result<(), UnifyError> {
    if list1.len() != list2.len() {
        return Err(UnifyError::LengthMismatch(list1.len(), list2.len()));
    }

    for (t1, t2) in list1.iter().zip(list2.iter()) {
        super::unify(t1, t2, subst)?;
    }

    Ok(())
}
```

```rust
// crates/xlog-logic/src/builtins/lists.rs
use xlog_logic::{Term, Substitution};
use crate::builtins::{BuiltinResult, BuiltinError};

/// member(X, List) - true if X is an element of List
pub fn member(x: &Term, list: &Term, subst: &mut Substitution) -> BuiltinResult {
    match list {
        Term::List(items) => {
            for item in items {
                if unify(x, item, &mut subst.clone()).is_ok() {
                    return Ok(true);
                }
            }
            Ok(false)
        }
        Term::ListCons { head, tail } => {
            if unify(x, head, &mut subst.clone()).is_ok() {
                return Ok(true);
            }
            member(x, tail, subst)
        }
        _ => Err(BuiltinError::TypeMismatch("list", list.type_name())),
    }
}

/// append(L1, L2, L3) - L3 is the concatenation of L1 and L2
pub fn append(l1: &Term, l2: &Term, l3: &Term, subst: &mut Substitution) -> BuiltinResult {
    match l1 {
        Term::List(items) if items.is_empty() => {
            unify(l2, l3, subst)?;
            Ok(true)
        }
        Term::List(items) => {
            let mut result = match l2 {
                Term::List(l2_items) => l2_items.clone(),
                _ => return Err(BuiltinError::TypeMismatch("list", l2.type_name())),
            };
            for item in items.iter().rev() {
                result.insert(0, item.clone());
            }
            unify(&Term::List(result), l3, subst)?;
            Ok(true)
        }
        Term::ListCons { head, tail } => {
            // append([H|T1], L2, [H|T3]) :- append(T1, L2, T3)
            match l3 {
                Term::Variable(v) => {
                    // Generate result
                    let t3 = Term::Variable(fresh_var());
                    let result = Term::ListCons {
                        head: head.clone(),
                        tail: Box::new(t3.clone()),
                    };
                    unify(l3, &result, subst)?;
                    append(tail, l2, &t3, subst)
                }
                Term::ListCons { head: h3, tail: t3 } => {
                    unify(head, h3, subst)?;
                    append(tail, l2, t3, subst)
                }
                _ => Err(BuiltinError::TypeMismatch("list", l3.type_name())),
            }
        }
        _ => Err(BuiltinError::TypeMismatch("list", l1.type_name())),
    }
}

/// reverse(List, Reversed)
pub fn reverse(list: &Term, reversed: &Term, subst: &mut Substitution) -> BuiltinResult {
    let items = match list {
        Term::List(items) => items.clone(),
        _ => return Err(BuiltinError::TypeMismatch("list", list.type_name())),
    };

    let rev_items: Vec<Term> = items.into_iter().rev().collect();
    unify(&Term::List(rev_items), reversed, subst)?;
    Ok(true)
}

/// length(List, N)
pub fn length(list: &Term, n: &Term, subst: &mut Substitution) -> BuiltinResult {
    let len = match list {
        Term::List(items) => items.len(),
        Term::ListCons { head: _, tail } => {
            1 + list_length(tail)?
        }
        _ => return Err(BuiltinError::TypeMismatch("list", list.type_name())),
    };

    unify(&Term::Number(len as f64), n, subst)?;
    Ok(true)
}

/// select(X, List, Rest) - Rest is List with one occurrence of X removed
pub fn select(x: &Term, list: &Term, rest: &Term, subst: &mut Substitution) -> BuiltinResult {
    match list {
        Term::List(items) => {
            for (i, item) in items.iter().enumerate() {
                if unify(x, item, &mut subst.clone()).is_ok() {
                    let mut new_items = items.clone();
                    new_items.remove(i);
                    if unify(&Term::List(new_items), rest, subst).is_ok() {
                        return Ok(true);
                    }
                }
            }
            Ok(false)
        }
        _ => Err(BuiltinError::TypeMismatch("list", list.type_name())),
    }
}

/// nth(N, List, X) - X is the N-th element (0-indexed)
pub fn nth(n: &Term, list: &Term, x: &Term, subst: &mut Substitution) -> BuiltinResult {
    let index = match n {
        Term::Number(n) => *n as usize,
        _ => return Err(BuiltinError::TypeMismatch("number", n.type_name())),
    };

    let items = match list {
        Term::List(items) => items,
        _ => return Err(BuiltinError::TypeMismatch("list", list.type_name())),
    };

    if index >= items.len() {
        return Ok(false);
    }

    unify(&items[index], x, subst)?;
    Ok(true)
}

/// last(List, X) - X is the last element
pub fn last(list: &Term, x: &Term, subst: &mut Substitution) -> BuiltinResult {
    let items = match list {
        Term::List(items) => items,
        _ => return Err(BuiltinError::TypeMismatch("list", list.type_name())),
    };

    if items.is_empty() {
        return Ok(false);
    }

    unify(&items[items.len() - 1], x, subst)?;
    Ok(true)
}
```

```rust
// crates/xlog-logic/src/builtins/meta.rs
use xlog_logic::{Term, Substitution};
use crate::builtins::{BuiltinResult, BuiltinError};

/// ground(X) - succeeds if X is fully instantiated
pub fn ground(x: &Term, subst: &Substitution) -> BuiltinResult {
    Ok(is_ground(x, subst))
}

fn is_ground(term: &Term, subst: &Substitution) -> bool {
    match term {
        Term::Variable(v) => {
            match subst.get(v) {
                Some(bound) => is_ground(bound, subst),
                None => false,
            }
        }
        Term::Atom(_) | Term::Number(_) | Term::String(_) => true,
        Term::Compound { args, .. } => {
            args.iter().all(|a| is_ground(a, subst))
        }
        Term::List(items) => {
            items.iter().all(|i| is_ground(i, subst))
        }
        Term::ListCons { head, tail } => {
            is_ground(head, subst) && is_ground(tail, subst)
        }
    }
}

/// var(X) - succeeds if X is an unbound variable
pub fn var(x: &Term, subst: &Substitution) -> BuiltinResult {
    match x {
        Term::Variable(v) => Ok(subst.get(v).is_none()),
        _ => Ok(false),
    }
}

/// nonvar(X) - succeeds if X is not an unbound variable
pub fn nonvar(x: &Term, subst: &Substitution) -> BuiltinResult {
    Ok(!matches!(var(x, subst), Ok(true)))
}

/// atom(X) - succeeds if X is an atom
pub fn atom(x: &Term, _subst: &Substitution) -> BuiltinResult {
    Ok(matches!(x, Term::Atom(_)))
}

/// number(X) - succeeds if X is a number
pub fn number(x: &Term, _subst: &Substitution) -> BuiltinResult {
    Ok(matches!(x, Term::Number(_)))
}

/// compound(X) - succeeds if X is a compound term
pub fn compound(x: &Term, _subst: &Substitution) -> BuiltinResult {
    Ok(matches!(x, Term::Compound { .. } | Term::List(_) | Term::ListCons { .. }))
}

/// functor(Term, Functor, Arity)
pub fn functor(term: &Term, func: &Term, arity: &Term, subst: &mut Substitution) -> BuiltinResult {
    match term {
        Term::Compound { functor, args, .. } => {
            unify(&Term::Atom(functor.clone()), func, subst)?;
            unify(&Term::Number(args.len() as f64), arity, subst)?;
            Ok(true)
        }
        Term::Atom(a) => {
            unify(&Term::Atom(a.clone()), func, subst)?;
            unify(&Term::Number(0.0), arity, subst)?;
            Ok(true)
        }
        Term::Variable(_) => {
            // Construction mode: functor(X, foo, 2) -> X = foo(_, _)
            let f = match func {
                Term::Atom(a) => a.clone(),
                _ => return Err(BuiltinError::TypeMismatch("atom", func.type_name())),
            };
            let n = match arity {
                Term::Number(n) => *n as usize,
                _ => return Err(BuiltinError::TypeMismatch("number", arity.type_name())),
            };

            if n == 0 {
                unify(term, &Term::Atom(f), subst)?;
            } else {
                let args: Vec<Term> = (0..n).map(|_| Term::Variable(fresh_var())).collect();
                unify(term, &Term::Compound { functor: f, args, span: None }, subst)?;
            }
            Ok(true)
        }
        _ => Ok(false),
    }
}

/// T =.. L (univ) - deconstruct/construct term
pub fn univ(term: &Term, list: &Term, subst: &mut Substitution) -> BuiltinResult {
    match term {
        Term::Compound { functor, args, .. } => {
            // Deconstruction: foo(a, b) =.. [foo, a, b]
            let mut items = vec![Term::Atom(functor.clone())];
            items.extend(args.iter().cloned());
            unify(&Term::List(items), list, subst)?;
            Ok(true)
        }
        Term::Atom(a) => {
            // atom =.. [atom]
            unify(&Term::List(vec![Term::Atom(a.clone())]), list, subst)?;
            Ok(true)
        }
        Term::Variable(_) => {
            // Construction: X =.. [foo, a, b] -> X = foo(a, b)
            match list {
                Term::List(items) if !items.is_empty() => {
                    let functor = match &items[0] {
                        Term::Atom(a) => a.clone(),
                        _ => return Err(BuiltinError::TypeMismatch("atom", items[0].type_name())),
                    };

                    if items.len() == 1 {
                        unify(term, &Term::Atom(functor), subst)?;
                    } else {
                        let args = items[1..].to_vec();
                        unify(term, &Term::Compound { functor, args, span: None }, subst)?;
                    }
                    Ok(true)
                }
                _ => Err(BuiltinError::TypeMismatch("non-empty list", list.type_name())),
            }
        }
        _ => Ok(false),
    }
}

/// arg(N, Term, Arg) - Arg is the N-th argument of Term (1-indexed)
pub fn arg(n: &Term, term: &Term, x: &Term, subst: &mut Substitution) -> BuiltinResult {
    let index = match n {
        Term::Number(n) => (*n as usize).saturating_sub(1), // 1-indexed
        _ => return Err(BuiltinError::TypeMismatch("number", n.type_name())),
    };

    match term {
        Term::Compound { args, .. } => {
            if index >= args.len() {
                return Ok(false);
            }
            unify(&args[index], x, subst)?;
            Ok(true)
        }
        _ => Err(BuiltinError::TypeMismatch("compound", term.type_name())),
    }
}
```

```rust
// crates/xlog-runtime/src/eval/findall.rs
use xlog_logic::{Term, Goal, Program};
use crate::eval::{EvalContext, Solution};

/// findall(Template, Goal, List) - collect all solutions
pub fn findall(
    template: &Term,
    goal: &Goal,
    list: &Term,
    ctx: &mut EvalContext,
) -> Result<bool, EvalError> {
    let mut solutions = Vec::new();

    // Save current substitution
    let saved_subst = ctx.substitution.clone();

    // Find all solutions
    for solution in ctx.solve_goal(goal)? {
        // Apply substitution to template
        let result = solution.apply(template);
        solutions.push(result);
    }

    // Restore substitution (findall doesn't bind variables from goal)
    ctx.substitution = saved_subst;

    // Unify list with collected solutions
    let result_list = Term::List(solutions);
    ctx.unify(list, &result_list)
}

/// bagof(Template, Goal, List) - like findall but fails if no solutions
pub fn bagof(
    template: &Term,
    goal: &Goal,
    list: &Term,
    ctx: &mut EvalContext,
) -> Result<bool, EvalError> {
    let mut solutions = Vec::new();

    let saved_subst = ctx.substitution.clone();

    for solution in ctx.solve_goal(goal)? {
        let result = solution.apply(template);
        solutions.push(result);
    }

    ctx.substitution = saved_subst;

    if solutions.is_empty() {
        return Ok(false); // bagof fails if no solutions
    }

    let result_list = Term::List(solutions);
    ctx.unify(list, &result_list)
}

/// setof(Template, Goal, List) - like bagof but sorted and unique
pub fn setof(
    template: &Term,
    goal: &Goal,
    list: &Term,
    ctx: &mut EvalContext,
) -> Result<bool, EvalError> {
    let mut solutions = Vec::new();

    let saved_subst = ctx.substitution.clone();

    for solution in ctx.solve_goal(goal)? {
        let result = solution.apply(template);
        if !solutions.contains(&result) {
            solutions.push(result);
        }
    }

    ctx.substitution = saved_subst;

    if solutions.is_empty() {
        return Ok(false);
    }

    // Sort solutions (requires term ordering)
    solutions.sort_by(|a, b| term_compare(a, b));

    let result_list = Term::List(solutions);
    ctx.unify(list, &result_list)
}
```

```rust
// crates/xlog-logic/src/builtins/higher.rs

/// maplist(Pred, List) - apply Pred to each element
pub fn maplist1(pred: &Term, list: &Term, ctx: &mut EvalContext) -> Result<bool, EvalError> {
    let items = match list {
        Term::List(items) => items,
        _ => return Err(EvalError::TypeMismatch("list", list.type_name())),
    };

    for item in items {
        let goal = build_goal(pred, &[item.clone()])?;
        if !ctx.prove_goal(&goal)? {
            return Ok(false);
        }
    }

    Ok(true)
}

/// maplist(Pred, List1, List2) - apply Pred to corresponding elements
pub fn maplist2(
    pred: &Term,
    list1: &Term,
    list2: &Term,
    ctx: &mut EvalContext,
) -> Result<bool, EvalError> {
    let items1 = extract_list(list1)?;

    match list2 {
        Term::Variable(_) => {
            // Generate mode: maplist(double, [1,2,3], Ys)
            let mut results = Vec::new();
            for item in &items1 {
                let result_var = Term::Variable(fresh_var());
                let goal = build_goal(pred, &[item.clone(), result_var.clone()])?;
                if !ctx.prove_goal(&goal)? {
                    return Ok(false);
                }
                results.push(ctx.substitution.apply(&result_var));
            }
            ctx.unify(list2, &Term::List(results))
        }
        Term::List(items2) => {
            // Check mode: maplist(pred, [1,2], [a,b])
            if items1.len() != items2.len() {
                return Ok(false);
            }
            for (i1, i2) in items1.iter().zip(items2.iter()) {
                let goal = build_goal(pred, &[i1.clone(), i2.clone()])?;
                if !ctx.prove_goal(&goal)? {
                    return Ok(false);
                }
            }
            Ok(true)
        }
        _ => Err(EvalError::TypeMismatch("list or variable", list2.type_name())),
    }
}

fn build_goal(pred: &Term, args: &[Term]) -> Result<Goal, EvalError> {
    match pred {
        Term::Atom(name) => {
            Ok(Goal::Call {
                predicate: name.clone(),
                args: args.to_vec(),
            })
        }
        Term::Compound { functor, args: pred_args, .. } => {
            // Partial application: maplist(foo(X), [1,2,3])
            let mut all_args = pred_args.clone();
            all_args.extend(args.iter().cloned());
            Ok(Goal::Call {
                predicate: functor.clone(),
                args: all_args,
            })
        }
        _ => Err(EvalError::TypeMismatch("predicate", pred.type_name())),
    }
}
```

```rust
use pyo3::prelude::*;
use std::collections::HashMap;

pub struct TermEmbedding {
    pub name: String,
    pub weights: PyObject,           // torch.nn.Embedding or raw tensor
    pub vocab: HashMap<String, usize>,
    pub trainable: bool,
    pub optimizer: Option<PyObject>,
    pub embedding_dim: usize,
}

impl TermEmbedding {
    pub fn new_learnable(
        py: Python<'_>,
        name: String,
        vocab_size: usize,
        embedding_dim: usize,
    ) -> PyResult<Self> {
        let torch = py.import("torch")?;
        let nn = torch.getattr("nn")?;

        let weights = nn.call_method1("Embedding", (vocab_size, embedding_dim))?;

        Ok(Self {
            name,
            weights: weights.into(),
            vocab: HashMap::new(),
            trainable: true,
            optimizer: None,
            embedding_dim,
        })
    }

    pub fn new_pretrained(
        name: String,
        weights: PyObject,
        embedding_dim: usize,
        trainable: bool,
    ) -> Self {
        Self {
            name,
            weights,
            vocab: HashMap::new(),
            trainable,
            optimizer: None,
            embedding_dim,
        }
    }

    pub fn add_to_vocab(&mut self, terms: &[String]) {
        for term in terms {
            if !self.vocab.contains_key(term) {
                let idx = self.vocab.len();
                self.vocab.insert(term.clone(), idx);
            }
        }
    }

    pub fn lookup(&self, py: Python<'_>, terms: &[String]) -> PyResult<PyObject> {
        let indices: Vec<usize> = terms.iter()
            .map(|t| self.vocab.get(t).copied().unwrap_or(0))
            .collect();

        let torch = py.import("torch")?;
        let idx_tensor = torch.call_method1("tensor", (indices,))?;

        self.weights.call_method1(py, "__call__", (idx_tensor,))
    }
}

pub struct EmbeddingRegistry {
    embeddings: HashMap<String, TermEmbedding>,
}

impl EmbeddingRegistry {
    pub fn new() -> Self {
        Self { embeddings: HashMap::new() }
    }

    pub fn register(&mut self, embedding: TermEmbedding) {
        self.embeddings.insert(embedding.name.clone(), embedding);
    }

    pub fn get(&self, name: &str) -> Option<&TermEmbedding> {
        self.embeddings.get(name)
    }

    pub fn get_mut(&mut self, name: &str) -> Option<&mut TermEmbedding> {
        self.embeddings.get_mut(name)
    }

    pub fn names(&self) -> Vec<&str> {
        self.embeddings.keys().map(|s| s.as_str()).collect()
    }
}
```

```rust
#[pymethods]
impl Program {
    #[pyo3(signature = (name, vocab_size=None, embedding_dim=None, weights=None, trainable=true, optimizer=None, lr=1e-3))]
    fn register_embedding(
        &mut self,
        py: Python<'_>,
        name: String,
        vocab_size: Option<usize>,
        embedding_dim: Option<usize>,
        weights: Option<PyObject>,
        trainable: bool,
        optimizer: Option<PyObject>,
        lr: f64,
    ) -> PyResult<()> {
        let embedding = if let Some(w) = weights {
            // Pretrained
            let dim: usize = w.getattr(py, "shape")?.get_item(py, 1)?.extract(py)?;
            TermEmbedding::new_pretrained(name, w, dim, trainable)
        } else {
            // Learnable
            let vs = vocab_size.ok_or_else(|| {
                PyErr::new::<pyo3::exceptions::PyValueError, _>("vocab_size required for learnable embedding")
            })?;
            let dim = embedding_dim.ok_or_else(|| {
                PyErr::new::<pyo3::exceptions::PyValueError, _>("embedding_dim required for learnable embedding")
            })?;

            let mut emb = TermEmbedding::new_learnable(py, name, vs, dim)?;

            // Create optimizer
            if let Some(opt_class) = optimizer {
                let params = emb.weights.call_method0(py, "parameters")?;
                let kwargs = pyo3::types::PyDict::new(py);
                kwargs.set_item("lr", lr)?;
                emb.optimizer = Some(opt_class.call(py, (params,), Some(kwargs))?);
            }

            emb
        };

        self.embeddings.register(embedding);
        Ok(())
    }

    fn embedding_names(&self) -> Vec<String> {
        self.embeddings.names().iter().map(|s| s.to_string()).collect()
    }

    fn add_to_vocab(&mut self, name: String, terms: Vec<String>) -> PyResult<()> {
        self.embeddings.get_mut(&name)
            .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyValueError, _>("Embedding not found"))?
            .add_to_vocab(&terms);
        Ok(())
    }
}
```

```rust
#[derive(Debug, Clone, Copy)]
pub enum TensorOp {
    Dot,
    Cosine,
    Sigmoid,
    Softmax,
    Relu,
    Add,
    Mul,
    Sub,
    Norm,
}

impl TensorOp {
    pub fn from_name(name: &str) -> Option<Self> {
        match name {
            "dot" => Some(Self::Dot),
            "cosine" => Some(Self::Cosine),
            "sigmoid" => Some(Self::Sigmoid),
            "softmax" => Some(Self::Softmax),
            "relu" => Some(Self::Relu),
            "add" => Some(Self::Add),
            "mul" => Some(Self::Mul),
            "sub" => Some(Self::Sub),
            "norm" => Some(Self::Norm),
            _ => None,
        }
    }

    pub fn evaluate(&self, py: Python<'_>, args: &[PyObject]) -> PyResult<PyObject> {
        let torch = py.import("torch")?;
        let f = torch.getattr("nn")?.getattr("functional")?;

        match self {
            Self::Dot => {
                // dot(x, y) -> scalar
                let x = &args[0];
                let y = &args[1];
                torch.call_method1("dot", (x, y))
            }
            Self::Cosine => {
                // cosine(x, y) -> scalar
                let x = args[0].call_method1(py, "unsqueeze", (0,))?;
                let y = args[1].call_method1(py, "unsqueeze", (0,))?;
                f.call_method1("cosine_similarity", (x, y))
            }
            Self::Sigmoid => {
                torch.call_method1("sigmoid", (&args[0],))
            }
            Self::Softmax => {
                f.call_method("softmax", (&args[0],), Some([("dim", -1)].into_py_dict(py)))
            }
            Self::Relu => {
                f.call_method1("relu", (&args[0],))
            }
            Self::Add => {
                torch.call_method1("add", (&args[0], &args[1]))
            }
            Self::Mul => {
                torch.call_method1("mul", (&args[0], &args[1]))
            }
            Self::Sub => {
                torch.call_method1("sub", (&args[0], &args[1]))
            }
            Self::Norm => {
                torch.call_method("norm", (&args[0],), Some([("p", 2)].into_py_dict(py)))
            }
        }
    }

    pub fn backward(&self, py: Python<'_>, args: &[PyObject], grad_out: &PyObject) -> PyResult<Vec<PyObject>> {
        // Implement gradient computation for each op
        // For now, rely on PyTorch autograd
        match self {

                // d/dx dot(x,y) = y, d/dy dot(x,y) = x
                Ok(vec![args[1].clone(), args[0].clone()])
            }
            Self::Sigmoid => {
                // d/dx sigmoid(x) = sigmoid(x) * (1 - sigmoid(x))
                let torch = py.import("torch")?;
                let s = torch.call_method1("sigmoid", (&args[0],))?;
                let one = torch.call_method1("ones_like", (&s,))?;
                let grad = torch.call_method1("mul", (&s, torch.call_method1("sub", (&one, &s))?))?;
                let result = torch.call_method1("mul", (grad_out, &grad))?;
                Ok(vec![result.into()])
            }
            _ => {
                // Use autograd for complex cases
```

```rust
// crates/pyxlog/src/dataloader.rs
use pyo3::prelude::*;
use rand::seq::SliceRandom;

#[pyclass]
pub struct DataLoader {
    queries: Vec<String>,
    batch_size: usize,
    shuffle: bool,
    current_epoch: Vec<String>,
    position: usize,
}

#[pymethods]
impl DataLoader {
    #[new]
    #[pyo3(signature = (queries, batch_size=32, shuffle=true))]
    fn new(queries: Vec<String>, batch_size: usize, shuffle: bool) -> Self {
        Self {
            queries,
            batch_size,
            shuffle,
            current_epoch: Vec::new(),
            position: 0,
        }
    }

    fn __iter__(mut slf: PyRefMut<'_, Self>) -> PyRefMut<'_, Self> {
        slf.position = 0;
        slf.current_epoch = slf.queries.clone();
        if slf.shuffle {
            let mut rng = rand::thread_rng();
            slf.current_epoch.shuffle(&mut rng);
        }
        slf
    }

    fn __next__(mut slf: PyRefMut<'_, Self>) -> Option<Vec<String>> {
        if slf.position >= slf.current_epoch.len() {
            return None;
        }

        let end = (slf.position + slf.batch_size).min(slf.current_epoch.len());
        let batch = slf.current_epoch[slf.position..end].to_vec();
        slf.position = end;

        Some(batch)
    }

    fn __len__(&self) -> usize {
        (self.queries.len() + self.batch_size - 1) / self.batch_size
    }
}
```

```rust
// crates/pyxlog/src/stop.rs
use pyo3::prelude::*;

#[pyclass]
#[derive(Clone)]
pub struct StopCondition {
    conditions: Vec<StopConditionType>,
    combine_or: bool, // true = OR, false = AND
    history: Vec<f64>,
}

#[derive(Clone)]
enum StopConditionType {
    Epochs(usize),
    Threshold { accuracy: f64 },
    Plateau { patience: usize, min_delta: f64 },
}

#[pymethods]
impl StopCondition {
    #[staticmethod]
    fn epochs(n: usize) -> Self {
        Self {
            conditions: vec![StopConditionType::Epochs(n)],
            combine_or: true,
            history: Vec::new(),
        }
    }

    #[staticmethod]
    #[pyo3(signature = (accuracy=0.99))]
    fn threshold(accuracy: f64) -> Self {
        Self {
            conditions: vec![StopConditionType::Threshold { accuracy }],
            combine_or: true,
            history: Vec::new(),
        }
    }

    #[staticmethod]
    #[pyo3(signature = (patience=10, min_delta=0.001))]
    fn plateau(patience: usize, min_delta: f64) -> Self {
        Self {
            conditions: vec![StopConditionType::Plateau { patience, min_delta }],
            combine_or: true,
            history: Vec::new(),
        }
    }

    fn __or__(&self, other: &Self) -> Self {
        let mut combined = self.conditions.clone();
        combined.extend(other.conditions.clone());
        Self {
            conditions: combined,
            combine_or: true,
            history: Vec::new(),
        }
    }

    fn __and__(&self, other: &Self) -> Self {
        let mut combined = self.conditions.clone();
        combined.extend(other.conditions.clone());
        Self {
            conditions: combined,
            combine_or: false,
            history: Vec::new(),
        }
    }

    #[pyo3(signature = (epoch, loss, accuracy=None))]
    fn should_stop(&mut self, epoch: usize, loss: f64, accuracy: Option<f64>) -> bool {
        self.history.push(loss);

        let results: Vec<bool> = self.conditions.iter().map(|c| {
            match c {
                StopConditionType::Epochs(n) => epoch >= *n,
                StopConditionType::Threshold { accuracy: thresh } => {
                    accuracy.map_or(false, |a| a >= *thresh)
                }
                StopConditionType::Plateau { patience, min_delta } => {
                    if self.history.len() < *patience + 1 {
                        return false;
                    }
                    let recent = &self.history[self.history.len() - patience - 1..];
                    let best = recent.iter().cloned().fold(f64::INFINITY, f64::min);
                    let current = recent.last().unwrap();
                    (best - current).abs() < *min_delta
                }
            }
        }).collect();

        if self.combine_or {
            results.iter().any(|&r| r)
        } else {
            results.iter().all(|&r| r)
        }
    }
}
```

```rust
pub fn train_model(
    py: Python<'_>,
    program: &mut crate::program::Program,
    queries: Vec<String>,
    epochs: usize,
    batch_size: usize,
    log_iter: usize,
    shuffle: bool,
    stop_condition: Option<crate::stop::StopCondition>,
    test_fn: Option<PyObject>,
    test_iter: usize,
) -> PyResult<TrainingHistory> {
    let loader = crate::dataloader::DataLoader::new(queries.clone(), batch_size, shuffle);
    let mut history = TrainingHistory::new();
    let mut stop = stop_condition.unwrap_or_else(|| crate::stop::StopCondition::epochs(epochs));
    let mut total_iters = 0;

    for epoch in 0..epochs {
        let mut epoch_loss = 0.0;
        let mut num_batches = 0;

        for batch in loader.iter(py)? {
            program.zero_grad()?;

            let mut batch_loss = 0.0;
            for query in &batch {
                batch_loss += program.forward_backward(query)?;
            }
            batch_loss /= batch.len() as f64;

            program.optimizer_step()?;

            epoch_loss += batch_loss;
            num_batches += 1;
            total_iters += 1;

            history.add_batch(batch_loss);

            if total_iters % log_iter == 0 {
                println!("Iter {}: loss={:.4}", total_iters, batch_loss);
            }

            if let Some(ref test) = test_fn {
                if total_iters % test_iter == 0 {
                    let accuracy: f64 = test.call1(py, (program,))?.extract(py)?;
                    history.accuracies.push(accuracy);
                    println!("Test accuracy: {:.2}%", accuracy * 100.0);

                    if stop.should_stop(epoch, epoch_loss / num_batches as f64, Some(accuracy)) {
                        println!("Stopping early at epoch {}", epoch + 1);
                        return Ok(history);
                    }
                }
            }
        }

        let avg_loss = epoch_loss / num_batches as f64;
        history.add_epoch(avg_loss);
        println!("Epoch {}/{}: avg_loss={:.4}", epoch + 1, epochs, avg_loss);

        if stop.should_stop(epoch + 1, avg_loss, None) {
            println!("Stopping at epoch {}", epoch + 1);
            break;
        }
    }

    Ok(history)
```

```python
# python/pyxlog/autograd.py
import torch
from typing import List, Tuple, Dict, Any
import pyxlog

class XLogLayer(torch.autograd.Function):
    """Custom autograd function for XLOG probabilistic inference.

    This layer enables end-to-end differentiable training through XLOG programs.
    It takes neural network outputs and queries, runs probabilistic inference,
    and returns query probabilities with full gradient support.
    """

    @staticmethod
    def forward(
        ctx: Any,
        program: "pyxlog.Program",
        queries: List[str],
        *network_outputs: torch.Tensor
    ) -> torch.Tensor:
        """
        Forward pass: run probabilistic inference.

        Args:
            ctx: Autograd context for saving tensors
            program: Compiled XLOG program
            queries: List of query strings
            network_outputs: Tensors from neural networks (ordered by network name)

        Returns:
            Tensor of query probabilities [batch_size]
        """
        # Save for backward
        ctx.program = program
        ctx.queries = queries
        ctx.num_networks = len(network_outputs)

        # Convert tensors to numpy for Rust interface
        network_names = program.network_names()
        outputs_dict = {}
        for i, (name, tensor) in enumerate(zip(network_names, network_outputs)):
            outputs_dict[name] = tensor.detach().cpu().numpy()

        # Run inference with neural outputs
        results, grad_cache = program._forward_with_cache(
            queries,
            outputs_dict
        )

        # Save gradient cache for backward
        ctx.save_for_backward(torch.tensor(grad_cache))
        ctx.network_names = network_names
        ctx.output_shapes = [t.shape for t in network_outputs]

        return torch.tensor(results.probabilities, dtype=torch.float32)

    @staticmethod
    def backward(
        ctx: Any,
        grad_output: torch.Tensor
    ) -> Tuple[None, None, ...]:
        """
        Backward pass: compute gradients w.r.t. neural network outputs.

        Uses the chain rule: dL/d(network_output) = dL/dP * dP/d(network_output)
        where P is the query probability.

        Args:
            ctx: Autograd context with saved tensors
            grad_output: Gradient of loss w.r.t. output probabilities

        Returns:
            Gradients for each input (None for program and queries)
        """
        grad_cache = ctx.saved_tensors[0].numpy()

        # Get gradients w.r.t. each network's outputs
        network_grads = ctx.program._backward(
            grad_cache,
            grad_output.numpy(),
            ctx.queries
        )

        # Convert to tensors with correct shapes
        result_grads = []
        for name, shape in zip(ctx.network_names, ctx.output_shapes):
            if name in network_grads:
                grad = torch.tensor(
                    network_grads[name],
                    dtype=torch.float32
                ).reshape(shape)
            else:
                grad = torch.zeros(shape, dtype=torch.float32)
            result_grads.append(grad)

        return (None, None) + tuple(result_grads)


class XLogModule(torch.nn.Module):
    """PyTorch module wrapper for XLOG programs.

    Example usage:
        program = pyxlog.Program.compile(source)
        xlog_module = XLogModule(program, ["digit_net"])

        # In forward:
        digit_probs = self.digit_net(images)
        query_probs = xlog_module(queries, digit_probs)
        loss = -torch.log(query_probs).mean()
    """

    def __init__(self, program: "pyxlog.Program", network_names: List[str]):
        super().__init__()
        self.program = program
```

```rust
use numpy::{PyArray1, PyArray2, PyReadonlyArray1, PyReadonlyArray2};
use std::collections::HashMap;

/// Gradient cache storing intermediate values for backward pass
#[derive(Clone)]
pub struct GradientCache {
    /// Leaf node contributions per network
    pub leaf_contributions: HashMap<String, Vec<LeafContribution>>,
    /// Circuit evaluation results
    pub circuit_values: Vec<f64>,
    /// Query to circuit mapping
    pub query_circuits: Vec<usize>,
}

#[derive(Clone)]
pub struct LeafContribution {
    pub leaf_id: usize,
    pub network: String,
    pub input_idx: usize,
    pub output_idx: usize,
    pub probability: f64,
    pub circuit_id: usize,
}

impl GradientCache {
    pub fn new() -> Self {
        Self {
            leaf_contributions: HashMap::new(),
            circuit_values: Vec::new(),
            query_circuits: Vec::new(),
        }
    }
}

/// Forward pass with gradient cache
pub fn forward_with_cache(
    program: &mut crate::program::Program,
    queries: &[String],
    network_outputs: HashMap<String, Vec<f64>>,
) -> PyResult<(Vec<f64>, GradientCache)> {
    let mut cache = GradientCache::new();
    let mut probabilities = Vec::with_capacity(queries.len());

    for (q_idx, query) in queries.iter().enumerate() {
        // Parse and ground query
        let grounded = program.ground_query(query)?;

        // Collect neural predicate instances
        let neural_instances = program.collect_neural_instances(&grounded)?;

        // Map network outputs to circuit leaves
        for instance in &neural_instances {
            let outputs = network_outputs.get(&instance.network)
                .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyValueError, _>(
                    format!("Network '{}' outputs not provided", instance.network)
                ))?;

            // Get probability for this instance's output
            let prob = outputs[instance.flat_index()];

            // Record leaf contribution for backward
            cache.leaf_contributions
                .entry(instance.network.clone())
                .or_insert_with(Vec::new)
                .push(LeafContribution {
                    leaf_id: instance.leaf_id,
                    network: instance.network.clone(),
                    input_idx: instance.input_idx,
                    output_idx: instance.output_idx,
                    probability: prob,
                    circuit_id: q_idx,
                });
        }

        // Run circuit evaluation
        let prob = program.evaluate_circuit_with_leaves(&grounded, &neural_instances)?;
        probabilities.push(prob);

        cache.circuit_values.push(prob);
        cache.query_circuits.push(q_idx);
    }

    Ok((probabilities, cache))
}

/// Backward pass computing gradients
pub fn backward(
    program: &crate::program::Program,
    cache: &GradientCache,
    grad_output: &[f64],
    queries: &[String],
) -> PyResult<HashMap<String, Vec<f64>>> {
    let mut network_grads: HashMap<String, Vec<f64>> = HashMap::new();

    // For each leaf contribution, compute gradient
    for (network, contributions) in &cache.leaf_contributions {
        let mut grads = vec![0.0; program.network_output_size(network)?];

        for contrib in contributions {
            let circuit_idx = contrib.circuit_id;
            let circuit_value = cache.circuit_values[circuit_idx];

            // Compute dP/d(leaf) using circuit derivative
            let leaf_derivative = program.circuit_leaf_derivative(
                circuit_idx,
                contrib.leaf_id,
            )?;

            // Chain rule: dL/d(output) = dL/dP * dP/d(leaf)
            let grad = grad_output[circuit_idx] * leaf_derivative;

            // Accumulate gradient for this network output
            let flat_idx = contrib.input_idx * program.network_num_outputs(network)?
                + contrib.output_idx;
            grads[flat_idx] += grad;
        }

        network_grads.insert(network.clone(), grads);
    }

    Ok(network_grads)
}

/// Python bindings
#[pymethods]
impl crate::program::Program {
    fn _forward_with_cache(
        &mut self,
        py: Python<'_>,
        queries: Vec<String>,
        network_outputs: HashMap<String, &PyArray2<f64>>,
    ) -> PyResult<(Py<PyArray1<f64>>, Vec<u8>)> {
        // Convert numpy arrays to Vec
        let outputs: HashMap<String, Vec<f64>> = network_outputs
            .into_iter()
            .map(|(k, v)| (k, v.readonly().as_array().iter().copied().collect()))
            .collect();

        let (probs, cache) = forward_with_cache(self, &queries, outputs)?;

        // Serialize cache for storage
        let cache_bytes = bincode::serialize(&cache)
            .map_err(|e| PyErr::new::<pyo3::exceptions::PyValueError, _>(e.to_string()))?;

        Ok((PyArray1::from_vec(py, probs).into(), cache_bytes))
    }

    fn _backward(
        &self,
        cache_bytes: Vec<u8>,
        grad_output: &PyArray1<f64>,
        queries: Vec<String>,
    ) -> PyResult<HashMap<String, Vec<f64>>> {
        let cache: GradientCache = bincode::deserialize(&cache_bytes)
            .map_err(|e| PyErr::new::<pyo3::exceptions::PyValueError, _>(e.to_string()))?;

        let grad_output: Vec<f64> = grad_output.readonly().as_array().iter().copied().collect();

        backward(self, &cache, &grad_output, &queries)
    }
```

```rust
// crates/pyxlog/src/loss.rs
use pyo3::prelude::*;
use std::collections::HashMap;

/// Compute semantic loss: -sum(weight_i * log P(query_i))
pub fn semantic_loss(
    program: &mut crate::program::Program,
    queries: Vec<(String, f64)>,  // (query, weight) pairs
    infoloss_weight: f64,
) -> PyResult<f64> {
    let mut total_loss = 0.0;
    let mut total_entropy = 0.0;

    for (query, weight) in &queries {
        // Evaluate query probability
        let result = program.evaluate_query(query)?;

        // NLL loss: -log P(query)
        let log_prob = result.probability.ln();
        total_loss -= weight * log_prob;

        // Entropy of neural outputs (for infoloss)
        if infoloss_weight > 0.0 {
            for (_, probs) in &result.neural_outputs {
                total_entropy += compute_entropy(probs);
            }
        }
    }

    // Add entropy regularization
    total_loss += infoloss_weight * total_entropy;

    Ok(total_loss)
}

/// Compute entropy: -sum(p * log(p))
fn compute_entropy(probs: &[f64]) -> f64 {
    probs.iter()
        .filter(|&&p| p > 1e-10)
        .map(|&p| -p * p.ln())
        .sum()
}

/// Result with gradient information
pub struct SemanticLossResult {
    pub loss: f64,
    pub gradients: HashMap<String, Vec<f64>>,
}

/// Compute semantic loss with gradients for PyTorch
pub fn semantic_loss_with_grad(
    program: &mut crate::program::Program,
    queries: Vec<(String, f64)>,
    infoloss_weight: f64,
) -> PyResult<SemanticLossResult> {
    let mut total_loss = 0.0;
    let mut all_gradients: HashMap<String, Vec<f64>> = HashMap::new();

    for (query, weight) in &queries {
        let result = program.evaluate_query_with_grad(query)?;

        // NLL loss
        let prob = result.probability;
        let log_prob = prob.ln();
        total_loss -= weight * log_prob;

        // Gradient of -log(p) w.r.t. p is -1/p
        let grad_scale = -weight / prob;

        // Accumulate gradients
        for (network, leaf_grads) in &result.leaf_gradients {
            let entry = all_gradients.entry(network.clone())
                .or_insert_with(|| vec![0.0; leaf_grads.len()]);
            for (i, g) in leaf_grads.iter().enumerate() {
                entry[i] += grad_scale * g;
            }
        }
    }

    // Add entropy gradient if needed
    if infoloss_weight > 0.0 {
        // d/dp (-p*log(p)) = -log(p) - 1
        for (network, entry) in &mut all_gradients {
            if let Some(probs) = program.get_network_outputs(network)? {
                for (i, (&p, grad)) in probs.iter().zip(entry.iter_mut()).enumerate() {
                    if p > 1e-10 {
                        *grad += infoloss_weight * (-p.ln() - 1.0);
                    }
                }
            }
        }
    }

    Ok(SemanticLossResult {
        loss: total_loss,
        gradients: all_gradients,
    })
}

#[pymethods]
impl crate::program::Program {
    #[pyo3(signature = (queries, infoloss=0.0))]
    fn semantic_loss(
        &mut self,
        py: Python<'_>,
        queries: PyObject,
        infoloss: f64,
    ) -> PyResult<PyObject> {
        // Parse queries input (string, list, or dict)
        let query_weights = parse_query_input(py, queries)?;

        // Compute loss with gradients
        let result = semantic_loss_with_grad(self, query_weights, infoloss)?;

        // Create PyTorch tensor with backward hook
        let torch = py.import("torch")?;

        let loss_tensor = torch.call_method1("tensor", (result.loss,))?
            .call_method1("requires_grad_", (true,))?;

        // Attach gradient function via custom backward hook
        // Store gradients in program for retrieval during backward
        self.pending_gradients = Some(result.gradients);

        Ok(loss_tensor.into())
    }
}

fn parse_query_input(py: Python<'_>, input: PyObject) -> PyResult<Vec<(String, f64)>> {
    // String: single query
    if let Ok(s) = input.extract::<String>(py) {
        return Ok(vec![(s, 1.0)]);
    }

    // List: multiple queries with weight 1.0
    if let Ok(list) = input.extract::<Vec<String>>(py) {
        return Ok(list.into_iter().map(|q| (q, 1.0)).collect());
    }

    // Dict: weighted queries
    if let Ok(dict) = input.extract::<HashMap<String, f64>>(py) {
        return Ok(dict.into_iter().collect());
    }

    Err(PyErr::new::<pyo3::exceptions::PyTypeError, _>(
        "queries must be str, list[str], or dict[str, float]"
    ))
}
```

```rust
// crates/xlog-neural/src/cache.rs
use lru::LruCache;
use std::hash::{Hash, Hasher};
use std::num::NonZeroUsize;

#[derive(Clone, Debug)]
pub struct CacheConfig {
    pub max_entries: usize,
    pub memory_budget_mb: Option<usize>,
}

impl Default for CacheConfig {
    fn default() -> Self {
        Self {
            max_entries: 10000,
            memory_budget_mb: None,
        }
    }
}

#[derive(Clone, Debug, Eq, PartialEq, Hash)]
pub struct InputKey {
    pub network: String,
    pub indices: Vec<usize>,
}

impl InputKey {
    pub fn new(network: &str, indices: Vec<usize>) -> Self {
        Self {
            network: network.to_string(),
            indices,
        }
    }
}

#[derive(Clone)]
pub struct CachedOutput {
    pub values: Vec<f64>,
    pub memory_size: usize,
}

impl CachedOutput {
    pub fn new(values: Vec<f64>) -> Self {
        let memory_size = values.len() * std::mem::size_of::<f64>();
        Self { values, memory_size }
    }
}

pub struct CacheStats {
    pub hits: u64,
    pub misses: u64,
}

impl CacheStats {
    pub fn hit_rate(&self) -> f64 {
        let total = self.hits + self.misses;
        if total == 0 {
            0.0
        } else {
            self.hits as f64 / total as f64
        }
    }
}

pub struct NeuralCache {
    entries: LruCache<InputKey, CachedOutput>,
    config: CacheConfig,
    current_memory: usize,
    hits: u64,
    misses: u64,
}

impl NeuralCache {
    pub fn new(config: CacheConfig) -> Self {
        let cap = NonZeroUsize::new(config.max_entries).unwrap_or(NonZeroUsize::new(1).unwrap());
        Self {
            entries: LruCache::new(cap),
            config,
            current_memory: 0,
            hits: 0,
            misses: 0,
        }
    }

    pub fn get(&mut self, key: &InputKey) -> Option<&Vec<f64>> {
        if let Some(entry) = self.entries.get(key) {
            self.hits += 1;
            Some(&entry.values)
        } else {
            self.misses += 1;
            None
        }
    }

    pub fn insert(&mut self, key: InputKey, values: Vec<f64>) {
        let output = CachedOutput::new(values);
        let entry_size = output.memory_size;

        // Evict if memory budget exceeded
        if let Some(budget) = self.config.memory_budget_mb {
            let budget_bytes = budget * 1024 * 1024;
            while self.current_memory + entry_size > budget_bytes {
                if let Some((_, evicted)) = self.entries.pop_lru() {
                    self.current_memory = self.current_memory.saturating_sub(evicted.memory_size);
                } else {
                    break;
                }
            }
        }

        // Insert (LRU will evict if max_entries exceeded)
        if let Some(old) = self.entries.push(key, output) {
            self.current_memory = self.current_memory.saturating_sub(old.1.memory_size);
        }
        self.current_memory += entry_size;
    }

    pub fn clear(&mut self) {
        self.entries.clear();
        self.current_memory = 0;
    }

    pub fn len(&self) -> usize {
        self.entries.len()
    }

    pub fn memory_usage(&self) -> usize {
        self.current_memory
    }

    pub fn stats(&self) -> CacheStats {
        CacheStats {
            hits: self.hits,
            misses: self.misses,
        }
    }
}

/// Cache registry for multiple networks
pub struct NetworkCacheRegistry {
    caches: std::collections::HashMap<String, NeuralCache>,
    default_config: CacheConfig,
}

impl NetworkCacheRegistry {
    pub fn new(default_config: CacheConfig) -> Self {
        Self {
            caches: std::collections::HashMap::new(),
            default_config,
        }
    }

    pub fn get_or_create(&mut self, network: &str) -> &mut NeuralCache {
        self.caches.entry(network.to_string())
            .or_insert_with(|| NeuralCache::new(self.default_config.clone()))
    }

    pub fn clear_all(&mut self) {
        for cache in self.caches.values_mut() {
            cache.clear();
        }
    }

    pub fn total_stats(&self) -> CacheStats {
        let (hits, misses) = self.caches.values()
            .map(|c| (c.hits, c.misses))
            .fold((0, 0), |(h1, m1), (h2, m2)| (h1 + h2, m1 + m2));
        CacheStats { hits, misses }
    }
}
```

```rust
// crates/xlog-neural/src/predicate.rs (add/modify)

/// Network evaluation mode
#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum NetworkMode {
    /// Output is probability distribution over labels
    Probabilistic,
    /// Output is raw tensor for downstream computation
    Deterministic,
}

impl NetworkHandle {
    /// Evaluate network and interpret output based on mode
    pub fn evaluate(
        &self,
        py: Python<'_>,
        inputs: &[GpuBuffer],
    ) -> PyResult<NetworkOutput> {
        // Run forward pass
        let raw_output = self.forward(py, inputs)?;

        match self.mode {
            NetworkMode::Probabilistic => {
                // Apply softmax if needed and return probabilities
                let probs = if self.needs_softmax {
                    apply_softmax(py, &raw_output)?
                } else {
                    raw_output
                };

                Ok(NetworkOutput::Probabilities {
                    values: probs,
                    labels: self.labels.clone(),
                })
            }
            NetworkMode::Deterministic => {
                // Return raw tensor for use in tensor predicates
                Ok(NetworkOutput::Tensor(raw_output))
            }
        }
    }
}

pub enum NetworkOutput {
    /// Probability distribution for classification
    Probabilities {
        values: GpuBuffer,
        labels: Vec<String>,
    },
    /// Raw tensor for embeddings/encoders
    Tensor(GpuBuffer),
}

impl NetworkOutput {
    /// Convert to circuit leaf weights (probabilistic only)
    pub fn to_leaf_weights(&self) -> Option<Vec<(String, f64)>> {
        match self {
            NetworkOutput::Probabilities { values, labels } => {
                Some(labels.iter().cloned()
                    .zip(values.to_vec().into_iter())
                    .collect())
            }
            NetworkOutput::Tensor(_) => None,
        }
    }

    /// Get raw tensor (deterministic only)
    pub fn as_tensor(&self) -> Option<&GpuBuffer> {
        match self {
            NetworkOutput::Tensor(t) => Some(t),
            _ => None,
        }
    }
}
```

```python
"""
MNIST Multi-Digit Addition Training

Trains a digit classifier using only multi-digit addition supervision.

Usage:
    python train.py --epochs 10
    python train.py --epochs 3 --test  # Quick test mode

def load_program():
    with open("examples/deepproblog/mnist_multidigit/program.xlog") as f:

def load_mnist():
    transform = transforms.Compose([
        transforms.ToTensor(),
        transforms.Normalize((0.1307,), (0.3081,))
    ])
    train_data = datasets.MNIST('./data', train=True, download=True, transform=transform)
    test_data = datasets.MNIST('./data', train=False, transform=transform)

    train_images = torch.stack([train_data[i][0] for i in range(len(train_data))])
    train_labels = [train_data[i][1] for i in range(len(train_data))]

    test_images = torch.stack([test_data[i][0] for i in range(len(test_data))])
    test_labels = [test_data[i][1] for i in range(len(test_data))]

    return (train_images, train_labels), (test_images, test_labels)

def generate_multidigit_queries(labels, n_queries, max_digits=3):
    """Generate multi-digit addition queries."""
    queries = []
    n = len(labels)

    for _ in range(n_queries):
        # Random number of digits (1-3)
        len1 = random.randint(1, max_digits)
        len2 = random.randint(1, max_digits)

        # Random indices
        indices1 = [random.randint(0, n-1) for _ in range(len1)]
        indices2 = [random.randint(0, n-1) for _ in range(len2)]

        # Compute expected sum
        num1 = int("".join(str(labels[i]) for i in indices1))
        num2 = int("".join(str(labels[i]) for i in indices2))
        expected_sum = num1 + num2

        # Format query
        list1 = "[" + ", ".join(str(i) for i in indices1) + "]"
        list2 = "[" + ", ".join(str(i) for i in indices2) + "]"
        query = f"multi_addition({list1}, {list2}, {expected_sum})"
        queries.append(query)

    return queries

def evaluate_digit_accuracy(program, test_images, test_labels, n_samples=1000):
    """Evaluate single-digit classification accuracy."""
    program.eval()
    correct = 0

    indices = random.sample(range(len(test_labels)), min(n_samples, len(test_labels)))

    for i in indices:
        result = program.query(f"digit({i}, D)")
        if result and int(result.get("D", -1)) == test_labels[i]:
            correct += 1

    return correct / len(indices)

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--epochs", type=int, default=10)
    parser.add_argument("--batch-size", type=int, default=32)
    parser.add_argument("--lr", type=float, default=1e-3)
    parser.add_argument("--test", action="store_true", help="Quick test mode")
    args = parser.parse_args()

    # Load data
    print("Loading MNIST...")
    (train_images, train_labels), (test_images, test_labels) = load_mnist()

    # Setup program
    print("Compiling program...")
    program = load_program()

    # Create and register network
    device = "cuda" if torch.cuda.is_available() else "cpu"
    net = MNISTNet().to(device)
    optimizer = torch.optim.Adam(net.parameters(), lr=args.lr)
    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=5)

    program.register_network("mnist_net", net, optimizer, scheduler=scheduler)
    program.add_tensor_source("train", train_images.to(device))
    program.add_tensor_source("test", test_images.to(device))

    # Generate training queries
    n_queries = 1000 if args.test else 10000
    train_queries = generate_multidigit_queries(train_labels, n_queries)

    # Initial accuracy
    program.set_active_tensor_source("test")
    initial_acc = evaluate_digit_accuracy(program, test_images, test_labels)

    # Training

    history = pyxlog.train_model(

        print("Accuracy improved")  # Test marker
```

```python
def generate_coin_data(n_samples):

def generate_queries(labels1, labels2, n_queries):
    """Generate win/lose queries."""
    queries = []
    n = len(labels1)

    for _ in range(n_queries):
        i1, i2 = random.randint(0, n-1), random.randint(0, n-1)

        # Determine outcome
        if labels1[i1] == labels2[i2]:
```

```rust
// crates/pyxlog/src/metrics.rs
use pyo3::prelude::*;
use std::collections::HashMap;

/// Compute accuracy: correct predictions / total predictions
#[pyfunction]
pub fn accuracy(predictions: Vec<String>, targets: Vec<String>) -> f64 {
    if predictions.len() != targets.len() || predictions.is_empty() {
        return 0.0;
    }

    let correct: usize = predictions.iter()
        .zip(targets.iter())
        .filter(|(p, t)| p == t)
        .count();

    correct as f64 / predictions.len() as f64
}

/// Compute confusion matrix for multiclass classification
#[pyfunction]
pub fn confusion_matrix(
    predictions: Vec<usize>,
    targets: Vec<usize>,
    num_classes: usize,
) -> Vec<Vec<usize>> {
    let mut matrix = vec![vec![0usize; num_classes]; num_classes];

    for (pred, target) in predictions.iter().zip(targets.iter()) {
        if *pred < num_classes && *target < num_classes {
            matrix[*target][*pred] += 1;
        }
    }

    matrix
}

/// Compute precision, recall, and F1 score for binary classification
#[pyfunction]
pub fn precision_recall_f1(
    predictions: Vec<usize>,
    targets: Vec<usize>,
    positive_class: usize,
) -> (f64, f64, f64) {
    let mut tp = 0usize;
    let mut fp = 0usize;
    let mut fn_ = 0usize;

    for (pred, target) in predictions.iter().zip(targets.iter()) {
        if *pred == positive_class && *target == positive_class {
            tp += 1;
        } else if *pred == positive_class && *target != positive_class {
            fp += 1;
        } else if *pred != positive_class && *target == positive_class {
            fn_ += 1;
        }
    }

    let precision = if tp + fp > 0 {
        tp as f64 / (tp + fp) as f64
    } else {
        0.0
    };

    let recall = if tp + fn_ > 0 {
        tp as f64 / (tp + fn_) as f64
    } else {
        0.0
    };

    let f1 = if precision + recall > 0.0 {
        2.0 * precision * recall / (precision + recall)
    } else {
        0.0
    };

    (precision, recall, f1)
}

/// Per-class metrics for multiclass classification
#[pyfunction]
pub fn classification_report(
    predictions: Vec<usize>,
    targets: Vec<usize>,
    num_classes: usize,
    class_names: Option<Vec<String>>,
) -> HashMap<String, HashMap<String, f64>> {
    let cm = confusion_matrix(predictions.clone(), targets.clone(), num_classes);
    let mut report = HashMap::new();

    for class_idx in 0..num_classes {
        let class_name = class_names.as_ref()
            .map(|names| names.get(class_idx).cloned().unwrap_or(format!("{}", class_idx)))
            .unwrap_or(format!("{}", class_idx));

        // True positives
        let tp = cm[class_idx][class_idx] as f64;

        // False positives (column sum minus diagonal)
        let fp: f64 = (0..num_classes).map(|i| cm[i][class_idx] as f64).sum::<f64>() - tp;

        // False negatives (row sum minus diagonal)
        let fn_: f64 = cm[class_idx].iter().sum::<f64>() - tp;

        let precision = if tp + fp > 0.0 { tp / (tp + fp) } else { 0.0 };
        let recall = if tp + fn_ > 0.0 { tp / (tp + fn_) } else { 0.0 };
        let f1 = if precision + recall > 0.0 {
            2.0 * precision * recall / (precision + recall)
        } else {
            0.0
        };

        let mut metrics = HashMap::new();
        metrics.insert("precision".to_string(), precision);
        metrics.insert("recall".to_string(), recall);
        metrics.insert("f1".to_string(), f1);
        metrics.insert("support".to_string(), cm[class_idx].iter().sum::<usize>() as f64);

        report.insert(class_name, metrics);
    }

    // Macro averages
    let mut macro_metrics = HashMap::new();
    macro_metrics.insert(
        "precision".to_string(),
        report.values().map(|m| m["precision"]).sum::<f64>() / num_classes as f64
    );
    macro_metrics.insert(
        "recall".to_string(),
        report.values().map(|m| m["recall"]).sum::<f64>() / num_classes as f64
    );
    macro_metrics.insert(
        "f1".to_string(),
        report.values().map(|m| m["f1"]).sum::<f64>() / num_classes as f64
    );

    report.insert("macro_avg".to_string(), macro_metrics);

    report
}

pub fn register_metrics(py: Python<'_>, m: &PyModule) -> PyResult<()> {
    m.add_function(wrap_pyfunction!(accuracy, m)?)?;
    m.add_function(wrap_pyfunction!(confusion_matrix, m)?)?;
    m.add_function(wrap_pyfunction!(precision_recall_f1, m)?)?;
    m.add_function(wrap_pyfunction!(classification_report, m)?)?;
    Ok(())
}
```

```python
"""Training metrics for neural-symbolic models."""

from pyxlog._pyxlog import (
    accuracy as _accuracy,
    confusion_matrix as _confusion_matrix,
    precision_recall_f1 as _precision_recall_f1,
    classification_report as _classification_report,
)
import numpy as np
from typing import List, Optional, Dict, Tuple, Union

def accuracy(
    predictions: List[Union[str, int]],
    targets: List[Union[str, int]]
) -> float:
    """
    Compute classification accuracy.

    Args:
        predictions: Predicted labels
        targets: Ground truth labels

    Returns:
        Accuracy as float between 0 and 1
    """
    # Convert to strings for comparison
    preds = [str(p) for p in predictions]
    targs = [str(t) for t in targets]
    return _accuracy(preds, targs)

def confusion_matrix(
    predictions: List[int],
    targets: List[int],
    num_classes: int
) -> np.ndarray:
    """
    Compute confusion matrix.

    Args:
        predictions: Predicted class indices
        targets: Ground truth class indices
        num_classes: Number of classes

    Returns:
        Confusion matrix as numpy array [num_classes, num_classes]
        where cm[i, j] = count of true class i predicted as class j
    """
    cm = _confusion_matrix(predictions, targets, num_classes)
    return np.array(cm)

def precision_recall_f1(
    predictions: List[int],
    targets: List[int],
    positive_class: int = 1
) -> Tuple[float, float, float]:
    """
    Compute precision, recall, and F1 for binary classification.

    Args:
        predictions: Predicted class indices
        targets: Ground truth class indices
        positive_class: Which class is considered positive

    Returns:
        Tuple of (precision, recall, f1)
    """
    return _precision_recall_f1(predictions, targets, positive_class)

def classification_report(
    predictions: List[int],
    targets: List[int],
    num_classes: int,
    class_names: Optional[List[str]] = None
) -> Dict[str, Dict[str, float]]:
    """
    Generate per-class classification report.

    Args:
        predictions: Predicted class indices
        targets: Ground truth class indices
        num_classes: Number of classes
        class_names: Optional names for each class

    Returns:
        Dictionary with metrics for each class and macro averages
    """
    return _classification_report(predictions, targets, num_classes, class_names)
```
