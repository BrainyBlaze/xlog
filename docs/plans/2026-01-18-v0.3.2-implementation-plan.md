```rust
//! Global symbol interning for reversible string-to-ID mapping.

use std::collections::HashMap;
use std::sync::{OnceLock, RwLock};

static REGISTRY: OnceLock<RwLock<SymbolRegistry>> = OnceLock::new();

struct SymbolRegistry {
    to_id: HashMap<String, u32>,
    to_string: Vec<String>,
}

impl SymbolRegistry {
    fn new() -> Self {
        Self {
            to_id: HashMap::new(),
            to_string: Vec::new(),
        }
    }
}

fn registry() -> &'static RwLock<SymbolRegistry> {
    REGISTRY.get_or_init(|| RwLock::new(SymbolRegistry::new()))
}

/// Intern a string, returning its unique ID.
/// Thread-safe. Returns existing ID if already interned.
pub fn intern(s: &str) -> u32 {
    let mut reg = registry().write().unwrap();
    if let Some(&id) = reg.to_id.get(s) {
        return id;
    }
    let id = reg.to_string.len() as u32;
    reg.to_string.push(s.to_string());
    reg.to_id.insert(s.to_string(), id);
    id
}

/// Resolve an ID to its string. Panics if ID is invalid.
pub fn resolve(id: u32) -> String {
    let reg = registry().read().unwrap();
    reg.to_string
        .get(id as usize)
        .cloned()
        .expect("invalid symbol ID")
}

/// Clear all symbols. For testing/REPL only.
pub fn clear() {
    let mut reg = registry().write().unwrap();
    reg.to_id.clear();
    reg.to_string.clear();
}

/// Number of interned symbols.
pub fn count() -> usize {
    registry().read().unwrap().to_string.len()
}

#[cfg(test)]

    fn setup() {

    }

    #[test]

    }

    #[test]
```

```rust
//! Module resolution and loading for XLOG programs.

use crate::ast::{Program, UseDecl};
use std::collections::{HashMap, HashSet};
use std::path::{Path, PathBuf};

/// A module path like ["utils", "math"]
pub type ModulePath = Vec<String>;

/// A loaded module with its exports
#[derive(Debug)]
pub struct LoadedModule {
    pub path: ModulePath,
    pub source_file: PathBuf,
    pub program: Program,
    pub exports: HashSet<String>,
}

/// Module resolution errors
#[derive(Debug, Clone)]
pub enum ModuleError {
    NotFound { path: ModulePath, searched: Vec<PathBuf> },
    CircularImport { cycle: Vec<ModulePath> },
    ImportConflict { name: String, module1: ModulePath, module2: ModulePath },
    PrivatePredicate { name: String, module: ModulePath },
    ParseError { path: PathBuf, message: String },
}

impl std::fmt::Display for ModuleError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            ModuleError::NotFound { path, searched } => {
                write!(f, "module not found: {}\nsearched in: {:?}",
                       path.join("/"), searched)
            }
            ModuleError::CircularImport { cycle } => {
                write!(f, "circular import detected: {}",
                       cycle.iter().map(|p| p.join("/")).collect::<Vec<_>>().join(" -> "))
            }
            ModuleError::ImportConflict { name, module1, module2 } => {
                write!(f, "ambiguous import `{}`: exported by both {} and {}",
                       name, module1.join("/"), module2.join("/"))
            }
            ModuleError::PrivatePredicate { name, module } => {
                write!(f, "cannot import private predicate `{}` from {}",
                       name, module.join("/"))
            }
            ModuleError::ParseError { path, message } => {
                write!(f, "parse error in {:?}: {}", path, message)
            }
        }
    }
}

impl std::error::Error for ModuleError {}

/// Resolves and loads modules
pub struct ModuleResolver {
    search_paths: Vec<PathBuf>,
    loaded: HashMap<String, LoadedModule>,  // path string -> module
    loading: HashSet<String>,  // currently loading (for cycle detection)
}

impl ModuleResolver {
    pub fn new(search_paths: Vec<PathBuf>) -> Self {
        Self {
            search_paths,
            loaded: HashMap::new(),
            loading: HashSet::new(),
        }
    }

    /// Find the file for a module path, searching relative to base_dir first
    pub fn find_module(&self, base_dir: &Path, module_path: &[String]) -> Option<PathBuf> {
        let relative_path = format!("{}.xlog", module_path.join("/"));

        // Try relative to base_dir first
        let candidate = base_dir.join(&relative_path);
        if candidate.exists() {
            return Some(candidate);
        }

        // Try search paths
        for search_path in &self.search_paths {
            let candidate = search_path.join(&relative_path);
            if candidate.exists() {
                return Some(candidate);
            }
        }

        None
    }

    /// Get exports from a program (all non-private predicates)
    pub fn get_exports(program: &Program) -> HashSet<String> {
        let mut exports = HashSet::new();

        // Add declared predicates that aren't private
        for pred in &program.predicates {
            if !pred.is_private {
                exports.insert(pred.name.clone());
            }
        }

        // Add rule head predicates (unless private)
        for rule in &program.rules {
            exports.insert(rule.head.predicate.clone());
        }

        exports
    }
}
```

```rust
pub struct FuncParam {

/// Conditional expression for recursive functions
#[derive(Debug, Clone, PartialEq)]
pub struct CondExpr {
    pub left: ArithExpr,
    pub op: CompOp,
    pub right: ArithExpr,
    pub then_branch: Box<FuncBody>,
    pub else_branch: Box<FuncBody>,
}

/// Function body - either arithmetic/conditional or predicate-based
#[derive(Debug, Clone, PartialEq)]
pub enum FuncBody {
    Arithmetic(ArithExpr),
    Conditional(CondExpr),
    Predicate {
        result: String,
        body: Vec<BodyLiteral>,
```

```rust
fn parse_func_param(pair: Pair<Rule>) -> FuncParam {
    let mut inner = pair.into_inner();
    let name = inner.next().unwrap().as_str().to_string();
    let typ = inner.next().map(|p| parse_type_spec(p.into_inner().next().unwrap()));
    FuncParam { name, typ }
}

fn parse_func_body(pair: Pair<Rule>) -> FuncBody {
    let inner = pair.into_inner().next().unwrap();
    match inner.as_rule() {
        Rule::func_body_pred => {
            let mut parts = inner.into_inner();
            let result = parts.next().unwrap().as_str().to_string();
            let body = parse_body(parts.next().unwrap());
            FuncBody::Predicate { result, body }
        }
        Rule::func_body_arith => {
            let arith = inner.into_inner().next().unwrap();
            match arith.as_rule() {
                Rule::cond_expr => FuncBody::Conditional(parse_cond_expr(arith)),
                _ => FuncBody::Arithmetic(parse_arith_expr(arith)),
            }
        }
        _ => unreachable!(),
    }
}

fn parse_cond_expr(pair: Pair<Rule>) -> CondExpr {
    let mut inner = pair.into_inner();
    let test = inner.next().unwrap();
    let mut test_inner = test.into_inner();
    let left = parse_arith_expr(test_inner.next().unwrap());
    let op = parse_cmp_op(test_inner.next().unwrap());
    let right = parse_arith_expr(test_inner.next().unwrap());
    let then_branch = Box::new(parse_func_body(inner.next().unwrap()));
    let else_branch = Box::new(parse_func_body(inner.next().unwrap()));
    CondExpr { left, op, right, then_branch, else_branch }
}

fn parse_func_def(pair: Pair<Rule>) -> FuncDef {
    let mut inner = pair.into_inner();
    let mut is_private = false;

    let first = inner.next().unwrap();
    let name_pair = if first.as_rule() == Rule::private_mod {
        is_private = true;
        inner.next().unwrap()
    } else {
        first
    };

    let name = name_pair.as_str().to_string();

    let mut params = Vec::new();
    let mut return_type = None;
    let mut body = None;

    for pair in inner {
        match pair.as_rule() {
            Rule::func_params => {
                params = pair.into_inner().map(parse_func_param).collect();
            }
            Rule::return_type => {
                return_type = Some(parse_type_spec(pair.into_inner().next().unwrap()));
            }
            Rule::func_body => {
                body = Some(parse_func_body(pair));
            }
            _ => {}
        }
    }

    FuncDef {
        name,
        params,
        return_type,
        body: body.unwrap(),
        is_private,
    }
}
```

```rust
//! Function registry and inline expansion for user-defined functions.

use crate::ast::{FuncDef, FuncBody, ArithExpr, BodyLiteral};
use std::collections::{HashMap, HashSet};

/// Registry of user-defined functions
#[derive(Debug, Default)]
pub struct FunctionRegistry {
    functions: HashMap<String, FuncDef>,
    call_graph: HashMap<String, HashSet<String>>,
}

/// Function-related errors
#[derive(Debug, Clone)]
pub enum FunctionError {
    DuplicateDefinition { name: String },
    RecursionWithoutBaseCase { name: String },
    UndefinedFunction { name: String },
    MaxRecursionDepth { name: String, depth: u32 },
}

impl std::fmt::Display for FunctionError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            FunctionError::DuplicateDefinition { name } => {
                write!(f, "duplicate function definition: {}", name)
            }
            FunctionError::RecursionWithoutBaseCase { name } => {
                write!(f, "recursive function `{}` has no base case", name)
            }
            FunctionError::UndefinedFunction { name } => {
                write!(f, "undefined function: {}", name)
            }
            FunctionError::MaxRecursionDepth { name, depth } => {
                write!(f, "maximum recursion depth ({}) exceeded in function `{}`", depth, name)
            }
        }
    }
}

impl std::error::Error for FunctionError {}

impl FunctionRegistry {
    pub fn new() -> Self {
        Self::default()
    }

    /// Register a function
    pub fn register(&mut self, func: FuncDef) -> Result<(), FunctionError> {
        if self.functions.contains_key(&func.name) {
            return Err(FunctionError::DuplicateDefinition { name: func.name });
        }

        // Build call graph entry
        let calls = self.extract_calls(&func.body);
        self.call_graph.insert(func.name.clone(), calls);
        self.functions.insert(func.name.clone(), func);

        Ok(())
    }

    /// Check if a function is recursive
    pub fn is_recursive(&self, name: &str) -> bool {
        self.calls_transitively(name, name, &mut HashSet::new())
    }

    fn calls_transitively(&self, from: &str, target: &str, visited: &mut HashSet<String>) -> bool {
        if visited.contains(from) {
            return false;
        }
        visited.insert(from.to_string());

        if let Some(calls) = self.call_graph.get(from) {
            if calls.contains(target) {
                return true;
            }
            for call in calls {
                if self.calls_transitively(call, target, visited) {
                    return true;
                }
            }
        }
        false
    }

    /// Extract function calls from a function body
    fn extract_calls(&self, body: &FuncBody) -> HashSet<String> {
        let mut calls = HashSet::new();
        self.extract_calls_from_body(body, &mut calls);
        calls
    }

    fn extract_calls_from_body(&self, body: &FuncBody, calls: &mut HashSet<String>) {
        match body {
            FuncBody::Arithmetic(expr) => self.extract_calls_from_expr(expr, calls),
            FuncBody::Conditional(cond) => {
                self.extract_calls_from_expr(&cond.left, calls);
                self.extract_calls_from_expr(&cond.right, calls);
                self.extract_calls_from_body(&cond.then_branch, calls);
                self.extract_calls_from_body(&cond.else_branch, calls);
            }
            FuncBody::Predicate { body, .. } => {
                // Predicate bodies don't contain function calls in this model
            }
        }
    }

    fn extract_calls_from_expr(&self, expr: &ArithExpr, calls: &mut HashSet<String>) {
        match expr {
            ArithExpr::Add(l, r) | ArithExpr::Sub(l, r) | ArithExpr::Mul(l, r)
            | ArithExpr::Div(l, r) | ArithExpr::Mod(l, r)
            | ArithExpr::Min(l, r) | ArithExpr::Max(l, r) | ArithExpr::Pow(l, r) => {
                self.extract_calls_from_expr(l, calls);
                self.extract_calls_from_expr(r, calls);
            }
            ArithExpr::Abs(e) | ArithExpr::Cast(e, _) => {
                self.extract_calls_from_expr(e, calls);
            }
            ArithExpr::Variable(_) | ArithExpr::Integer(_) | ArithExpr::Float(_) => {}
            // Note: Need to add FuncCall variant to ArithExpr to fully support this
        }
    }

    /// Validate all registered functions
    pub fn validate(&self) -> Result<(), FunctionError> {
        for (name, func) in &self.functions {
            if self.is_recursive(name) {
                // Check for base case (must have conditional form)
                if !self.has_base_case(&func.body) {
                    return Err(FunctionError::RecursionWithoutBaseCase { name: name.clone() });
                }
            }
        }
        Ok(())
    }

    fn has_base_case(&self, body: &FuncBody) -> bool {
        matches!(body, FuncBody::Conditional(_))
    }

    /// Get a function by name
    pub fn get(&self, name: &str) -> Option<&FuncDef> {
        self.functions.get(name)
    }
}
```
