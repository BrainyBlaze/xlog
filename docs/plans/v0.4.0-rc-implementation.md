// crates/xlog-prob/src/transform/negation_removal.rs
use xlog_logic::{Program, Rule, Goal, Term, Atom};
use crate::stratify::{stratify, Stratum};
use thiserror::Error;

#[derive(Error, Debug)]
pub enum NegationRemovalError {
    #[error("Program is not stratified")]
    NotStratified,
    #[error("Cannot compute finite domain for {0}")]
    InfiniteDomain(String),
    #[error("Negated predicate {0} has no definition")]
    UndefinedNegation(String),
}

/// Remove negation from a stratified program by computing complements
pub fn remove_negation(program: &Program) -> Result<Program, NegationRemovalError> {
    // 1. Compute stratification
    let strata = stratify(program).map_err(|_| NegationRemovalError::NotStratified)?;

    // 2. Process each stratum bottom-up
    let mut transformed = program.clone();
    transformed.rules = Vec::new();

    for stratum in &strata {
        for rule in &stratum.rules {
            if has_negation(rule) {
                // Generate complement predicates and rewrite rule
                let (complement_rules, rewritten) = transform_negated_rule(rule, &transformed)?;
                transformed.rules.extend(complement_rules);
                transformed.rules.push(rewritten);
            } else {
                transformed.rules.push(rule.clone());
            }
        }

        // Materialize predicates in this stratum for use in higher strata
        materialize_stratum(&mut transformed, stratum)?;
    }

    Ok(transformed)
}

fn has_negation(rule: &Rule) -> bool {
    rule.body.iter().any(|g| matches!(g, Goal::Not(_)))
}

fn transform_negated_rule(
    rule: &Rule,
    program: &Program,
) -> Result<(Vec<Rule>, Rule), NegationRemovalError> {
    let mut complement_rules = Vec::new();
    let mut new_body = Vec::new();

    for goal in &rule.body {
        match goal {
            Goal::Not(inner) => {
                // Generate complement predicate
                let (pred, args) = extract_call(inner)?;
                let complement_name = format!("not_{}", pred);

                // Compute domain for variables in negated atom
                let domain = compute_domain(&args, program)?;

                // Generate complement rules
                let complement = generate_complement(&pred, &args, &domain, program)?;
                complement_rules.extend(complement);

                // Replace \+ p(X) with not_p(X)
                new_body.push(Goal::Call {
                    predicate: complement_name,
                    args: args.clone(),
                });
            }
            other => {
                new_body.push(other.clone());
            }
        }
    }

    let rewritten = Rule {
        head: rule.head.clone(),
        body: new_body,
    };

    Ok((complement_rules, rewritten))
}

fn generate_complement(
    pred: &str,
    args: &[Term],
    domain: &[Vec<Term>],
    program: &Program,
) -> Result<Vec<Rule>, NegationRemovalError> {
    let complement_name = format!("not_{}", pred);
    let in_pred_name = format!("in_{}", pred);

    // Generate: in_p(X) :- p(X).
    let in_rule = Rule {
        head: Goal::Call {
            predicate: in_pred_name.clone(),
            args: args.to_vec(),
        },
        body: vec![Goal::Call {
            predicate: pred.to_string(),
            args: args.to_vec(),
        }],
    };

    // Generate: not_p(X) :- domain(X), \+ in_p(X).
    // But since we're removing negation, we enumerate the complement:
    // not_p(d1). not_p(d2). ... for all d in domain \ p

    let mut complement_rules = vec![in_rule];

    // For small domains, enumerate complement explicitly
    // This is the "aggregate lifting" approach for negation
    let domain_size: usize = domain.iter().map(|d| d.len()).product();

    if domain_size <= 10000 {
        // Enumerate all domain elements not in p
        for combo in domain_combinations(domain) {
            // Check if this combo is in p
            let in_p = program.query_ground(&pred, &combo)?;

            if !in_p {
                complement_rules.push(Rule {
                    head: Goal::Call {
                        predicate: complement_name.clone(),
                        args: combo,
                    },
                    body: vec![],
                });
            }
        }
    } else {
        // For large domains, use domain predicate + negation marker
        // This requires the caller to handle specially
        return Err(NegationRemovalError::InfiniteDomain(pred.to_string()));
    }

    Ok(complement_rules)
}

fn compute_domain(args: &[Term], program: &Program) -> Result<Vec<Vec<Term>>, NegationRemovalError> {
    // Extract domain from program facts or declarations
    let mut domains = Vec::new();

    for arg in args {
        match arg {
            Term::Variable(v) => {
                // Look for domain predicates like: domain_X(a). domain_X(b). ...
                // Or infer from facts
                let inferred = infer_variable_domain(v, program)?;
                domains.push(inferred);
            }
            Term::Atom(_) | Term::Number(_) => {
                // Ground term - domain is just itself
                domains.push(vec![arg.clone()]);
            }
            _ => {
                return Err(NegationRemovalError::InfiniteDomain(format!("{:?}", arg)));
            }
        }
    }

    Ok(domains)
}

fn domain_combinations(domains: &[Vec<Term>]) -> Vec<Vec<Term>> {
    if domains.is_empty() {
        return vec![vec![]];
    }

    let mut result = Vec::new();
    let rest = domain_combinations(&domains[1..]);

    for item in &domains[0] {
        for suffix in &rest {
            let mut combo = vec![item.clone()];
            combo.extend(suffix.iter().cloned());
            result.push(combo);
        }
    }

    result
}

// crates/xlog-prob/src/transform/aggregate_lift.rs
use xlog_logic::{Program, Rule, Goal, Term, Expr};
use std::collections::HashSet;

#[derive(Debug)]
pub struct LiftedProgram {
    pub rules: Vec<Rule>,
    pub mc_required: HashSet<String>,
}

/// Lift aggregates by enumerating all possible combinations
pub fn lift_aggregates(program: &Program, max_domain: usize) -> Result<LiftedProgram, LiftError> {
    let mut lifted = LiftedProgram {
        rules: Vec::new(),
        mc_required: HashSet::new(),
    };

    for rule in &program.rules {
        if let Some(agg) = find_aggregate(rule) {
            let domain = compute_aggregate_domain(&agg, program)?;
            let domain_size = domain.len();

            if domain_size <= max_domain {
                // Enumerate all 2^n combinations
                let enumerated = enumerate_aggregate(rule, &agg, &domain)?;
                lifted.rules.extend(enumerated);
            } else {
                // Mark for MC fallback
                lifted.mc_required.insert(rule.head_predicate().to_string());
                lifted.rules.push(rule.clone());
            }
        } else {
            lifted.rules.push(rule.clone());
        }
    }

    Ok(lifted)
}

#[derive(Debug)]
struct AggregateInfo {
    kind: AggregateKind,
    var: Term,
    goal: Goal,
    result_var: Term,
}

#[derive(Debug, Clone, Copy)]
enum AggregateKind {
    Count,
    Sum,
    Min,
    Max,
}

fn find_aggregate(rule: &Rule) -> Option<AggregateInfo> {
    for goal in &rule.body {
        if let Goal::Unify { left, right } = goal {
            if let Expr::Aggregate { kind, var, goal } = right {
                return Some(AggregateInfo {
                    kind: *kind,
                    var: var.clone(),
                    goal: *goal.clone(),
                    result_var: left.clone(),
                });
            }
        }
    }
    None
}

fn enumerate_aggregate(
    rule: &Rule,
    agg: &AggregateInfo,
    domain: &[Term],
) -> Result<Vec<Rule>, LiftError> {
    let mut rules = Vec::new();
    let n = domain.len();

    // Enumerate all 2^n subsets
    for mask in 0..(1 << n) {
        let mut included = Vec::new();
        let mut excluded = Vec::new();

        for (i, item) in domain.iter().enumerate() {
            if (mask >> i) & 1 == 1 {
                included.push(item.clone());
            } else {
                excluded.push(item.clone());
            }
        }

        // Compute aggregate value for this subset
        let value = compute_aggregate_value(&agg.kind, &included)?;

        // Build rule body with explicit inclusion/exclusion
        let mut body = Vec::new();

        // Include positive goals for included items
        for item in &included {
            body.push(substitute_goal(&agg.goal, &agg.var, item));
        }

        // Include negated goals for excluded items
        for item in &excluded {
            body.push(Goal::Not(Box::new(substitute_goal(&agg.goal, &agg.var, item))));
        }

        // Add other body goals from original rule (excluding the aggregate)
        for goal in &rule.body {
            if !is_aggregate_goal(goal) {
                body.push(goal.clone());
            }
        }

        // Create rule: head :- body, Result = value
        body.push(Goal::Unify {
            left: agg.result_var.clone(),
            right: Expr::Number(value as f64),
        });

        rules.push(Rule {
            head: rule.head.clone(),
            body,
        });
    }

    Ok(rules)
}

fn compute_aggregate_value(kind: &AggregateKind, items: &[Term]) -> Result<i64, LiftError> {
    match kind {
        AggregateKind::Count => Ok(items.len() as i64),
        AggregateKind::Sum => {
            items.iter()
                .map(|t| match t {
                    Term::Number(n) => Ok(*n as i64),
                    _ => Err(LiftError::NonNumericSum),
                })
                .sum()
        }
        AggregateKind::Min => {
            items.iter()
                .map(|t| match t {
                    Term::Number(n) => Ok(*n as i64),
                    _ => Err(LiftError::NonNumericMinMax),
                })
                .min()
                .unwrap_or(Ok(i64::MAX))
        }
        AggregateKind::Max => {
            items.iter()
                .map(|t| match t {
                    Term::Number(n) => Ok(*n as i64),
                    _ => Err(LiftError::NonNumericMinMax),
                })
                .max()
                .unwrap_or(Ok(i64::MIN))
        }
    }
}

fn substitute_goal(goal: &Goal, var: &Term, value: &Term) -> Goal {
    // Replace occurrences of var with value in goal
    goal.substitute(var, value)
}

// crates/xlog-prob/src/compile/compiler.rs
use crate::ddnnf::Ddnnf;
use xlog_logic::Cnf;
use thiserror::Error;

#[derive(Error, Debug)]
pub enum CompileError {
    #[error("Compiler binary not found: {0}")]
    BinaryNotFound(String),
    #[error("Compilation failed: {0}")]
    CompilationFailed(String),
    #[error("Invalid output: {0}")]
    InvalidOutput(String),
    #[error("Timeout after {0}ms")]
    Timeout(u64),
}

pub trait KnowledgeCompiler: Send + Sync {
    fn compile(&self, cnf: &Cnf) -> Result<Ddnnf, CompileError>;
    fn name(&self) -> &str;
}

pub struct CompilerRegistry {
    compilers: std::collections::HashMap<String, Box<dyn KnowledgeCompiler>>,
    default: String,
}

impl CompilerRegistry {
    pub fn new() -> Self {
        let mut registry = Self {
            compilers: std::collections::HashMap::new(),
            default: "d4".to_string(),
        };

        // Register available compilers
        if let Ok(d4) = crate::compile::D4Compiler::new() {
            registry.compilers.insert("d4".to_string(), Box::new(d4));
        }
        if let Ok(c2d) = crate::compile::C2DCompiler::new() {
            registry.compilers.insert("c2d".to_string(), Box::new(c2d));
        }
        if let Ok(minic2d) = crate::compile::MiniC2DCompiler::new() {
            registry.compilers.insert("minic2d".to_string(), Box::new(minic2d));
        }

        registry
    }

    pub fn get(&self, name: &str) -> Option<&dyn KnowledgeCompiler> {
        self.compilers.get(name).map(|c| c.as_ref())
    }

    pub fn default(&self) -> &dyn KnowledgeCompiler {
        self.compilers.get(&self.default).unwrap().as_ref()
    }

    pub fn set_default(&mut self, name: &str) -> Result<(), CompileError> {
        if self.compilers.contains_key(name) {
            self.default = name.to_string();
            Ok(())
        } else {
            Err(CompileError::BinaryNotFound(name.to_string()))
        }
    }
}

// crates/xlog-prob/src/compile/c2d.rs
use super::compiler::{KnowledgeCompiler, CompileError};
use crate::ddnnf::Ddnnf;
use xlog_logic::Cnf;
use std::path::PathBuf;
use std::process::Command;
use std::io::Write;
use tempfile::NamedTempFile;

pub struct C2DCompiler {
    binary_path: PathBuf,
    timeout_ms: u64,
}

impl C2DCompiler {
    pub fn new() -> Result<Self, CompileError> {
        let binary_path = find_c2d_binary()?;
        Ok(Self {
            binary_path,
            timeout_ms: 60000, // 1 minute default
        })
    }

    pub fn with_timeout(mut self, ms: u64) -> Self {
        self.timeout_ms = ms;
        self
    }
}

impl KnowledgeCompiler for C2DCompiler {
    fn compile(&self, cnf: &Cnf) -> Result<Ddnnf, CompileError> {
        // Write CNF to temp file in DIMACS format
        let mut cnf_file = NamedTempFile::new()
            .map_err(|e| CompileError::CompilationFailed(e.to_string()))?;

        write_dimacs(cnf_file.as_file_mut(), cnf)?;

        // Create output file path
        let nnf_path = cnf_file.path().with_extension("nnf");

        // Run c2d
        let output = Command::new(&self.binary_path)
            .arg("-in")
            .arg(cnf_file.path())
            .arg("-out")
            .arg(&nnf_path)
            .arg("-smooth")
            .output()
            .map_err(|e| CompileError::CompilationFailed(e.to_string()))?;

        if !output.status.success() {
            let stderr = String::from_utf8_lossy(&output.stderr);
            return Err(CompileError::CompilationFailed(stderr.to_string()));
        }

        // Parse NNF output
        let nnf_content = std::fs::read_to_string(&nnf_path)
            .map_err(|e| CompileError::InvalidOutput(e.to_string()))?;

        parse_nnf(&nnf_content)
    }

    fn name(&self) -> &str {
        "c2d"
    }
}

fn find_c2d_binary() -> Result<PathBuf, CompileError> {
    // Check vendored location first
    let vendored = PathBuf::from(env!("CARGO_MANIFEST_DIR"))
        .join("vendor")
        .join("c2d")
        .join(platform_dir())
        .join("c2d");

    if vendored.exists() {
        return Ok(vendored);
    }

    // Check PATH

fn write_dimacs(file: &mut std::fs::File, cnf: &Cnf) -> Result<(), CompileError> {

    writeln!(file, "p cnf {} {}", cnf.num_vars, cnf.clauses.len())

    for clause in &cnf.clauses {

            .collect::<Vec<_>>()
            .join(" ");
        writeln!(file, "{} 0", line)
            .map_err(|e| CompileError::CompilationFailed(e.to_string()))?;
    }

    Ok(())
}

fn parse_nnf(content: &str) -> Result<Ddnnf, CompileError> {
    // Parse c2d NNF format into DDNNF structure
    // Format: nnf <nodes> <edges> <vars>
    // Followed by node definitions

    let mut lines = content.lines();

    let header = lines.next()
        .ok_or_else(|| CompileError::InvalidOutput("Empty NNF file".to_string()))?;

    let parts: Vec<&str> = header.split_whitespace().collect();
    if parts.len() < 4 || parts[0] != "nnf" {
        return Err(CompileError::InvalidOutput("Invalid NNF header".to_string()));
    }

    let num_nodes: usize = parts[1].parse()
        .map_err(|_| CompileError::InvalidOutput("Invalid node count".to_string()))?;

    let mut nodes = Vec::with_capacity(num_nodes);

    for line in lines {
        let node = parse_nnf_node(line)?;
        nodes.push(node);
    }

    Ok(Ddnnf::from_nodes(nodes))
}

// crates/xlog-prob/src/compile/minic2d.rs
use super::{KnowledgeCompiler, CompilerError};
use crate::cnf::Cnf;
use crate::ddnnf::Ddnnf;
use std::io::{BufRead, BufReader, Write};
use std::path::PathBuf;
use std::process::{Command, Stdio};
use tempfile::NamedTempFile;

/// miniC2D knowledge compiler wrapper
///
/// miniC2D is a lightweight knowledge compiler optimized for small formulas.
/// It produces smaller circuits than c2d/D4 but may be slower on large inputs.
pub struct MiniC2DCompiler {
    binary_path: PathBuf,
}

impl MiniC2DCompiler {
    /// Create a new miniC2D compiler using the vendored binary
    pub fn new() -> Result<Self, CompilerError> {
        let binary_path = Self::find_binary()?;
        Self::validate_binary(&binary_path)?;
        Ok(Self { binary_path })
    }

    /// Find the miniC2D binary for the current platform
    fn find_binary() -> Result<PathBuf, CompilerError> {
        let vendor_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"))
            .join("vendor")
            .join("minic2d");

        #[cfg(target_os = "linux")]
        let binary = vendor_dir.join("linux-x86_64").join("miniC2D");

        #[cfg(target_os = "macos")]
        let binary = vendor_dir.join("darwin-arm64").join("miniC2D");

        #[cfg(target_os = "windows")]
        let binary = vendor_dir.join("windows-x86_64").join("miniC2D.exe");

        if binary.exists() {
            Ok(binary)
        } else {
            // Try PATH
            which::which("miniC2D")
                .map_err(|_| CompilerError::BinaryNotFound("miniC2D".to_string()))
        }
    }

    fn validate_binary(path: &PathBuf) -> Result<(), CompilerError> {
        let output = Command::new(path)
            .arg("--version")
            .output()
            .map_err(|e| CompilerError::ExecutionFailed(e.to_string()))?;

        if !output.status.success() {
            return Err(CompilerError::ExecutionFailed(
                "miniC2D version check failed".to_string()
            ));
        }

        Ok(())
    }

    /// Parse miniC2D NNF output format
    fn parse_nnf_output(&self, output: &str) -> Result<Ddnnf, CompilerError> {
        let mut nodes = Vec::new();
        let mut lines = output.lines();

        // Parse header: nnf <nodes> <edges> <vars>
        let header = lines.next()
            .ok_or_else(|| CompilerError::ParseError("Missing NNF header".to_string()))?;

        let parts: Vec<&str> = header.split_whitespace().collect();
        if parts.len() < 4 || parts[0] != "nnf" {
            return Err(CompilerError::ParseError(format!("Invalid header: {}", header)));
        }

        let n_nodes: usize = parts[1].parse()
            .map_err(|_| CompilerError::ParseError("Invalid node count".to_string()))?;

        // Parse nodes
        for line in lines {
            let line = line.trim();
            if line.is_empty() {
                continue;
            }

            let node = self.parse_node_line(line)?;
            nodes.push(node);
        }

        if nodes.len() != n_nodes {
            return Err(CompilerError::ParseError(format!(
                "Expected {} nodes, got {}", n_nodes, nodes.len()
            )));
        }

        Ok(Ddnnf::from_nodes(nodes))
    }

    fn parse_node_line(&self, line: &str) -> Result<crate::ddnnf::DdnnfNode, CompilerError> {
        use crate::ddnnf::DdnnfNode;

        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.is_empty() {
            return Err(CompilerError::ParseError("Empty node line".to_string()));
        }

        match parts[0] {
            "L" => {
                // Literal: L <var> or L -<var>
                let lit: i32 = parts.get(1)
                    .ok_or_else(|| CompilerError::ParseError("Missing literal".to_string()))?
                    .parse()
                    .map_err(|_| CompilerError::ParseError("Invalid literal".to_string()))?;
                Ok(DdnnfNode::Literal(lit))
            }
            "A" => {
                // And node: A <count> <child1> <child2> ...
                let count: usize = parts.get(1)
                    .ok_or_else(|| CompilerError::ParseError("Missing child count".to_string()))?
                    .parse()
                    .map_err(|_| CompilerError::ParseError("Invalid count".to_string()))?;

                let children: Vec<usize> = parts[2..2+count].iter()
                    .map(|s| s.parse().map_err(|_|
                        CompilerError::ParseError("Invalid child index".to_string())))
                    .collect::<Result<_, _>>()?;

                Ok(DdnnfNode::And(children))
            }
            "O" => {
                // Or node: O <var> <count> <child1> <child2> ...
                let _var: i32 = parts.get(1)
                    .ok_or_else(|| CompilerError::ParseError("Missing var".to_string()))?
                    .parse()
                    .map_err(|_| CompilerError::ParseError("Invalid var".to_string()))?;

                let count: usize = parts.get(2)
                    .ok_or_else(|| CompilerError::ParseError("Missing child count".to_string()))?
                    .parse()
                    .map_err(|_| CompilerError::ParseError("Invalid count".to_string()))?;

                let children: Vec<usize> = parts[3..3+count].iter()
                    .map(|s| s.parse().map_err(|_|
                        CompilerError::ParseError("Invalid child index".to_string())))
                    .collect::<Result<_, _>>()?;

                Ok(DdnnfNode::Or(children))
            }
            "T" => Ok(DdnnfNode::True),
            "F" => Ok(DdnnfNode::False),
            _ => Err(CompilerError::ParseError(format!("Unknown node type: {}", parts[0]))),
        }
    }
}

impl KnowledgeCompiler for MiniC2DCompiler {
    fn compile(&self, cnf: &Cnf) -> Result<Ddnnf, CompilerError> {
        // Write CNF to temp file
        let mut cnf_file = NamedTempFile::new()
            .map_err(|e| CompilerError::IoError(e.to_string()))?;

        cnf.write_dimacs(&mut cnf_file)
            .map_err(|e| CompilerError::IoError(e.to_string()))?;

        cnf_file.flush()
            .map_err(|e| CompilerError::IoError(e.to_string()))?;

        // Create output file
        let nnf_file = NamedTempFile::new()
            .map_err(|e| CompilerError::IoError(e.to_string()))?;

        // Run miniC2D
        let output = Command::new(&self.binary_path)
            .arg("-c")
            .arg(cnf_file.path())
            .arg("-o")
            .arg(nnf_file.path())
            .output()
            .map_err(|e| CompilerError::ExecutionFailed(e.to_string()))?;

        if !output.status.success() {
            return Err(CompilerError::ExecutionFailed(
                String::from_utf8_lossy(&output.stderr).to_string()
            ));
        }

        // Read and parse output
        let nnf_content = std::fs::read_to_string(nnf_file.path())
            .map_err(|e| CompilerError::IoError(e.to_string()))?;

        self.parse_nnf_output(&nnf_content)
    }

    fn name(&self) -> &str {
        "minic2d"
    }
}

// crates/xlog-prob/src/mc/importance.rs
use super::{McResult, McSampler};
use xlog_logic::Program;

#[derive(Debug, Clone, Copy)]
pub enum ImportanceStrategy {
    /// Bias toward evidence-satisfying worlds
    EvidenceGuided,
    /// Bias toward query-satisfying worlds
    QueryGuided,
    /// Learn proposal from initial samples
    Adaptive { warmup_samples: usize },
    /// Standard uniform sampling (baseline)
    Uniform,
}

pub struct ImportanceSampler {
    strategy: ImportanceStrategy,
    proposal: Option<ProposalDistribution>,
}

struct ProposalDistribution {
    weights: Vec<f64>, // Per-fact proposal weights
}

impl ImportanceSampler {
    pub fn new(strategy: ImportanceStrategy) -> Self {
        Self {
            strategy,
            proposal: None,
        }
    }

    pub fn sample(&self, program: &Program, n_samples: usize) -> ImportanceResult {
        let proposal = self.build_proposal(program);

        let mut samples = Vec::with_capacity(n_samples);
        let mut log_weights = Vec::with_capacity(n_samples);

        for _ in 0..n_samples {
            // Sample from proposal distribution
            let (world, log_proposal) = proposal.sample();

            // Compute true probability of world
            let log_true = program.log_probability(&world);

            // Importance weight = P(world) / Q(world)
            let log_weight = log_true - log_proposal;

            // Check if query is satisfied
            let query_satisfied = program.evaluate_query_in_world(&world);

            samples.push(query_satisfied as u8 as f64);
            log_weights.push(log_weight);
        }

        self.compute_result(&samples, &log_weights)
    }

    fn build_proposal(&self, program: &Program) -> ProposalDistribution {
        match self.strategy {
            ImportanceStrategy::Uniform => {
                // Use original probabilities
                ProposalDistribution {
                    weights: program.fact_probabilities().to_vec(),
                }
            }
            ImportanceStrategy::QueryGuided => {
                // Bias toward facts that appear in query derivation
                let query_facts = program.facts_relevant_to_query();
                let mut weights = program.fact_probabilities().to_vec();

                for (i, w) in weights.iter_mut().enumerate() {
                    if query_facts.contains(&i) {
                        // Increase probability of relevant facts
                        *w = (*w * 2.0).min(0.99);
                    }
                }

                ProposalDistribution { weights }
            }
            ImportanceStrategy::EvidenceGuided => {
                // Bias toward facts consistent with evidence
                let evidence_facts = program.facts_required_by_evidence();
                let mut weights = program.fact_probabilities().to_vec();

                for i in evidence_facts {
                    weights[i] = 0.99; // Force evidence-required facts
                }

                ProposalDistribution { weights }
            }
            ImportanceStrategy::Adaptive { warmup_samples } => {
                // Run warmup with uniform, then learn proposal
                let uniform = ProposalDistribution {
                    weights: program.fact_probabilities().to_vec(),
                };

                let mut fact_counts = vec![0usize; program.num_facts()];
                let mut success_counts = vec![0usize; program.num_facts()];

                for _ in 0..warmup_samples {
                    let (world, _) = uniform.sample();
                    let query_satisfied = program.evaluate_query_in_world(&world);

                    for (i, &active) in world.iter().enumerate() {
                        if active {
                            fact_counts[i] += 1;
                            if query_satisfied {
                                success_counts[i] += 1;
                            }
                        }
                    }
                }

                // Compute proposal weights based on success rate
                let weights: Vec<f64> = fact_counts.iter()
                    .zip(success_counts.iter())
                    .map(|(&total, &success)| {
                        if total == 0 {
                            0.5
                        } else {
                            (success as f64 / total as f64).clamp(0.1, 0.9)
                        }
                    })
                    .collect();

                ProposalDistribution { weights }
            }
        }
    }

    fn compute_result(&self, samples: &[f64], log_weights: &[f64]) -> ImportanceResult {
        // Convert log weights to normalized weights
        let max_log = log_weights.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
        let weights: Vec<f64> = log_weights.iter()
            .map(|&lw| (lw - max_log).exp())
            .collect();

        let weight_sum: f64 = weights.iter().sum();
        let normalized: Vec<f64> = weights.iter().map(|w| w / weight_sum).collect();

        // Weighted mean
        let mean: f64 = samples.iter()
            .zip(normalized.iter())
            .map(|(s, w)| s * w)
            .sum();

        // Weighted variance
        let variance: f64 = samples.iter()
            .zip(normalized.iter())
            .map(|(s, w)| w * (s - mean).powi(2))
            .sum();

        let ess = self.effective_sample_size(&weights);

        ImportanceResult {
            mean,
            variance,
            effective_sample_size: ess,
            n_samples: samples.len(),
        }
    }

    pub fn effective_sample_size(&self, weights: &[f64]) -> f64 {
        let sum: f64 = weights.iter().sum();
        let sum_sq: f64 = weights.iter().map(|w| w * w).sum();

        if sum_sq == 0.0 {
            0.0
        } else {
            (sum * sum) / sum_sq
        }
    }
}

#[derive(Debug)]
pub struct ImportanceResult {
    pub mean: f64,
    pub variance: f64,
    pub effective_sample_size: f64,
    pub n_samples: usize,
}

// kernels/mc_importance.cu
extern "C" __global__ void importance_sample_kernel(
    const float* fact_probs,      // Original probabilities [n_facts]
    const float* proposal_probs,  // Proposal probabilities [n_facts]
    float* samples,               // Output: sampled worlds [n_samples, n_facts]
    float* log_weights,           // Output: importance weights [n_samples]
    int n_samples,
    int n_facts,
    unsigned long long seed
) {
    int sample_idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (sample_idx >= n_samples) return;

    // Initialize RNG
    curandState state;
    curand_init(seed, sample_idx, 0, &state);

    float log_true = 0.0f;
    float log_proposal = 0.0f;

    for (int i = 0; i < n_facts; i++) {
        float u = curand_uniform(&state);
        float q = proposal_probs[i];
        float p = fact_probs[i];

        bool sampled = u < q;
        samples[sample_idx * n_facts + i] = sampled ? 1.0f : 0.0f;

        // Accumulate log probabilities
        if (sampled) {
            log_true += logf(p);
            log_proposal += logf(q);
        } else {
            log_true += logf(1.0f - p);
            log_proposal += logf(1.0f - q);
        }
    }

    log_weights[sample_idx] = log_true - log_proposal;
}

pub trait ApproximateEngine: Send + Sync {
    fn evaluate(&self, program: &Program, query: &str) -> Result<ApproxResult, ApproxError>;
    fn evaluate_with_grad(&self, program: &Program, query: &str) -> Result<ApproxResultWithGrad, ApproxError>;

#[derive(Debug)]
pub struct ApproxResult {
    pub probability: f64,
    pub proofs_found: usize,
    pub proofs_explored: usize,
}

#[derive(Debug)]
pub struct ApproxResultWithGrad {
    pub probability: f64,
    pub proofs: Vec<Proof>,
    grad_cache: GradientCache,
}

impl ApproxResultWithGrad {
    pub fn backward(&self, grad_output: f64) -> Result<HashMap<String, Vec<f64>>, ApproxError> {
        // REINFORCE gradient: d/dθ E[f(x)] ≈ E[f(x) * d/dθ log p(x|θ)]
        // For geometric mean: use log-derivative trick

        let mut gradients = HashMap::new();

        for proof in &self.proofs {
            let proof_weight = proof.probability / self.probability;
            let grad_scale = grad_output * proof_weight;

            for (network, leaf_grads) in &proof.leaf_gradients {
                let entry = gradients.entry(network.clone()).or_insert_with(Vec::new);
                for (i, g) in leaf_grads.iter().enumerate() {
                    if entry.len() <= i {
                        entry.resize(i + 1, 0.0);
                    }
                    entry[i] += grad_scale * g;
                }
            }
        }

        Ok(gradients)
    }
}

// crates/xlog-prob/src/approximate/geometric_mean.rs
use super::{ApproximateEngine, ApproxResult, ApproxResultWithGrad, ApproxError};
use xlog_logic::Program;

pub struct GeometricMeanEngine {
    k: usize,           // Top-k proofs
    exploration: f64,   // Exploration probability (epsilon-greedy)
    max_depth: usize,   // Maximum proof depth
}

impl GeometricMeanEngine {
    pub fn new(k: usize, exploration: f64) -> Self {
        Self {
            k,
            exploration,
            max_depth: 100,
        }
    }
}

impl ApproximateEngine for GeometricMeanEngine {
    fn evaluate(&self, program: &Program, query: &str) -> Result<ApproxResult, ApproxError> {
        let mut proofs = Vec::new();
        let mut frontier = vec![PartialProof::start(query)];
        let mut explored = 0;

        while !frontier.is_empty() && proofs.len() < self.k {
            // Sort by probability (descending)
            frontier.sort_by(|a, b| b.log_prob.partial_cmp(&a.log_prob).unwrap());

            // Take best or explore
            let next = if rand::random::<f64>() < self.exploration && frontier.len() > 1 {
                // Exploration: random choice
                let idx = rand::random::<usize>() % frontier.len();
                frontier.remove(idx)
            } else {
                // Exploitation: best
                frontier.remove(0)
            };

            explored += 1;

            if next.is_complete() {
                proofs.push(next.to_proof());
                continue;
            }

            if next.depth >= self.max_depth {
                continue;
            }

            // Expand: try all applicable rules
            for expansion in program.expand(&next)? {
                frontier.push(expansion);
            }
        }

        // Compute geometric mean of proof probabilities
        let probability = if proofs.is_empty() {
            0.0
        } else {
            let log_sum: f64 = proofs.iter().map(|p| p.log_probability).sum();
            (log_sum / proofs.len() as f64).exp()
        };

        Ok(ApproxResult {
            probability,
            proofs_found: proofs.len(),
            proofs_explored: explored,
        })
    }

    fn evaluate_with_grad(&self, program: &Program, query: &str) -> Result<ApproxResultWithGrad, ApproxError> {
        // Same as evaluate but track gradient information
        let mut proofs = Vec::new();
        let mut frontier = vec![PartialProof::start_with_grad(query)];

        while !frontier.is_empty() && proofs.len() < self.k {
            frontier.sort_by(|a, b| b.log_prob.partial_cmp(&a.log_prob).unwrap());
            let next = frontier.remove(0);

            if next.is_complete() {
                proofs.push(next.to_proof_with_grad());
                continue;
            }

            for expansion in program.expand_with_grad(&next)? {
                frontier.push(expansion);
            }
        }

        let probability = if proofs.is_empty() {
            0.0
        } else {
            let log_sum: f64 = proofs.iter().map(|p| p.log_probability).sum();
            (log_sum / proofs.len() as f64).exp()
        };

        Ok(ApproxResultWithGrad {
            probability,
            proofs,
            grad_cache: GradientCache::new(),
        })
    }
}

    classify_type(Encoding, ProblemType),
    extract_numbers(Encoding, Numbers),
    apply_operation(ProblemType, Numbers, Answer).

% Operation application

apply_operation(subtraction, [A, B], C) :- C is A - B.
apply_operation(multiplication, [A, B], C) :- C is A * B.

% Multi-step problems (permutation tables)
% The numbers can appear in any order in the text
solve_multi_step(Words, Answer) :-
    encode_text(Words, Encoding),
    extract_all_numbers(Encoding, AllNums),
    extract_operations(Encoding, Ops),
    permutation_solve(AllNums, Ops, Answer).

% Try different number orderings
permutation_solve([A, B], [Op], Result) :-
    apply_operation(Op, [A, B], Result).
permutation_solve([A, B], [Op], Result) :-
    apply_operation(Op, [B, A], Result).

% For three numbers, two operations
permutation_solve([A, B, C], [Op1, Op2], Result) :-
    apply_operation(Op1, [A, B], Intermediate),
    apply_operation(Op2, [Intermediate, C], Result).
% ... other permutations

% Number list extraction (using attention)
nn(attention, [E, Position], Weight) :: attention_weight(E, Position, Weight).

extract_all_numbers(Encoding, Numbers) :-
    findall(N, (
        member(Pos, [0, 1, 2, 3, 4]),
        attention_weight(Encoding, Pos, W),
        W > 0.5,

    elif op == 'subtraction':
        answer = a - b
    else:

# examples/deepproblog/forth_wap/model.py
import torch
import torch.nn as nn

class SharedRNN(nn.Module):
    """Shared RNN encoder for word sequences."""

    def __init__(self, embed_dim=128, hidden_dim=256):
        super().__init__()
        self.rnn = nn.LSTM(embed_dim, hidden_dim, batch_first=True, bidirectional=True)
        self.fc = nn.Linear(hidden_dim * 2, hidden_dim)

    def forward(self, x):
        # x: [batch, seq_len, embed_dim]
        _, (h, _) = self.rnn(x)
        h = torch.cat([h[0], h[1]], dim=-1)
        return self.fc(h)

class TypeClassifier(nn.Module):
    """Classify problem type from encoding."""

    def __init__(self, hidden_dim=256):
        super().__init__()
        self.fc1 = nn.Linear(hidden_dim, 64)
        self.fc2 = nn.Linear(64, 3)

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        return torch.softmax(self.fc2(x), dim=-1)

class NumberExtractor(nn.Module):
    """Extract numbers from encoding."""

    def __init__(self, hidden_dim=256, max_numbers=5):
        super().__init__()
        self.fc1 = nn.Linear(hidden_dim, 128)
        self.fc2 = nn.Linear(128, max_numbers * 21)  # 0-20 for each position
        self.max_numbers = max_numbers

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = self.fc2(x)
        return x.view(-1, self.max_numbers, 21)  # [batch, max_numbers, 21]
