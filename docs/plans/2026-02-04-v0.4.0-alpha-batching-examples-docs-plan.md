**Step 3: Implement template mapping in Rust**

Add mapping structs near `CompiledProgram` in `crates/pyxlog/src/lib.rs`:

```rust
#[derive(Debug, Clone)]
struct TemplateInputMapping {
    head_arg_index: usize,
    predicate: String,
    input_position: usize,
}

#[derive(Debug, Clone)]
struct TemplateMapping {
    pred_name: String,
    inputs: Vec<TemplateInputMapping>,
}
```

Build mappings during `Program::compile` (after `neural_registry` is created):

```rust
fn build_template_mappings(
    ast: &xlog_logic::ast::Program,
    registry: &NeuralPredicateRegistry,
) -> Result<StdHashMap<String, TemplateMapping>, String> {
    let mut mappings = StdHashMap::new();
    for rule in ast.rules.iter() {
        let head_pred = rule.head.predicate.clone();
        let mut inputs = Vec::new();
        for lit in rule.body.iter() {
            let atom = match lit.atom() {
                Some(a) => a,
                None => continue,
            };
            let info = match registry.get(&atom.predicate) {
                Some(i) => i,
                None => continue,
            };
            for &input_pos in &info.input_positions {
                let term = atom.terms.get(input_pos).ok_or_else(|| {
                    format!("Invalid input position {} for predicate {}", input_pos, atom.predicate)
                })?;
                let var = term.variable_name().ok_or_else(|| {
                    format!("Neural input must be a variable in predicate {}", atom.predicate)
                })?;
                let head_idx = rule
                    .head
                    .terms
                    .iter()
                    .position(|t| t.variable_name() == Some(var))
                    .ok_or_else(|| {
                        format!("Neural input var {} not in rule head for {}", var, head_pred)
                    })?;
                inputs.push(TemplateInputMapping {
                    head_arg_index: head_idx,
                    predicate: atom.predicate.clone(),
                    input_position: input_pos,
                });
            }
        }
        if !inputs.is_empty() {
            inputs.sort_by_key(|i| i.head_arg_index);
            mappings.insert(
                head_pred.clone(),
                TemplateMapping {
                    pred_name: head_pred,
                    inputs,
                },
            );
        }
    }
    Ok(mappings)
}
```

Add helper `parse_query_args` (strings only) to return argument list without heuristic popping:

```rust
fn parse_query_args(&self, query: &str) -> PyResult<(Vec<String>, String)> {
    let query = query.trim();
    let paren_start = query.find('(').ok_or_else(|| PyValueError::new_err("Invalid query"))?;
    let paren_end = query.rfind(')').ok_or_else(|| PyValueError::new_err("Invalid query"))?;
    let pred = query[..paren_start].to_string();
    let args = query[paren_start + 1..paren_end]
        .split(',')
        .map(|s| s.trim().to_string())
        .collect::<Vec<_>>();
    Ok((args, pred))
}
```

In `forward_backward_complex_tensor`, replace the per-input loop with per-network batching:

```rust
// Build ordered calls from template mapping (order = head_arg_index)
let mut calls: Vec<(usize, String, String)> = Vec::new();
for input in mapping.inputs.iter() {
    let input_idx = /* parsed from query args */;
    let info = self.neural_registry.get(&input.predicate).unwrap();
    calls.push((input_idx, input.predicate.clone(), info.network.clone()));
}

// Group by network name, preserve call order indices
let mut grouped: StdHashMap<String, Vec<(usize, usize)>> = StdHashMap::new();
for (call_idx, (input_idx, _pred, net)) in calls.iter().enumerate() {
    grouped.entry(net.clone()).or_default().push((call_idx, *input_idx));
}
```

For each network group, stack inputs and forward once:

```rust
let torch = py.import_bound("torch")?;
let mut out_tensors: Vec<PyObject> = vec![py.None(); calls.len()];
let mut grad_tensors: Vec<PyObject> = vec![py.None(); calls.len()];
let mut prob_bufs: Vec<xlog_cuda::CudaBuffer> = Vec::with_capacity(calls.len());
let mut grad_bufs: Vec<xlog_cuda::CudaBuffer> = Vec::with_capacity(calls.len());

for (network_name, entries) in grouped.iter() {
    let handle = self.network_registry.get(network_name).ok_or_else(|| {
        PyValueError::new_err(format!("Network '{}' not registered", network_name))
    })?;
    let module = handle.module().ok_or_else(|| {
        PyValueError::new_err(format!("Network '{}' has no module", network_name))
    })?;

    let mut batch_inputs = Vec::with_capacity(entries.len());
    for &(_call_idx, input_idx) in entries {
        let input_tensor = self.get_input_tensor(py, input_idx)?;
        batch_inputs.push(input_tensor);
    }
    let batch = torch.call_method1("stack", (batch_inputs,))?;
    let output_batch = module.call_method1(py, "__call__", (batch,))?;
    let output_batch = output_batch.bind(py).call_method0("contiguous")?;

    for (row_idx, (call_idx, _)) in entries.iter().enumerate() {
        let row = output_batch.call_method1("select", (0i32, row_idx))?;
        let row = row.call_method0("contiguous")?;
        let output_detached = row.call_method0("detach")?;

        let managed = dlpack_from_py(&output_detached)?;
        let prob_buf = self.output_provider
            .from_dlpack_tensors_with_schema(schema_f32.clone(), vec![managed])
            .map_err(|e| PyRuntimeError::new_err(format!("DLPack import failed: {}", e)))?;

        let grad_tensor = torch.call_method1("zeros_like", (&row,))?;
        let grad_tensor = grad_tensor.call_method0("contiguous")?;
        let grad_managed = dlpack_from_py(&grad_tensor)?;
        let grad_buf = self.output_provider
            .from_dlpack_tensors_with_schema(schema_f32.clone(), vec![grad_managed])
            .map_err(|e| PyRuntimeError::new_err(format!("DLPack import failed: {}", e)))?;

        out_tensors[*call_idx] = row.into();
        grad_tensors[*call_idx] = grad_tensor.into();
        prob_bufs.push(prob_buf);
        grad_bufs.push(grad_buf);
    }
}
```

class CoinNet(torch.nn.Module):

        super().__init__()
        self.conv = torch.nn.Sequential(
            torch.nn.Conv2d(3, 16, 3, stride=2, padding=1),

import argparse
from pathlib import Path
import sys
import os

ROOT = Path(__file__).resolve().parent
REPO = ROOT.parents[2]
sys.path.insert(0, str(REPO))

import torch
import pyxlog
from model import CoinNet
from scripts.neural_datasets import DatasetManifest, verify_manifest

DATA_ROOT = ROOT / "data" / "coins"
MANIFEST = ROOT / "dataset.json"


def verify_data():
    if not MANIFEST.exists():
        raise FileNotFoundError("dataset.json missing; generate via scripts.neural_datasets")
    manifest = DatasetManifest.load(MANIFEST)
    verify_manifest(ROOT, manifest)
    return manifest


def load_images():
    if not DATA_ROOT.exists():
        raise FileNotFoundError("Dataset missing: examples/neural/02_coins/data/coins")
    # expect data/coins/heads/*.png and data/coins/tails/*.png
    import torchvision
    transform = torchvision.transforms.Compose([
        torchvision.transforms.Resize((64, 64)),
        torchvision.transforms.ToTensor(),
    ])
    dataset = torchvision.datasets.ImageFolder(str(DATA_ROOT), transform=transform)
    images = torch.stack([img for img, _ in dataset])
    labels = [label for _, label in dataset]
    return images, labels


def create_program():
    source = (ROOT / "program.xlog").read_text()
    return pyxlog.Program.compile(source)


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--mode", choices=["ci", "dev", "release"], required=True)
    args = parser.parse_args()

    verify_data()
    images, labels = load_images()

    program = create_program()
    net = CoinNet()
    optim = torch.optim.Adam(net.parameters(), lr=1e-3)
    program.register_network("coin_net", net, optim, batching=True)
    program.add_tensor_source("train", images)

    # Train on symbolic sums of two coin predictions (0..2)
    pairs = [(2 * i, 2 * i + 1) for i in range(len(labels) // 2)]
    queries = []
    for a, b in pairs:
        target = int(labels[a]) + int(labels[b])
        queries.append(f"addition({a}, {b}, {target})")
    if args.mode == "ci":
        queries = queries[: min(32, len(queries))]
    program.train_epoch(queries, batch_size=8)


if __name__ == "__main__":
    main()
