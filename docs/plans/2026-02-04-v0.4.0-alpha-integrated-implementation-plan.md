```python
# scripts/neural_datasets.py
from __future__ import annotations

import json
from dataclasses import dataclass
from pathlib import Path
from typing import Dict, List, Optional


@dataclass(frozen=True)
class DatasetSplit:
    name: str
    count: int


@dataclass(frozen=True)
class DatasetArtifact:
    name: str
    url: str
    sha256: str


@dataclass
class DatasetManifest:
    name: str
    provider: str
    version: str
    splits: Dict[str, DatasetSplit]
    ci_subset: Dict[str, int]
    artifacts: List[DatasetArtifact]
    metrics: Dict[str, float]

    @staticmethod
    def load(path: Path) -> "DatasetManifest":
        data = json.loads(Path(path).read_text())
        splits = {
            k: DatasetSplit(name=k, count=int(v["count"]))
            for k, v in data.get("splits", {}).items()
        }
        artifacts = [
            DatasetArtifact(**a) for a in data.get("artifacts", [])
        ]
        return DatasetManifest(
            name=data["name"],
            provider=data["provider"],
            version=data.get("version", ""),
            splits=splits,
            ci_subset={k: int(v) for k, v in data.get("ci_subset", {}).items()},
            artifacts=artifacts,
            metrics={k: float(v) for k, v in data.get("metrics", {}).items()},
        )

    def save(self, path: Path) -> None:
        payload = {
            "name": self.name,
            "provider": self.provider,
            "version": self.version,
            "splits": {k: {"count": v.count} for k, v in self.splits.items()},
            "ci_subset": self.ci_subset,
            "artifacts": [a.__dict__ for a in self.artifacts],
            "metrics": self.metrics,
        }
        Path(path).write_text(json.dumps(payload, indent=2) + "\n")
```

```python
# scripts/neural_datasets.py
import hashlib
import urllib.request


def verify_sha256(path: Path, expected_hex: str) -> bool:
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for chunk in iter(lambda: f.read(1024 * 1024), b""):
            h.update(chunk)
    return h.hexdigest() == expected_hex


def download_file(url: str, dest: Path) -> None:
    dest.parent.mkdir(parents=True, exist_ok=True)
    with urllib.request.urlopen(url) as r, open(dest, "wb") as f:
        f.write(r.read())
```

```python
    print("+", " ".join(cmd))

        if train.exists():
```

```rust
// crates/pyxlog/src/neural_registry.rs
use std::collections::HashMap;
use xlog_logic::{parse_program, NeuralLabel};

#[derive(Debug, Clone)]
pub struct NeuralPredicateInfo {
    pub predicate: String,
    pub network: String,
    pub labels: Vec<String>,
}

#[derive(Debug, Default, Clone)]
pub struct NeuralPredicateRegistry {
    by_predicate: HashMap<String, NeuralPredicateInfo>,
}

impl NeuralPredicateRegistry {
    pub fn from_source(source: &str) -> Result<Self, String> {
        let ast = parse_program(source).map_err(|e| e.to_string())?;
        let mut registry = NeuralPredicateRegistry::default();

        for np in ast.neural_predicates {
            let labels = np
                .labels
                .unwrap_or_default()
                .into_iter()
                .map(|l| match l {
                    NeuralLabel::Integer(i) => i.to_string(),
                    NeuralLabel::Symbol(s) => s,
                })
                .collect::<Vec<_>>();

            registry.by_predicate.insert(
                np.predicate.predicate,
                NeuralPredicateInfo {
                    predicate: np.predicate.predicate,
                    network: np.network,
                    labels,
                },
            );
        }

        Ok(registry)
    }

    pub fn get(&self, predicate: &str) -> Option<&NeuralPredicateInfo> {
        self.by_predicate.get(predicate)
    }
}
```

```python
import torch

class CoinNet(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.conv = torch.nn.Sequential(
            torch.nn.Conv2d(3, 16, 3, stride=2, padding=1),
            torch.nn.ReLU(),
            torch.nn.Conv2d(16, 32, 3, stride=2, padding=1),
            torch.nn.ReLU(),
            torch.nn.AdaptiveAvgPool2d((1, 1)),
        )
        self.fc = torch.nn.Linear(32, 2)

    def forward(self, x):
        x = self.conv(x).view(x.size(0), -1)
        return torch.softmax(self.fc(x), dim=-1)
```

```python
ROOT = Path(__file__).resolve().parent

MANIFEST = ROOT / "dataset.json"


def load_dataset(mode: str):
    manifest = DatasetManifest.load(MANIFEST)
    # Use a public coin dataset mirrored as an ImageFolder (manual download).
    # Dataset must exist under data/coins/{train, test}/{heads, tails}
    if not (DATA / "coins" / "train").exists():
        raise SystemExit("Dataset missing: examples/neural/02_coins/data/coins")

    transform = transforms.Compose([
        transforms.Resize((64, 64)),
        transforms.ToTensor(),
    ])

    train_ds = datasets.ImageFolder(DATA / "coins" / "train", transform=transform)
    test_ds = datasets.ImageFolder(DATA / "coins" / "test", transform=transform)

    if mode == "ci":
        train_ds.samples = train_ds.samples[: manifest.ci_subset["train"]]
        test_ds.samples = test_ds.samples[: manifest.ci_subset["test"]]

    return train_ds, test_ds


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--mode", choices=["ci", "dev", "release"], required=True)
    parser.add_argument("--epochs", type=int, default=2)
    parser.add_argument("--batch-size", type=int, default=32)
    args = parser.parse_args()

    train_ds, _ = load_dataset(args.mode)
    loader = torch.utils.data.DataLoader(train_ds, batch_size=args.batch_size, shuffle=True)

    program = pyxlog.Program.compile(Path(ROOT / "program.xlog").read_text())
    net = CoinNet()
    opt = torch.optim.Adam(net.parameters(), lr=1e-3)
    program.register_network("coin_net", net, opt)

    program.add_tensor_source("train", images)

    for _ in range(args.epochs):

    return 0
```

```python
import argparse
from pathlib import Path

import numpy as np

def load_svhn_two_digit(mode: str, limit: int):
    # Expect SVHN files extracted under data/svhn/train/ and digitStruct.mat present
    mat_path = DATA / "train" / "digitStruct.mat"
    if not mat_path.exists():
        raise SystemExit("SVHN data missing. Place SVHN train/ with digitStruct.mat under examples/neural/03_mnist_multidigit/data/svhn")

    with h5py.File(mat_path, "r") as f:
        bbox = f["digitStruct"]["bbox"]
        names = f["digitStruct"]["name"]

        def get_name(idx):
            return "".join(chr(c[0]) for c in f[names[idx][0]].value)

        def get_labels(idx):
            labels_ref = bbox[idx][0]["label"]
            if labels_ref.shape[0] == 1:
                return [int(labels_ref[0][0])]
            return [int(f[r[0]][0][0]) for r in labels_ref]

        images = []
        labels = []
        for i in range(len(names)):
            digits = get_labels(i)
            if len(digits) != 2:
                continue
            img_path = DATA / "train" / get_name(i)
            if not img_path.exists():
                continue
            images.append(Image.open(img_path).convert("RGB"))
            labels.append(digits)
            if len(images) >= limit:
                break

    transform = transforms.Compose([
        transforms.Resize((64, 64)),
        transforms.ToTensor(),
    ])
    tensor = torch.stack([transform(im) for im in images])
    return tensor, labels


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--mode", choices=["ci", "dev", "release"], required=True)
    parser.add_argument("--epochs", type=int, default=2)
    args = parser.parse_args()

    limit = 128 if args.mode == "ci" else 2048
    images, labels = load_svhn_two_digit(args.mode, limit)

    program = pyxlog.Program.compile((ROOT / "program.xlog").read_text())
    left_net = DigitNet()
    right_net = DigitNet()
    program.register_network("left_net", left_net, torch.optim.Adam(left_net.parameters(), lr=1e-3))
    program.register_network("right_net", right_net, torch.optim.Adam(right_net.parameters(), lr=1e-3))

    program.add_tensor_source("train", images)
    queries = [f"number({i}, {labels[i][0]*10 + labels[i][1]})" for i in range(len(labels))]

    for _ in range(args.epochs):
        program.train_epoch(queries, batch_size=16)

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
import argparse
from pathlib import Path

from torchvision import transforms

def inkml_to_image(path: Path, size=128):
    tree = ET.parse(path)
    root = tree.getroot()
    img = Image.new("L", (size, size), color=255)
    draw = ImageDraw.Draw(img)
    for trace in root.findall(".//{*}trace"):
        points = []
        for pt in trace.text.strip().split(","):
            x, y = pt.strip().split(" ")
            points.append((float(x), float(y)))
        if len(points) >= 2:
            draw.line(points, fill=0, width=2)
    return img


def label_from_latex(latex: str):
    if "=" in latex:
        return "eq"
    if "+" in latex:
        return "add"
    if "-" in latex:
        return "sub"
    return "mul"


def load_crohme(mode: str, limit: int):
    if not DATA.exists():
        raise SystemExit("CROHME dataset missing. Place under examples/neural/04_hwf/data/crohme")
    inkml_files = list(DATA.glob("**/*.inkml"))
    labels = []
    images = []
    for f in inkml_files[:limit]:
        # Assume label is stored in accompanying .txt with same stem
        label_path = f.with_suffix(".txt")
        if not label_path.exists():
            continue
        latex = label_path.read_text().strip()
        label = label_from_latex(latex)
        images.append(inkml_to_image(f))
        labels.append(label)

    transform = transforms.Compose([transforms.Resize((128, 128)), transforms.ToTensor()])
    tensor = torch.stack([transform(im) for im in images])
    return tensor, labels


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--mode", choices=["ci", "dev", "release"], required=True)
    args = parser.parse_args()

    limit = 64 if args.mode == "ci" else 1024
    images, labels = load_crohme(args.mode, limit)

    program = pyxlog.Program.compile((ROOT / "program.xlog").read_text())
    net = HWFNet()
    program.register_network("hw_net", net, torch.optim.Adam(net.parameters(), lr=1e-3))
    program.add_tensor_source("train", images)

    queries = [f"expr_type({i}, {labels[i]})" for i in range(len(labels))]
    for _ in range(2):
        program.train_epoch(queries, batch_size=8)
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```python
import argparse

SUITS = ["C","D","H","S"]


def load_cards(limit):
    if not DATA.exists():
        raise SystemExit("Card dataset missing. Place under examples/neural/05_poker/data/cards")
    images = []
    labels = []
    for img in DATA.glob("**/*.jpg"):
        name = img.stem.upper()
        # Expect filename like \"AS\" for Ace of Spades
        rank = name[:-1]
        suit = name[-1]

        labels.append((rank, suit))
        if len(images) >= limit:

    transform = transforms.Compose([transforms.Resize((64,64)), transforms.ToTensor()])
```

import argparse
from pathlib import Path
import torch
from datasets import load_dataset
import pyxlog

from model import RelNet

ROOT = Path(__file__).resolve().parent


def build_vocab(texts, max_tokens=10000):
    vocab = {\"<pad>\": 0}
    for t in texts:
        for w in t.lower().split():
            if w not in vocab and len(vocab) < max_tokens:
                vocab[w] = len(vocab)
    return vocab


def encode(text, vocab, max_len=64):
    tokens = [vocab.get(w, 0) for w in text.lower().split()[:max_len]]
    return tokens + [0] * (max_len - len(tokens))


def main():
    parser = argparse.ArgumentParser()
    parser.add_argument(\"--mode\", choices=[\"ci\",\"dev\",\"release\"], required=True)
    args = parser.parse_args()
