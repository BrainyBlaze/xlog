## 2. GPU Memory & Kernel Execution

We must execute the super-graph efficiently. Evaluating an  join space naively will OOM the GPU. We use a custom CUDA kernel to extract *only* the active rules from the Straight-Through Gumbel-Softmax mask.

### 2.1 CUDA Kernel (`kernels/neural.cu`)

We add a kernel to extract the non-zero indices of the mask. This runs in microseconds.

```cuda
#include <stdint.h>

extern "C" __global__ void extract_nonzero_indices(
    const float* mask_tensor,    // N x N x N contiguous mask
    uint32_t N,
    uint32_t* out_i,             // Extracted Body1 relations
    uint32_t* out_j,             // Extracted Body2 relations
    uint32_t* out_k,             // Extracted Head relations
    uint32_t* active_count       // Atomic counter
) {
    uint32_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    uint32_t total = N * N * N;
    
    if (idx >= total) return;

    // Check if the ST Gumbel-Softmax snapped this to 1.0
    if (mask_tensor[idx] > 0.5f) {
        uint32_t i = idx / (N * N);
        uint32_t j = (idx / N) % N;
        uint32_t k = idx % N;

        uint32_t pos = atomicAdd(active_count, 1);
        out_i[pos] = i;
        out_j[pos] = j;
        out_k[pos] = k;
    }
}

```

### 2.2 CUDA Provider (`crates/xlog-cuda/src/provider.rs`)

Wrap the kernel in a safe Rust API using `cudarc`.

```rust
impl CudaKernelProvider {
    pub fn extract_active_rule_indices(&self, mask: &CudaColumn, n: usize) -> Result<Vec<(u32, u32, u32)>> {
        let total = n * n * n;
        let block_size = 256;
        let grid_size = (total + block_size - 1) / block_size;

        let out_i = self.memory.alloc::<u32>(total)?;
        let out_j = self.memory.alloc::<u32>(total)?;
        let out_k = self.memory.alloc::<u32>(total)?;
        let mut count = self.memory.alloc_zeros::<u32>(1)?;

        unsafe {
            self.module.get_function("extract_nonzero_indices")?
                .launch(
                    LaunchConfig { grid_dim: (grid_size as u32, 1, 1), block_dim: (block_size as u32, 1, 1), shared_mem_bytes: 0 },
                    (&mask.as_device_ptr(), n as u32, &out_i, &out_j, &out_k, &count)
                )?;
        }

        let active_count = count.to_host()?[0] as usize;
        let i_host = out_i.slice(0..active_count).to_host()?;
        let j_host = out_j.slice(0..active_count).to_host()?;
        let k_host = out_k.slice(0..active_count).to_host()?;

        Ok(izip!(i_host, j_host, k_host).collect())
    }
}
```

## 3. The Hot Loop Executor (`crates/xlog-runtime/src/executor.rs`)

The `Executor` handles the `TensorMaskedJoin` by dynamically dispatching the active combinations to the highly-optimized `hash_join_v2`.

```rust
impl Executor {
    fn execute_tensor_masked_join(&mut self, node: &RirNode) -> Result<CudaBuffer> {
        let RirNode::TensorMaskedJoin { mask_name, schema_size, left_keys, right_keys, target_rel } = node else { unreachable!() };
        
        // 1. Fetch the ST-Gumbel mask provided by PyTorch via DLPack
        let mask_col = self.neural_registry.get_mask(mask_name)?;
        
        // 2. Run the extraction kernel
        let active_rules = self.provider.extract_active_rule_indices(&mask_col, *schema_size)?;
        
        let mut target_results = Vec::new();
        
        // 3. Dispatch native joins
        for (i, j, k) in active_rules {
            if RelId(k) != *target_rel { continue; } // Only process rules aiming at the target relation
            
            let left_buf = self.store.get_by_id(RelId(i))?;
            let right_buf = self.store.get_by_id(RelId(j))?;
            
            // XLOG's ultra-fast bucketed hash join
            let joined_buf = self.provider.hash_join_v2(
                &left_buf, &right_buf, left_keys.clone(), right_keys.clone(), JoinType::Inner
            )?;
            
            // 4. Tag Provenance: This execution path relies on W_{i,j,k} being True
            let prov_joined = self.prob_engine.tag_tensor_provenance(joined_buf, mask_name, i, j, k)?;
            target_results.push(prov_joined);
        }
        
        // 5. Union all results for this relation
        self.provider.union_all(&target_results)
    }
}

```

```rust
    /// Injects the ST-Gumbel-Softmax mask into XLOG
    pub fn set_rule_mask(&mut self, name: &str, capsule: &PyCapsule) -> PyResult<()> {

        Ok(grads.into_py_capsule(py)?)
    }

    /// TIIPS: Extract current WMIR state so a Neural Teacher can inspect it
    pub fn extract_wmir_state(&self, py: Python) -> PyResult<HashMap<String, PyObject>> {
        let mut state = HashMap::new();
        for (name, rel_id) in &self.executor.name_to_rel {
            if let Ok(buf) = self.executor.store.get_by_id(*rel_id) {
                state.insert(name.clone(), buf.to_dlpack(py)?);
            }
        }
        Ok(state)
    }

    /// TIIPS: Evaluate with a hallucinated Transductive Subgoal as the target
    pub fn evaluate_with_subgoal(&mut self, target_rel: &str, subgoal_dlpack: &PyCapsule) -> PyResult<PyObject> {

        // Compute cross entropy loss inside XLOG between target_buf and derived rel
        let loss = self.prob_engine.compute_subgoal_loss(target_rel, &target_buf)?;
        
        // Trigger backward pass
        self.prob_engine.backward(loss)?;
        Ok(py.None()) // Gradients are now ready to be fetched
    }
```

```python
import pyxlog
import torch
import torch.nn.functional as F

class DTS_MindMachine:
    def __init__(self, xlog_source, schema_size):
        self.program = pyxlog.Program.compile(xlog_source, prob_engine="exact_ddnnf")
        self.N = schema_size
        
        # The continuous rule tensor (Inductive Student)
        self.W = torch.randn((self.N, self.N, self.N), requires_grad=True, device='cuda')
        self.optimizer = torch.optim.Adam([self.W], lr=0.1)
        
        # The diffusion backbone (Transductive Teacher)
        self.neural_backbone = load_diffusion_model()

    def train_step(self, raw_evidence):
        self.optimizer.zero_grad()
        
        # --- 1. TIIPS Selective Transductive Guidance ---
        # Get current state of logic engine
        current_state = self.program.extract_wmir_state()
        
        # If logic is stuck (metastable), use the neural backbone to hallucinate 
        # a transductive subgoal (e.g. predicting intermediate variables directly from pixels)
        subgoal_tensor = self.neural_backbone.predict_subgoal(raw_evidence, current_state)
        subgoal_capsule = torch.utils.dlpack.to_dlpack(subgoal_tensor)
        
        # --- 2. Continuous to Discrete (ST Gumbel-Softmax) ---
        M_soft = F.gumbel_softmax(self.W, tau=0.5, hard=False)
        index = M_soft.max(dim=-1, keepdim=True)[1]
        M_hard = torch.zeros_like(self.W).scatter_(-1, index, 1.0)
        
        # Straight-through proxy: Hard in forward pass, soft in backward pass
        M = (M_hard - M_soft).detach() + M_soft 
        
        # --- 3. XLOG Execution ---
        # Inject the mask into the GPU runtime via zero-copy
        self.program.set_rule_mask("W_mask", torch.utils.dlpack.to_dlpack(M))
        
        # Evaluate logic targeting the transductive subgoal, triggering backward pass
        self.program.evaluate_with_subgoal("target_relation", subgoal_capsule)
        
        # --- 4. Backpropagation ---
        # Extract the Product T-Norm gradients computed by the XGCF circuit
        grad_capsule = self.program.get_rule_gradients("W_mask", self.N)
        xgcf_grads = torch.from_dlpack(grad_capsule)
        
        # Backprop through the Gumbel-Softmax onto the W tensor
        M.backward(xgcf_grads)
        self.optimizer.step()
        
        # --- 5. Meta-Rule Commit ---
        if self._check_convergence():
            self._harden_and_commit_rule(M_hard)
```
