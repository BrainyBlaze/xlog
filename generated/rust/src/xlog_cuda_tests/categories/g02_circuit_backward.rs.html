1//! Circuit backward kernel tests
2//!
3//! Tests the backward evaluation kernels for XGCF circuits, covering adjoint propagation,
4//! gradient accumulation, and numerical verification. These tests verify correct gradient
5//! computation for neural-symbolic integration.

15/// Run all tests in this category.
16pub fn run_all(ctx: &TestContext) -> CategoryResult {
17    let mut results = CategoryResult::new("circuit_backward_kernel");
18    let start = Instant::now();
19
20    // Adjoint propagation checks.
21    results.add_result(test_backward_root_adjoint(ctx));
22    results.add_result(test_backward_and_propagate(ctx));
23    results.add_result(test_backward_or_propagate(ctx));
24    results.add_result(test_backward_decision_propagate(ctx));
25
26    // Gradient accumulation checks.
27    results.add_result(test_backward_lit_grad(ctx));
28    results.add_result(test_backward_decision_grad(ctx));
29    results.add_result(test_backward_gradient_accumulation(ctx));
30    results.add_result(test_backward_negative_lit_grad(ctx));
31
32    // Numerical verification checks.
33    results.add_result(test_backward_vs_numerical_diff(ctx));
34    results.add_result(test_backward_chain_rule(ctx));
35    results.add_result(test_backward_gradient_stability(ctx));
36    results.add_result(test_backward_log_space_accuracy(ctx));
37
38    results.set_duration(start.elapsed());
39    results
40}

46/// Test 1: Verify root adjoint is initialized to 1.0
47///
48/// The backward pass starts with adj[root] = 1.0, representing the gradient of the
49/// output with respect to itself.
50fn test_backward_root_adjoint(ctx: &TestContext) -> TestResult {
51    let start = Instant::now();
52
53    let spec = tiny_xgcf_spec();
54    let run = match run_tiny_xgcf_backward(ctx, &spec) {
55        Ok(r) => r,
56        Err(e) => {
57            return TestResult::error(
58                "test_backward_root_adjoint",
59                start.elapsed(),
60                format!("Failed to run backward pass: {}", e),
61            );
62        }
63    };
64
65    // Verify root adjoint is 1.0
66    let root_adj = run.adj[spec.root as usize];
67    if (root_adj - 1.0).abs() > 1e-10 {
68        return TestResult::error(
69            "test_backward_root_adjoint",
70            start.elapsed(),
71            format!(
72                "Root adjoint should be 1.0, got {} (diff={})",
73                root_adj,
74                (root_adj - 1.0).abs()
75            ),
76        );
77    }
78
79    if let Err(e) = ctx.sync_and_check() {
80        return TestResult::error(
81            "test_backward_root_adjoint",
82            start.elapsed(),
83            format!("Sync failed: {}", e),
84        );
85    }
86
87    TestResult::passed("test_backward_root_adjoint", start.elapsed())
88}

90/// Test 2: AND node adjoint propagation
91///
92/// For AND nodes, each child receives the parent's adjoint unchanged because
93/// d(log(a*b))/d(log(a)) = 1 and d(log(a*b))/d(log(b)) = 1.
94fn test_backward_and_propagate(ctx: &TestContext) -> TestResult {
95    let start = Instant::now();
96
97    let spec = gen_and_circuit();
98    // AND circuit: node 0 = Lit(+1), node 1 = Lit(+2), node 2 = AND(0,1) [root]
99    let run = match run_tiny_xgcf_backward(ctx, &spec) {
100        Ok(r) => r,
101        Err(e) => {
102            return TestResult::error(
103                "test_backward_and_propagate",
104                start.elapsed(),
105                format!("Failed to run backward pass: {}", e),
106            );
107        }
108    };
109
110    // Root adjoint should be 1.0
111    let root_adj = run.adj[spec.root as usize];
112    if (root_adj - 1.0).abs() > 1e-10 {
113        return TestResult::error(
114            "test_backward_and_propagate",
115            start.elapsed(),
116            format!("Root adjoint should be 1.0, got {}", root_adj),
117        );
118    }
119
120    // For AND, children get parent's adjoint (chain rule: d(a+b)/da = 1)
121    // Child 0 and child 1 should each have adjoint = root_adj = 1.0
122    let child0_adj = run.adj[0];
123    let child1_adj = run.adj[1];
124
125    if (child0_adj - root_adj).abs() > 1e-10 {
126        return TestResult::error(
127            "test_backward_and_propagate",
128            start.elapsed(),
129            format!(
130                "AND child 0 adjoint should equal parent adjoint {}, got {}",
131                root_adj, child0_adj
132            ),
133        );
134    }
135
136    if (child1_adj - root_adj).abs() > 1e-10 {
137        return TestResult::error(
138            "test_backward_and_propagate",
139            start.elapsed(),
140            format!(
141                "AND child 1 adjoint should equal parent adjoint {}, got {}",
142                root_adj, child1_adj
143            ),
144        );
145    }
146
147    if let Err(e) = ctx.sync_and_check() {
148        return TestResult::error(
149            "test_backward_and_propagate",
150            start.elapsed(),
151            format!("Sync failed: {}", e),
152        );
153    }
154
155    TestResult::passed("test_backward_and_propagate", start.elapsed())
156}

158/// Test 3: OR node adjoint propagation
159///
160/// For OR nodes (logsumexp), children get adjoint weighted by softmax:
161/// adj[child_i] += adj[parent] * exp(val[child_i] - val[parent])
162fn test_backward_or_propagate(ctx: &TestContext) -> TestResult {
163    let start = Instant::now();
164
165    let spec = gen_or_circuit();
166    // OR circuit: node 0 = Lit(+1), node 1 = Lit(+2), node 2 = OR(0,1) [root]
167    let run = match run_tiny_xgcf_backward(ctx, &spec) {
168        Ok(r) => r,
169        Err(e) => {
170            return TestResult::error(
171                "test_backward_or_propagate",
172                start.elapsed(),
173                format!("Failed to run backward pass: {}", e),
174            );
175        }
176    };
177
178    // Root adjoint should be 1.0
179    let root_adj = run.adj[spec.root as usize];
180    let root_val = run.values[spec.root as usize];
181
182    if (root_adj - 1.0).abs() > 1e-10 {
183        return TestResult::error(
184            "test_backward_or_propagate",
185            start.elapsed(),
186            format!("Root adjoint should be 1.0, got {}", root_adj),
187        );
188    }
189
190    // For OR (logsumexp), child adjoint = parent_adj * exp(child_val - parent_val)
191    let child0_val = run.values[0];
192    let child1_val = run.values[1];
193
194    let expected_adj0 = root_adj * (child0_val - root_val).exp();
195    let expected_adj1 = root_adj * (child1_val - root_val).exp();
196
197    let child0_adj = run.adj[0];
198    let child1_adj = run.adj[1];
199
200    if (child0_adj - expected_adj0).abs() > 1e-10 {
201        return TestResult::error(
202            "test_backward_or_propagate",
203            start.elapsed(),
204            format!(
205                "OR child 0 adjoint should be {}, got {} (diff={})",
206                expected_adj0,
207                child0_adj,
208                (child0_adj - expected_adj0).abs()
209            ),
210        );
211    }
212
213    if (child1_adj - expected_adj1).abs() > 1e-10 {
214        return TestResult::error(
215            "test_backward_or_propagate",
216            start.elapsed(),
217            format!(
218                "OR child 1 adjoint should be {}, got {} (diff={})",
219                expected_adj1,
220                child1_adj,
221                (child1_adj - expected_adj1).abs()
222            ),
223        );
224    }
225
226    // Verify softmax weights sum to 1
227    let softmax_sum = expected_adj0 + expected_adj1;
228    if (softmax_sum - 1.0).abs() > 1e-10 {
229        return TestResult::error(
230            "test_backward_or_propagate",
231            start.elapsed(),
232            format!("OR softmax weights should sum to 1.0, got {}", softmax_sum),
233        );
234    }
235
236    if let Err(e) = ctx.sync_and_check() {
237        return TestResult::error(
238            "test_backward_or_propagate",
239            start.elapsed(),
240            format!("Sync failed: {}", e),
241        );
242    }
243
244    TestResult::passed("test_backward_or_propagate", start.elapsed())
245}

247/// Test 4: Decision node adjoint propagation
248///
249/// Decision nodes propagate adjoint to their children weighted by branch probability:
250/// adj[child_false] += adj[parent] * exp(log_false + val_false - val_parent)
251/// adj[child_true] += adj[parent] * exp(log_true + val_true - val_parent)
252fn test_backward_decision_propagate(ctx: &TestContext) -> TestResult {
253    let start = Instant::now();
254
255    let spec = gen_decision_circuit();
256    // Decision circuit: node 0 = CONST1, node 1 = Lit(+1), node 2 = Decision(var2, 0, 1) [root]
257    let run = match run_tiny_xgcf_backward(ctx, &spec) {
258        Ok(r) => r,
259        Err(e) => {
260            return TestResult::error(
261                "test_backward_decision_propagate",
262                start.elapsed(),
263                format!("Failed to run backward pass: {}", e),
264            );
265        }
266    };
267
268    // Root adjoint should be 1.0
269    let root_adj = run.adj[spec.root as usize];
270    let root_val = run.values[spec.root as usize];
271
272    if (root_adj - 1.0).abs() > 1e-10 {
273        return TestResult::error(
274            "test_backward_decision_propagate",
275            start.elapsed(),
276            format!("Root adjoint should be 1.0, got {}", root_adj),
277        );
278    }
279
280    // Decision var is at index 2 in the decision_var array for node 2
281    let decision_var = spec.decision_var[spec.root as usize] as usize;
282    let log_false = spec.var_log_false[decision_var];
283    let log_true = spec.var_log_true[decision_var];
284
285    let val_false = run.values[0]; // CONST1 = 0.0
286    let val_true = run.values[1]; // Lit(+1)
287
288    // Expected child adjoints based on branch probability
289    let expected_adj_false = root_adj * (log_false + val_false - root_val).exp();
290    let expected_adj_true = root_adj * (log_true + val_true - root_val).exp();
291
292    let child_false_adj = run.adj[0];
293    let child_true_adj = run.adj[1];
294
295    if (child_false_adj - expected_adj_false).abs() > 1e-10 {
296        return TestResult::error(
297            "test_backward_decision_propagate",
298            start.elapsed(),
299            format!(
300                "Decision false-child adjoint should be {}, got {} (diff={})",
301                expected_adj_false,
302                child_false_adj,
303                (child_false_adj - expected_adj_false).abs()
304            ),
305        );
306    }
307
308    if (child_true_adj - expected_adj_true).abs() > 1e-10 {
309        return TestResult::error(
310            "test_backward_decision_propagate",
311            start.elapsed(),
312            format!(
313                "Decision true-child adjoint should be {}, got {} (diff={})",
314                expected_adj_true,
315                child_true_adj,
316                (child_true_adj - expected_adj_true).abs()
317            ),
318        );
319    }
320
321    // Verify branch probabilities sum to 1
322    let prob_sum = expected_adj_false + expected_adj_true;
323    if (prob_sum - 1.0).abs() > 1e-10 {
324        return TestResult::error(
325            "test_backward_decision_propagate",
326            start.elapsed(),
327            format!(
328                "Decision branch probabilities should sum to 1.0, got {}",
329                prob_sum
330            ),
331        );
332    }
333
334    if let Err(e) = ctx.sync_and_check() {
335        return TestResult::error(
336            "test_backward_decision_propagate",
337            start.elapsed(),
338            format!("Sync failed: {}", e),
339        );
340    }
341
342    TestResult::passed("test_backward_decision_propagate", start.elapsed())
343}

349/// Test 5: Positive literal gradient contribution
350///
351/// A positive literal Lit(+v) contributes to grad_true[v] with the node's adjoint.
352fn test_backward_lit_grad(ctx: &TestContext) -> TestResult {
353    let start = Instant::now();
354
355    let spec = gen_single_lit_circuit(1);
356    // Single literal circuit: node 0 = Lit(+1) [root]
357    let run = match run_tiny_xgcf_backward(ctx, &spec) {
358        Ok(r) => r,
359        Err(e) => {
360            return TestResult::error(
361                "test_backward_lit_grad",
362                start.elapsed(),
363                format!("Failed to run backward pass: {}", e),
364            );
365        }
366    };
367
368    // Positive literal should contribute to grad_true
369    let expected_grad_true = 1.0; // adjoint of root = 1.0
370    let expected_grad_false = 0.0;
371
372    if (run.grad_true[1] - expected_grad_true).abs() > 1e-10 {
373        return TestResult::error(
374            "test_backward_lit_grad",
375            start.elapsed(),
376            format!(
377                "Positive literal grad_true[1] should be {}, got {}",
378                expected_grad_true, run.grad_true[1]
379            ),
380        );
381    }
382
383    if (run.grad_false[1] - expected_grad_false).abs() > 1e-10 {
384        return TestResult::error(
385            "test_backward_lit_grad",
386            start.elapsed(),
387            format!(
388                "Positive literal grad_false[1] should be {}, got {}",
389                expected_grad_false, run.grad_false[1]
390            ),
391        );
392    }
393
394    // Verify against expected values from spec
395    for (i, (&actual, &expected)) in run
396        .grad_true
397        .iter()
398        .zip(spec.expected_grad_true.iter())
399        .enumerate()
400    {
401        if (actual - expected).abs() > 1e-10 {
402            return TestResult::error(
403                "test_backward_lit_grad",
404                start.elapsed(),
405                format!(
406                    "grad_true[{}] mismatch: expected {}, got {}",
407                    i, expected, actual
408                ),
409            );
410        }
411    }
412
413    if let Err(e) = ctx.sync_and_check() {
414        return TestResult::error(
415            "test_backward_lit_grad",
416            start.elapsed(),
417            format!("Sync failed: {}", e),
418        );
419    }
420
421    TestResult::passed("test_backward_lit_grad", start.elapsed())
422}

679/// Test 9: Compare GPU gradients vs finite differences
680///
681/// Validates gradient correctness by comparing against numerical differentiation.
682fn test_backward_vs_numerical_diff(ctx: &TestContext) -> TestResult {
683    let start = Instant::now();
684
685    let spec = gen_decision_circuit();
686    let run = match run_tiny_xgcf_backward(ctx, &spec) {
687        Ok(r) => r,
688        Err(e) => {
689            return TestResult::error(
690                "test_backward_vs_numerical_diff",
691                start.elapsed(),
692                format!("Failed to run backward pass: {}", e),
693            );
694        }
695    };
696
697    let eps = 1e-6;
698    let tolerance = 1e-5;
699
700    // Check numerical gradients for each variable
701    for var in 1..=spec.num_vars {
702        let (num_grad_true, num_grad_false) = match numerical_gradient(ctx, &spec, var, eps) {
703            Ok(g) => g,
704            Err(e) => {
705                return TestResult::error(
706                    "test_backward_vs_numerical_diff",
707                    start.elapsed(),
708                    format!(
709                        "Failed to compute numerical gradient for var {}: {}",
710                        var, e
711                    ),
712                );
713            }
714        };
715
716        let gpu_grad_true = run.grad_true[var];
717        let gpu_grad_false = run.grad_false[var];
718

723            return TestResult::error(
724                "test_backward_vs_numerical_diff",
725                start.elapsed(),

727                    "var {} grad_true mismatch: GPU={}, numerical={} (diff={})",
728                    var, gpu_grad_true, num_grad_true, diff_true
729                ),
730            );
731        }
732
733        if diff_false > tolerance {
734            return TestResult::error(
735                "test_backward_vs_numerical_diff",
736                start.elapsed(),
737                format!(
738                    "var {} grad_false mismatch: GPU={}, numerical={} (diff={})",
739                    var, gpu_grad_false, num_grad_false, diff_false

741            );
742        }
743    }

747            "test_backward_vs_numerical_diff",

756/// Test 10: Chain rule through multiple levels
757///
758/// Verifies that the chain rule is correctly applied through a deep chain of nodes.
759fn test_backward_chain_rule(ctx: &TestContext) -> TestResult {
760    let start = Instant::now();
761
762    let depth = 10;
763    let spec = gen_deep_chain_circuit(depth);
764    let run = match run_tiny_xgcf_backward(ctx, &spec) {
765        Ok(r) => r,
766        Err(e) => {
767            return TestResult::error(
768                "test_backward_chain_rule",
769                start.elapsed(),
770                format!("Failed to run backward pass: {}", e),
771            );
772        }
773    };
774
775    // In a chain of ANDs, adjoint should propagate unchanged through all levels
776    // Root adjoint = 1.0, and each AND passes it to its child unchanged
777    for i in 0..=depth {
778        let adj = run.adj[i];
779        if (adj - 1.0).abs() > 1e-10 {
780            return TestResult::error(
781                "test_backward_chain_rule",
782                start.elapsed(),
783                format!(
784                    "Node {} adjoint should be 1.0 (chain rule through AND), got {}",
785                    i, adj
786                ),
787            );
788        }
789    }
790
791    // The gradient should reach the literal at the bottom
792    // grad_true[1] should be 1.0 (the adjoint of the literal)
793    if (run.grad_true[1] - 1.0).abs() > 1e-10 {
794        return TestResult::error(
795            "test_backward_chain_rule",
796            start.elapsed(),
797            format!(
798                "grad_true[1] should be 1.0 after chain rule, got {}",
799                run.grad_true[1]
800            ),
801        );
802    }
803
804    // Verify against expected values from spec
805    for (i, (&actual, &expected)) in run
806        .grad_true
807        .iter()
808        .zip(spec.expected_grad_true.iter())
809        .enumerate()
810    {
811        if (actual - expected).abs() > 1e-10 {
812            return TestResult::error(
813                "test_backward_chain_rule",
814                start.elapsed(),
815                format!(
816                    "grad_true[{}] mismatch: expected {}, got {}",
817                    i, expected, actual
818                ),
819            );
820        }
821    }
822
823    // Compare with numerical gradient
824    let eps = 1e-6;
825    let tolerance = 1e-5;
826    let (num_grad_true, _) = match numerical_gradient(ctx, &spec, 1, eps) {
827        Ok(g) => g,
828        Err(e) => {
829            return TestResult::error(
830                "test_backward_chain_rule",
831                start.elapsed(),
832                format!("Failed to compute numerical gradient: {}", e),
833            );
834        }
835    };
836
837    let diff = (run.grad_true[1] - num_grad_true).abs();
838    if diff > tolerance {
839        return TestResult::error(
840            "test_backward_chain_rule",
841            start.elapsed(),
842            format!(
843                "Chain rule gradient differs from numerical: GPU={}, numerical={} (diff={})",
844                run.grad_true[1], num_grad_true, diff
845            ),
846        );
847    }
848
849    if let Err(e) = ctx.sync_and_check() {
850        return TestResult::error(
851            "test_backward_chain_rule",
852            start.elapsed(),
853            format!("Sync failed: {}", e),
854        );
855    }
856
857    TestResult::passed("test_backward_chain_rule", start.elapsed())
858}

860/// Test 11: Gradient stability with extreme values
861///
862/// Verifies that gradients remain finite (no NaN/Inf) even with extreme log values.
863fn test_backward_gradient_stability(ctx: &TestContext) -> TestResult {
864    let start = Instant::now();
865
866    const LIT: u8 = 2;
867    const AND: u8 = 3;
868
869    // Create a circuit with extreme values
870    // AND(Lit(+1), Lit(+2)) where var1 has an extremely small probability
871    let spec = TinyXgcfSpec {
872        num_nodes: 3,
873        num_vars: 2,
874        root: 2,
875        node_type: vec![LIT, LIT, AND],
876        child_offsets: vec![0, 0, 0, 2],
877        child_indices: vec![0, 1],
878        lit: vec![1, 2, 0],
879        decision_var: vec![0, 0, 0],
880        decision_child_false: vec![0, 0, 0],
881        decision_child_true: vec![0, 0, 0],
882        level_nodes: vec![0, 1, 2],
883        levels: vec![(0, 2), (2, 1)],
884        // Extreme value: log(1e-300) for var1
885        var_log_true: vec![0.0, 1e-300_f64.ln(), 0.5_f64.ln()],
886        var_log_false: vec![0.0, (1.0 - 1e-300_f64).ln(), 0.5_f64.ln()],
887        expected_values: vec![
888            1e-300_f64.ln(),
889            0.5_f64.ln(),
890            1e-300_f64.ln() + 0.5_f64.ln(),
891        ],
892        expected_grad_true: vec![0.0, 1.0, 1.0],
893        expected_grad_false: vec![0.0, 0.0, 0.0],
894    };
895
896    let run = match run_tiny_xgcf_backward(ctx, &spec) {
897        Ok(r) => r,
898        Err(e) => {
899            return TestResult::error(
900                "test_backward_gradient_stability",
901                start.elapsed(),
902                format!("Failed to run backward pass: {}", e),
903            );
904        }
905    };
906
907    // Check that no values are NaN or Inf
908    for (i, &val) in run.values.iter().enumerate() {
909        if val.is_nan() {
910            return TestResult::error(
911                "test_backward_gradient_stability",
912                start.elapsed(),
913                format!("values[{}] is NaN with extreme input values", i),
914            );
915        }
916        // Values can be -inf for log(0), but not +inf
917        if val.is_infinite() && val.is_sign_positive() {
918            return TestResult::error(
919                "test_backward_gradient_stability",
920                start.elapsed(),
921                format!("values[{}] is +Inf with extreme input values", i),
922            );
923        }
924    }
925
926    for (i, &adj) in run.adj.iter().enumerate() {
927        if adj.is_nan() {
928            return TestResult::error(
929                "test_backward_gradient_stability",
930                start.elapsed(),
931                format!("adj[{}] is NaN with extreme input values", i),
932            );
933        }
934        if adj.is_infinite() {
935            return TestResult::error(
936                "test_backward_gradient_stability",
937                start.elapsed(),
938                format!("adj[{}] is Inf with extreme input values", i),
939            );
940        }
941    }
942
943    for (i, &g) in run.grad_true.iter().enumerate() {
944        if g.is_nan() {
945            return TestResult::error(
946                "test_backward_gradient_stability",
947                start.elapsed(),
948                format!("grad_true[{}] is NaN with extreme input values", i),
949            );
950        }
951        if g.is_infinite() {
952            return TestResult::error(
953                "test_backward_gradient_stability",
954                start.elapsed(),
955                format!("grad_true[{}] is Inf with extreme input values", i),
956            );
957        }
958    }
959
960    for (i, &g) in run.grad_false.iter().enumerate() {
961        if g.is_nan() {
962            return TestResult::error(
963                "test_backward_gradient_stability",
964                start.elapsed(),
965                format!("grad_false[{}] is NaN with extreme input values", i),
966            );
967        }
968        if g.is_infinite() {
969            return TestResult::error(
970                "test_backward_gradient_stability",
971                start.elapsed(),
972                format!("grad_false[{}] is Inf with extreme input values", i),
973            );
974        }
975    }
976
977    if let Err(e) = ctx.sync_and_check() {
978        return TestResult::error(
979            "test_backward_gradient_stability",
980            start.elapsed(),
981            format!("Sync failed: {}", e),
982        );
983    }
984
985    TestResult::passed("test_backward_gradient_stability", start.elapsed())
986}

988/// Test 12: Log-space accuracy with tight tolerance
989///
990/// Verifies that gradient computations in log space achieve relative error < 1e-10.
991fn test_backward_log_space_accuracy(ctx: &TestContext) -> TestResult {
992    let start = Instant::now();
993
994    let spec = tiny_xgcf_spec();
995    let run = match run_tiny_xgcf_backward(ctx, &spec) {
996        Ok(r) => r,
997        Err(e) => {
998            return TestResult::error(
999                "test_backward_log_space_accuracy",
1000                start.elapsed(),
1001                format!("Failed to run backward pass: {}", e),
1002            );
1003        }
1004    };
1005
1006    let tolerance = 1e-10;
1007
1008    // Check relative error for all gradients
1009    for (i, (&actual, &expected)) in run
1010        .grad_true
1011        .iter()
1012        .zip(spec.expected_grad_true.iter())
1013        .enumerate()
1014    {
1015        let abs_diff = (actual - expected).abs();
1016        let rel_err = if expected.abs() > 1e-15 {
1017            abs_diff / expected.abs()
1018        } else {
1019            abs_diff // Use absolute error for near-zero expected values
1020        };
1021
1022        if abs_diff > tolerance && rel_err > tolerance {
1023            return TestResult::error(
1024                "test_backward_log_space_accuracy",
1025                start.elapsed(),
1026                format!(
1027                    "grad_true[{}] exceeds tolerance: expected {}, got {} (abs_diff={}, rel_err={})",
1028                    i, expected, actual, abs_diff, rel_err
1029                ),
1030            );
1031        }
1032    }
1033
1034    for (i, (&actual, &expected)) in run
1035        .grad_false
1036        .iter()
1037        .zip(spec.expected_grad_false.iter())
1038        .enumerate()
1039    {
1040        let abs_diff = (actual - expected).abs();
1041        let rel_err = if expected.abs() > 1e-15 {
1042            abs_diff / expected.abs()
1043        } else {
1044            abs_diff // Use absolute error for near-zero expected values
1045        };
1046
1047        if abs_diff > tolerance && rel_err > tolerance {
1048            return TestResult::error(
1049                "test_backward_log_space_accuracy",
1050                start.elapsed(),
1051                format!(
1052                    "grad_false[{}] exceeds tolerance: expected {}, got {} (abs_diff={}, rel_err={})",
1053                    i, expected, actual, abs_diff, rel_err
1054                ),
1055            );
1056        }
1057    }
1058
1059    // Check forward values accuracy as well (they affect backward correctness)
1060    for (i, (&actual, &expected)) in run
1061        .values
1062        .iter()
1063        .zip(spec.expected_values.iter())
1064        .enumerate()
1065    {
1066        // Handle -inf comparison specially
1067        if expected.is_infinite() && expected.is_sign_negative() {
1068            if !actual.is_infinite() || !actual.is_sign_negative() {
1069                return TestResult::error(
1070                    "test_backward_log_space_accuracy",
1071                    start.elapsed(),
1072                    format!("values[{}] should be -inf, got {}", i, actual),
1073                );
1074            }
1075            continue;
1076        }
1077
1078        let abs_diff = (actual - expected).abs();
1079        let rel_err = if expected.abs() > 1e-15 {
1080            abs_diff / expected.abs()
1081        } else {
1082            abs_diff
1083        };
1084
1085        if abs_diff > tolerance && rel_err > tolerance {
1086            return TestResult::error(
1087                "test_backward_log_space_accuracy",
1088                start.elapsed(),
1089                format!(
1090                    "values[{}] exceeds tolerance: expected {}, got {} (abs_diff={}, rel_err={})",
1091                    i, expected, actual, abs_diff, rel_err
1092                ),
1093            );
1094        }
1095    }
1096
1097    if let Err(e) = ctx.sync_and_check() {
1098        return TestResult::error(
1099            "test_backward_log_space_accuracy",
1100            start.elapsed(),
1101            format!("Sync failed: {}", e),
1102        );
1103    }
1104
1105    TestResult::passed("test_backward_log_space_accuracy", start.elapsed())
1106}
