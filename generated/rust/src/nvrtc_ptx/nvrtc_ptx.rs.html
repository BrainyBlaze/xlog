1use std::fs;
2use std::path::PathBuf;
3
4use cudarc::nvrtc::{compile_ptx_with_opts, CompileOptions};
5
6fn kernels_dir() -> PathBuf {
7    let manifest_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
8    let packaged_dir = manifest_dir.join("kernels");
9    if packaged_dir.is_dir() {
10        return packaged_dir;
11    }
12
13    manifest_dir
14        .parent()
15        .expect("crates/")
16        .parent()
17        .expect("repo root")
18        .join("kernels")
19}
20
21fn main() {
22    let kernel = std::env::args().nth(1).unwrap_or_else(|| "sat".to_string());
23    let kernels_dir = kernels_dir();
24    let cu_path = kernels_dir.join(format!("{}.cu", kernel));
25    let ptx_path = kernels_dir.join(format!("{}.ptx", kernel));
26
27    let src = fs::read_to_string(&cu_path)
28        .unwrap_or_else(|e| panic!("Failed to read {}: {}", cu_path.display(), e));
29
30    // Keep options minimal so this works in environments that only have NVRTC available.
31    // Default sm_75 is the lowest arch supported by CUDA 13+.
32    // Leak is fine — this is a one-shot CLI tool.
33    let arch: &'static str = Box::leak(
34        std::env::var("XLOG_NVRTC_ARCH")
35            .unwrap_or_else(|_| "sm_75".to_string())
36            .into_boxed_str(),
37    );
38    let opts = CompileOptions {
39        arch: Some(arch),
40        ..Default::default()
41    };
42
43    let ptx =
44        compile_ptx_with_opts(&src, opts).unwrap_or_else(|e| panic!("NVRTC compile failed: {e}"));
45
46    fs::write(&ptx_path, ptx.to_src())
47        .unwrap_or_else(|e| panic!("Failed to write {}: {}", ptx_path.display(), e));
48
49    println!("wrote {}", ptx_path.display());
50}
