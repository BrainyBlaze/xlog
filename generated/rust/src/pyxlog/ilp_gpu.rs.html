23/// Task descriptor for COO fill (one per mask-query x candidate).
24pub(crate) struct CooTask {
25    pub(crate) d_mask: TrackedCudaSlice<u8>,
26    pub(crate) fact_indices_idx: usize,
27    pub(crate) cidx: u32,
28    pub(crate) num_query: u32,
29}

35/// Phase C (non-chunked): single-pass COO fill on device.
36///
37/// Over-allocates COO arrays at `upper_bound` and fills sentinels for unused
38/// slots. Returns `(d_coo_facts, d_coo_cands, actual_nnz)` where actual_nnz
39/// equals upper_bound (sentinels sort to end and are ignored later).
40pub(crate) fn build_coo_single(
41    provider: &CudaKernelProvider,
42    tasks: &[CooTask],
43    fact_indices_buffers: &[TrackedCudaSlice<u32>],
44    num_facts: u32,
45    num_cands: u32,
46    upper_bound: u32,
47) -> PyResult<(TrackedCudaSlice<u32>, TrackedCudaSlice<u32>, u32)> {
48    let num_tasks = tasks.len();
49
50    let mut offsets_host = vec![0u32; num_tasks];
51    {
52        let mut running = 0u32;
53        for i in 0..num_tasks {
54            offsets_host[i] = running;
55            running += tasks[i].num_query;
56        }
57        debug_assert_eq!(running, upper_bound);
58    }
59
60    let mut d_offsets = provider
61        .memory()
62        .alloc::<u32>(num_tasks)
63        .map_err(|e| types::gpu_err("alloc d_offsets", e))?;
64    provider
65        .device()
66        .inner()
67        .htod_sync_copy_into(&offsets_host, &mut d_offsets)
68        .map_err(|e| types::gpu_err("htod d_offsets", e))?;
69
70    let sentinel_fact = num_facts;
71    let sentinel_cand = num_cands;
72    let mut d_coo_facts = provider
73        .memory()
74        .alloc::<u32>(upper_bound as usize)
75        .map_err(|e| types::gpu_err("alloc coo_facts", e))?;
76    let mut d_coo_cands = provider
77        .memory()
78        .alloc::<u32>(upper_bound as usize)
79        .map_err(|e| types::gpu_err("alloc coo_cands", e))?;
80    {
81        let sentinel_facts_vec = vec![sentinel_fact; upper_bound as usize];
82        let sentinel_cands_vec = vec![sentinel_cand; upper_bound as usize];
83        provider
84            .device()
85            .inner()
86            .htod_sync_copy_into(&sentinel_facts_vec, &mut d_coo_facts)
87            .map_err(|e| types::gpu_err("sentinel facts", e))?;
88        provider
89            .device()
90            .inner()
91            .htod_sync_copy_into(&sentinel_cands_vec, &mut d_coo_cands)
92            .map_err(|e| types::gpu_err("sentinel cands", e))?;
93    }
94
95    for (task_idx, task) in tasks.iter().enumerate() {
96        let d_prefix = provider
97            .scan_u8_mask_device(&task.d_mask, task.num_query)
98            .map_err(|e| types::gpu_err("scan mask", e))?;
99
100        provider
101            .ilp_coo_fill_from_mask_launch(
102                &task.d_mask,
103                &d_prefix,
104                &fact_indices_buffers[task.fact_indices_idx],
105                task_idx as u32,
106                task.cidx,
107                task.num_query,
108                &d_offsets,
109                &mut d_coo_facts,
110                &mut d_coo_cands,
111            )
112            .map_err(|e| types::gpu_err("coo fill", e))?;
113    }
114
115    Ok((d_coo_facts, d_coo_cands, upper_bound))
116}

118/// Phase C (chunked): two-pass GPU-only bounded-memory merge.
119///
120/// Pass 1: Count NNZ per task on device (bounded temp per chunk).
121/// Pass 2: Fill COO at pre-computed offsets (bounded temp per chunk).
122/// The final COO buffer is exact-NNZ sized and may exceed `coo_chunk_budget`.
123///
124/// Returns `(d_coo_facts, d_coo_cands, total_nnz)`, or `None` if total_nnz
125/// is zero (caller should handle the empty-COO path).
126pub(crate) fn build_coo_chunked(
127    provider: &CudaKernelProvider,
128    tasks: &[CooTask],
129    fact_indices_buffers: &[TrackedCudaSlice<u32>],
130    _num_facts: u32,
131    coo_chunk_budget: u64,
132) -> PyResult<Option<(TrackedCudaSlice<u32>, TrackedCudaSlice<u32>, u32)>> {
133    let num_tasks = tasks.len();
134    let max_queries_per_chunk = (coo_chunk_budget / 8).max(1) as u32;
135
136    // Allocate task_counts array (num_tasks + 1 for exclusive scan).
137    let tc_len = num_tasks + 1;
138    let mut d_task_counts = provider
139        .memory()
140        .alloc::<u32>(tc_len)
141        .map_err(|e| types::gpu_err("alloc task_counts", e))?;
142    // Zero-init the entire array (counts default to 0, last slot = 0 for scan).
143    {
144        let zeros = vec![0u32; tc_len];
145        provider
146            .device()
147            .inner()
148            .htod_sync_copy_into(&zeros, &mut d_task_counts)
149            .map_err(|e| types::gpu_err("zero task_counts", e))?;
150    }
151
152    // ── Pass 1: Count NNZ per task ──
153    {
154        let mut chunk_start = 0usize;
155        while chunk_start < tasks.len() {
156            let mut chunk_end = chunk_start;
157            let mut chunk_sum = 0u32;
158            while chunk_end < tasks.len() {
159                let nq = tasks[chunk_end].num_query;
160                if chunk_sum + nq > max_queries_per_chunk && chunk_sum > 0 {
161                    break;
162                }
163                chunk_sum += nq;
164                chunk_end += 1;
165            }
166            if chunk_end == chunk_start {
167                chunk_end = chunk_start + 1;
168            }
169
170            for task_idx in chunk_start..chunk_end {
171                let task = &tasks[task_idx];
172                provider
173                    .count_mask_into_slot(
174                        &task.d_mask,
175                        task.num_query,
176                        &mut d_task_counts,
177                        task_idx,
178                    )
179                    .map_err(|e| types::gpu_err("count_mask_into_slot", e))?;
180            }
181
182            chunk_start = chunk_end;
183        }
184    }
185
186    // Synchronize to ensure all count kernels have completed.
187    provider
188        .device()
189        .synchronize()
190        .map_err(|e| types::gpu_err("sync pass1", e))?;
191
192    // Compute per-task write offsets via exclusive scan.
193    // d_task_counts[0..num_tasks] has counts; after scan,
194    // d_task_counts[i] = sum of counts[0..i] = write offset for task i.
195    // d_task_counts[num_tasks] = total_nnz.
196    provider
197        .exclusive_scan_u32_inplace(&mut d_task_counts, tc_len as u32)
198        .map_err(|e| types::gpu_err("scan task_counts", e))?;
199
200    // Read total_nnz from the last element (metadata-only, untracked).
201    let total_nnz: u32 = provider
202        .dtoh_scalar_untracked(&d_task_counts, num_tasks)
203        .map_err(|e| types::gpu_err("read total_nnz", e))?;
204
205    if total_nnz == 0 {
206        return Ok(None);
207    }
208
209    // Allocate exact-NNZ global COO buffers (may exceed coo_chunk_budget).
210    let mut d_coo_facts = provider
211        .memory()
212        .alloc::<u32>(total_nnz as usize)
213        .map_err(|e| types::gpu_err("alloc global coo_facts", e))?;
214    let mut d_coo_cands = provider
215        .memory()
216        .alloc::<u32>(total_nnz as usize)
217        .map_err(|e| types::gpu_err("alloc global coo_cands", e))?;
218
219    // ── Pass 2: Fill COO at pre-computed offsets ──
220    // d_task_counts now contains offsets (after exclusive scan).
221    // ilp_coo_fill_from_mask_launch reads d_offsets[offset_idx] for the
222    // write base position. We pass d_task_counts as d_offsets with
223    // offset_idx = task_idx.
224    {
225        let mut chunk_start = 0usize;
226        while chunk_start < tasks.len() {
227            let mut chunk_end = chunk_start;
228            let mut chunk_sum = 0u32;
229            while chunk_end < tasks.len() {
230                let nq = tasks[chunk_end].num_query;
231                if chunk_sum + nq > max_queries_per_chunk && chunk_sum > 0 {
232                    break;
233                }
234                chunk_sum += nq;
235                chunk_end += 1;
236            }
237            if chunk_end == chunk_start {
238                chunk_end = chunk_start + 1;
239            }
240
241            for task_idx in chunk_start..chunk_end {
242                let task = &tasks[task_idx];
243
244                let d_prefix = provider
245                    .scan_u8_mask_device(&task.d_mask, task.num_query)
246                    .map_err(|e| types::gpu_err("scan mask", e))?;
247
248                provider
249                    .ilp_coo_fill_from_mask_launch(
250                        &task.d_mask,
251                        &d_prefix,
252                        &fact_indices_buffers[task.fact_indices_idx],
253                        task_idx as u32,
254                        task.cidx,
255                        task.num_query,
256                        &d_task_counts, // offsets after scan
257                        &mut d_coo_facts,
258                        &mut d_coo_cands,
259                    )
260                    .map_err(|e| types::gpu_err("coo fill", e))?;
261                // d_prefix dropped here (bounded temp).
262            }
263
264            chunk_start = chunk_end;
265        }
266    }
267
268    Ok(Some((d_coo_facts, d_coo_cands, total_nnz)))
269}

271// ---------------------------------------------------------------------------
272// Phase D: Sort COO + device-side CSR build
273// ---------------------------------------------------------------------------
274
275/// Sort all COO entries by fact index and build CSR row offsets.
276///
277/// In the non-chunked path, sentinels (fact = num_facts) sort to the end
278/// and are ignored by the histogram kernel's `f < num_facts` guard. In the
279/// chunked path, the COO is exact-NNZ (no sentinels).
280pub(crate) fn sort_and_build_csr(
281    provider: &CudaKernelProvider,
282    d_coo_facts: &mut TrackedCudaSlice<u32>,
283    d_coo_cands: &mut TrackedCudaSlice<u32>,
284    actual_nnz: u32,
285    num_facts: u32,
286) -> PyResult<TrackedCudaSlice<u32>> {
287    let mut scratch = xlog_cuda::provider::RadixSortScratch::new(provider, actual_nnz)
288        .map_err(|e| types::gpu_err("sort scratch", e))?;
289    provider
290        .radix_sort_u32_pairs(d_coo_facts, d_coo_cands, actual_nnz, &mut scratch)
291        .map_err(|e| types::gpu_err("radix sort", e))?;
292
293    let d_hist = provider
294        .ilp_csr_histogram_launch(d_coo_facts, actual_nnz, num_facts)
295        .map_err(|e| types::gpu_err("csr histogram", e))?;
296
297    let mut d_row_offsets = provider
298        .memory()
299        .alloc::<u32>((num_facts + 1) as usize)
300        .map_err(|e| types::gpu_err("alloc row_offsets", e))?;
301    {
302        let zeros = vec![0u32; (num_facts + 1) as usize];
303        provider
304            .device()
305            .inner()
306            .htod_sync_copy_into(&zeros, &mut d_row_offsets)
307            .map_err(|e| types::gpu_err("zero row_offsets", e))?;
308    }
309    {
310        let mut dst_view = d_row_offsets
311            .try_slice_mut(0..num_facts as usize)
312            .ok_or_else(|| PyRuntimeError::new_err("row_offsets slice failed"))?;
313        provider
314            .device()
315            .inner()
316            .dtod_copy(&d_hist, &mut dst_view)
317            .map_err(|e| types::gpu_err("dtod hist->row_offsets", e))?;
318    }
319    provider
320        .exclusive_scan_u32_inplace(&mut d_row_offsets, num_facts + 1)
321        .map_err(|e| types::gpu_err("scan row_offsets", e))?;
322
323    Ok(d_row_offsets)
324}

330/// Run forward credit-gather, NLL loss reduction, and backward gradient
331/// scatter on device. Returns `(loss_capsule, grad_capsule)` DLPack objects.
332pub(crate) fn forward_backward_reduce(
333    provider: &CudaKernelProvider,
334    py: Python<'_>,
335    d_row_offsets: &TrackedCudaSlice<u32>,
336    d_coo_cands: &TrackedCudaSlice<u32>,
337    cand_col: &xlog_cuda::CudaColumn,
338    d_is_positive: &TrackedCudaSlice<u8>,
339    num_facts: u32,
340    num_cands: u32,
341    is_f64: bool,
342) -> PyResult<(PyObject, PyObject)> {
343    if is_f64 {
344        let eps_f64 = 1e-8f64;
345
346        let (credit_out, loss_contrib) = provider
347            .ilp_credit_forward_f64_launch(
348                d_row_offsets,
349                d_coo_cands,
350                cand_col,
351                d_is_positive,
352                num_facts,
353                eps_f64,
354            )
355            .map_err(|e| types::gpu_err("forward f64", e))?;
356
357        let d_total_loss = provider
358            .ilp_reduce_sum_f64_launch(&loss_contrib, num_facts)
359            .map_err(|e| types::gpu_err("reduce f64", e))?;
360
361        let d_grad = provider
362            .ilp_credit_backward_f64_launch(
363                d_row_offsets,
364                d_coo_cands,
365                &credit_out,
366                d_is_positive,
367                num_facts,
368                num_cands,
369            )
370            .map_err(|e| types::gpu_err("backward f64", e))?;
371
372        export_loss_grad_device(
373            provider,
374            py,
375            d_total_loss,
376            d_grad,
377            num_cands,
378            ScalarType::F64,
379        )
380    } else {
381        let eps_f32 = 1e-8f32;
382
383        let (credit_out, loss_contrib) = provider
384            .ilp_credit_forward_f32_launch(
385                d_row_offsets,
386                d_coo_cands,
387                cand_col,
388                d_is_positive,
389                num_facts,
390                eps_f32,
391            )
392            .map_err(|e| types::gpu_err("forward f32", e))?;
393
394        let d_total_loss = provider
395            .ilp_reduce_sum_f32_launch(&loss_contrib, num_facts)
396            .map_err(|e| types::gpu_err("reduce f32", e))?;
397
398        let d_grad = provider
399            .ilp_credit_backward_f32_launch(
400                d_row_offsets,
401                d_coo_cands,
402                &credit_out,
403                d_is_positive,
404                num_facts,
405                num_cands,
406            )
407            .map_err(|e| types::gpu_err("backward f32", e))?;
408
409        export_loss_grad_device(
410            provider,
411            py,
412            d_total_loss,
413            d_grad,
414            num_cands,
415            ScalarType::F32,
416        )
417    }
418}

424/// Export device-resident loss scalar and grad vector as DLPack capsules.
425///
426/// Handles both f32 and f64 scalar types. Pass `scalar_type` matching the
427/// element type `T` of the incoming slices.
428pub(crate) fn export_loss_grad_device<T: GpuScalar>(
429    provider: &CudaKernelProvider,
430    py: Python<'_>,
431    d_loss_val: TrackedCudaSlice<T>,
432    d_grad: TrackedCudaSlice<T>,
433    num_cands: u32,
434    scalar_type: ScalarType,
435) -> PyResult<(PyObject, PyObject)> {
436    let schema = Schema::new(vec![("col_0".to_string(), scalar_type)]);
437
438    let mut d_loss_nrows = provider
439        .memory()
440        .alloc::<u32>(1)
441        .map_err(|e| types::gpu_err("alloc", e))?;
442    provider
443        .device()
444        .inner()
445        .htod_sync_copy_into(&[1u32], &mut d_loss_nrows)
446        .map_err(|e| types::gpu_err("htod", e))?;
447
448    let loss_buf = xlog_cuda::CudaBuffer::from_columns(
449        vec![d_loss_val.into_bytes().into()],
450        1,
451        d_loss_nrows,
452        schema.clone(),
453    );
454    let loss_dl = provider
455        .to_dlpack_table(loss_buf)
456        .column(0)
457        .map_err(|e| types::gpu_err("DLPack loss", e))?;
458    let loss_capsule = dlpack_capsule_from_tensor(py, loss_dl)?;
459
460    let mut d_grad_nrows = provider
461        .memory()
462        .alloc::<u32>(1)
463        .map_err(|e| types::gpu_err("alloc", e))?;
464    provider
465        .device()
466        .inner()
467        .htod_sync_copy_into(&[num_cands], &mut d_grad_nrows)
468        .map_err(|e| types::gpu_err("htod", e))?;
469
470    let grad_buf = xlog_cuda::CudaBuffer::from_columns(
471        vec![d_grad.into_bytes().into()],
472        num_cands as u64,
473        d_grad_nrows,
474        schema,
475    );
476    let grad_dl = provider
477        .to_dlpack_table(grad_buf)
478        .column(0)
479        .map_err(|e| types::gpu_err("DLPack grad", e))?;
480    let grad_capsule = dlpack_capsule_from_tensor(py, grad_dl)?;
481
482    Ok((loss_capsule, grad_capsule))
