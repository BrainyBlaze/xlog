6/// Convert XlogError to PyErr (PyRuntimeError).
7///
8/// Cannot use From impl (orphan rule: From, XlogError, PyErr all foreign).
9pub(crate) fn xlog_err(e: impl std::fmt::Display) -> PyErr {
10    PyRuntimeError::new_err(e.to_string())
11}

23pub(crate) fn scalar_type_name(typ: &ScalarType) -> String {
24    match *typ {
25        ScalarType::U32 => "u32".to_string(),
26        ScalarType::U64 => "u64".to_string(),
27        ScalarType::I32 => "i32".to_string(),
28        ScalarType::I64 => "i64".to_string(),
29        ScalarType::F32 => "f32".to_string(),
30        ScalarType::F64 => "f64".to_string(),
31        ScalarType::Bool => "bool".to_string(),
32        ScalarType::Symbol => "symbol".to_string(),
33    }
34}

44/// Epsilon value for numerical stability in log computations.
45pub(crate) const NLL_EPSILON: f64 = 1e-38;
46
47/// Compute negative log-likelihood loss from probability.
48#[inline]
49pub(crate) fn nll_loss_value(probability: f64) -> f64 {
50    -(probability.max(NLL_EPSILON)).ln()
51}

53/// Create a PyTorch tensor from a scalar f64 value.
54pub(crate) fn create_torch_tensor(
55    py: pyo3::Python<'_>,
56    value: f64,
57) -> pyo3::PyResult<pyo3::PyObject> {
58    let torch = py.import("torch")?;
59    let tensor = torch.call_method1("tensor", (value,))?;
60    Ok(tensor.into())
