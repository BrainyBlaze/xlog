12/// Train a program for multiple epochs.
13///
14/// This is the main training entry point that runs the full training loop:
15/// - For each epoch: shuffle queries (optional), process batches, record stats
16/// - Supports learning rate scheduling via scheduler_step() after each epoch
17///   once at least one optimizer step has been performed
18///
19/// # Arguments
20/// * `program` - Compiled program with registered networks
21/// * `queries` - Training queries (e.g., ["addition(0, 1, 5)", "addition(2, 3, 7)")
22/// * `epochs` - Number of training epochs (default: 10)
23/// * `batch_size` - Number of queries per batch (default: 32)
24/// * `log_iter` - Log progress every N batches (default: 100)
25/// * `shuffle` - Whether to shuffle queries each epoch (default: true)
26///
27/// # Returns
28/// TrainingHistory with epoch and batch losses
29#[pyfunction]
30#[pyo3(signature = (program, queries, epochs=10, batch_size=32, log_iter=100, shuffle=true, max_grad_norm=None, val_queries=None, patience=None))]
31pub fn train_model(
32    py: Python<'_>,
33    program: &mut CompiledProgram,
34    queries: Vec<String>,
35    epochs: usize,
36    batch_size: usize,
37    log_iter: usize,
38    shuffle: bool,
39    max_grad_norm: Option<f64>,
40    val_queries: Option<Vec<String>>,
41    patience: Option<usize>,
42) -> PyResult<TrainingHistory> {
43    use rand::seq::SliceRandom;
44    use rand::thread_rng;
45    use std::time::Instant;
46
47    // Validate: val_queries and patience must both be present or both absent
48    match (&val_queries, &patience) {
49        (Some(_), None) | (None, Some(_)) => {
50            return Err(pyo3::exceptions::PyValueError::new_err(
51                "val_queries and patience must both be provided for early stopping",
52            ));
53        }
54        _ => {}
55    }
56
57    let mut history = TrainingHistory::new();
58    let mut best_val_loss = f64::INFINITY;
59    let mut epochs_without_improvement = 0usize;
60
61    for epoch in 0..epochs {
62        let mut epoch_queries = queries.clone();
63
64        if shuffle {
65            let mut rng = thread_rng();
66            epoch_queries.shuffle(&mut rng);
67        }
68
69        let epoch_start = Instant::now();
70        let stats = program.train_epoch_internal(
71            py,
72            &epoch_queries,
73            batch_size,
74            log_iter,
75            max_grad_norm,
76            &mut history,
77        )?;
78        history.add_epoch(stats.avg_loss, epoch_start.elapsed().as_secs_f64());
79
80        println!(
81            "Epoch {}/{}: avg_loss={:.6}",
82            epoch + 1,
83            epochs,
84            stats.avg_loss
85        );
86        use std::io::Write;
87        let _ = std::io::stdout().flush();
88
89        // Early stopping check
90        if let (Some(ref val_q), Some(pat)) = (&val_queries, patience) {
91            let val_loss = program.evaluate_loss(val_q.clone())?;
92            if val_loss < best_val_loss {
93                best_val_loss = val_loss;
94                epochs_without_improvement = 0;
95            } else {
96                epochs_without_improvement += 1;
97            }
98            if epochs_without_improvement >= pat {
99                history.stopped_early = true;
100                break;
101            }
102        }
103    }
104
105    Ok(history)
106}

108/// GPU-native training loop — no per-query host synchronization.

189impl TrainingHistory {
190    pub(super) fn new() -> Self {
191        Self {
192            epoch_losses: Vec::new(),
193            epoch_times: Vec::new(),
194            batch_losses: Vec::new(),
195            stopped_early: false,
196        }
197    }
198
199    pub(super) fn add_epoch(&mut self, loss: f64, epoch_time_sec: f64) {
200        self.epoch_losses.push(loss);
201        self.epoch_times.push(epoch_time_sec);
202    }
203
204    pub(super) fn add_batch(&mut self, loss: f64) {
205        self.batch_losses.push(loss);
206    }
207}
