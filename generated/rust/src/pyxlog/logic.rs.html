30    #[pyo3(signature = (source, device=0, memory_mb=32768, prob_engine=None))]
31    pub fn compile(
32        source: &str,
33        device: usize,
34        memory_mb: u64,
35        prob_engine: Option<String>,
36    ) -> PyResult<CompiledProgram> {
37        if memory_mb == 0 {
38            return Err(PyValueError::new_err("memory_mb must be > 0"));
39        }
40
41        let mut config = GpuConfig::default();
42        config.device_ordinal = device;
43        config.memory_bytes = memory_mb * 1024 * 1024;
44
45        // Parse the AST to get prob_engine and neural predicates
46        let ast = xlog_logic::parse_program(source).map_err(types::xlog_err)?;
47
48        // Extract declared neural network names
49        let declared_networks: HashSet<String> = ast
50            .neural_predicates
51            .iter()
52            .map(|np| np.network.clone())
53            .collect();
54        // Build by-network form index: network name -> is_embedding
55        let mut declared_network_forms: HashMap<String, bool> = HashMap::new();
56        for np in &ast.neural_predicates {
57            let is_embedding = np.labels.is_none();
58            match declared_network_forms.get(&np.network) {
59                Some(&existing_form) if existing_form != is_embedding => {
60                    return Err(PyValueError::new_err(format!(
61                        "network '{}' is declared as both classification and embedding; \
62                         each network name must have a single form",
63                        np.network
64                    )));
65                }
66                _ => {
67                    declared_network_forms.insert(np.network.clone(), is_embedding);
68                }
69            }
70        }
71
72        let neural_registry = NeuralPredicateRegistry::from_ast(&ast).map_err(types::val_err)?;
73
74        let engine = match prob_engine {
75            Some(s) => parse_prob_engine_override(&s)?,
76            None => ast.prob_engine(),
77        };
78
79        let program = match engine {
80            ProbEngine::ExactDdnnf => CompiledProbProgram::Exact(
81                ExactDdnnfProgram::compile_source_with_gpu(source, config)
82                    .map_err(types::xlog_err)?,
83            ),
84            ProbEngine::Mc => CompiledProbProgram::Mc(
85                McProgram::compile_source_with_gpu(source, config).map_err(types::xlog_err)?,
86            ),
87        };
88        let provider = provider_from_config(config).map_err(types::xlog_err)?;
89
90        Ok(CompiledProgram {
91            program,
92            output_provider: Arc::new(provider),
93            network_registry: NetworkRegistry::new(),
94            neural_registry,
95            declared_networks,
96            declared_network_forms,
97            tensor_sources: TensorSourceRegistry::new(),
98            domain_source: None,
99            domain_ids: Vec::new(),
100            _source: source.to_string(),
101            ast,
102            _gpu_config: config,
103            _prob_engine: engine,
104            query_signature_cache: StdHashMap::new(),
105            circuit_cache: StdHashMap::new(),
106            circuit_cache_hits: 0,
107            circuit_cache_misses: 0,
108            template_compile_count: 0,
109            batch_queries: true,
110            last_compile_profile: None,
111        })
112    }
113}

140    #[pyo3(signature = (dlpack_inputs=None, memory_mb=None))]
141    pub fn evaluate(
142        &self,
143        py: Python<'_>,
144        dlpack_inputs: Option<&Bound<'_, PyDict>>,
145        memory_mb: Option<u64>,
146    ) -> PyResult<LogicEvalResult> {
147        enforce_call_memory_limit(&self.provider, memory_mb)?;
148        let mut inputs: HashMap<String, xlog_cuda::CudaBuffer> = HashMap::new();
149
150        if let Some(dict) = dlpack_inputs {
151            for (k, v) in dict.iter() {
152                let name: String = k.extract()?;
153                let schema = self.program.schema(&name).ok_or_else(|| {
154                    PyValueError::new_err(format!(
155                        "Unknown input relation {} (not present in compiled schemas)",
156                        name
157                    ))
158                })?;
159
160                let tensors = collect_dlpack_columns(
161                    &v,
162                    &format!(
163                        "Input relation {} must be a sequence of DLPack columns",
164                        name
165                    ),
166                )?;
167
168                let buffer = self
169                    .provider
170                    .from_dlpack_tensors_with_schema(schema.clone(), tensors)
171                    .map_err(types::xlog_err)?;
172
173                inputs.insert(name, buffer);
174            }
175        }
176
177        let result = self
178            .program
179            .evaluate(self.provider.clone(), inputs)
180            .map_err(types::xlog_err)?;
181        pack_logic_result_with_provider(py, &self.provider, result)
182    }

220    pub fn put_relation(
221        &mut self,
222        name: String,
223        dlpack_columns: &Bound<'_, PyAny>,
224    ) -> PyResult<()> {
225        if name.starts_with("__") {
226            return Err(PyValueError::new_err(format!(
227                "Relation {} is internal and cannot be stored in a persistent session",
228                name
229            )));
230        }
231        let schema = self.program.schema(&name).ok_or_else(|| {
232            PyValueError::new_err(format!(
233                "Unknown relation {} (not present in compiled schemas)",
234                name
235            ))
236        })?;
237        let tensors = collect_dlpack_columns(
238            dlpack_columns,
239            &format!("Relation {} must be a sequence of DLPack columns", name),
240        )?;
241        let buffer = self
242            .provider
243            .from_dlpack_tensors_with_schema(schema.clone(), tensors)
244            .map_err(types::xlog_err)?;
245        self.relation_store.put(&name, buffer);
246        self.evaluation_store = None;
247        self.session_runtime = None;
248        self.last_delta_stats = None;
249        Ok(())
250    }

252    #[pyo3(signature = (memory_mb=None))]
253    pub fn evaluate(
254        &mut self,
255        py: Python<'_>,
256        memory_mb: Option<u64>,
257    ) -> PyResult<LogicEvalResult> {
258        enforce_call_memory_limit(&self.provider, memory_mb)?;
259        let result = if let Some(store) = &self.evaluation_store {
260            self.program
261                .evaluate_cached_relation_store(self.provider.clone(), store)
262                .map_err(types::xlog_err)?
263        } else {
264            if self.session_runtime.is_none() {
265                self.session_runtime = Some(
266                    self.program
267                        .create_session_runtime(self.provider.clone(), &self.relation_store, false)
268                        .map_err(types::xlog_err)?,
269                );
270            }
271            let runtime = self.session_runtime.as_mut().ok_or_else(|| {
272                PyRuntimeError::new_err("session runtime unavailable during evaluation")
273            })?;
274            let (result, store) = self
275                .program
276                .evaluate_with_session_runtime(self.provider.clone(), runtime)
277                .map_err(types::xlog_err)?;
278            self.evaluation_store = Some(store);
279            result
280        };
281        pack_logic_result_with_provider(py, &self.provider, result)
282    }

304    #[pyo3(signature = (name, insert_columns=None, delete_columns=None))]
305    pub fn apply_relation_delta(
306        &mut self,
307        py: Python<'_>,
308        name: String,
309        insert_columns: Option<&Bound<'_, PyAny>>,
310        delete_columns: Option<&Bound<'_, PyAny>>,
311    ) -> PyResult<PyObject> {
312        if insert_columns.is_none() && delete_columns.is_none() {
313            return Err(PyValueError::new_err(
314                "apply_relation_delta requires insert_columns, delete_columns, or both",
315            ));
316        }
317        let insert = insert_columns
318            .map(|columns| self.relation_delta_buffer(&name, columns))
319            .transpose()?;
320        let delete = delete_columns
321            .map(|columns| self.relation_delta_buffer(&name, columns))
322            .transpose()?;
323        self.apply_single_relation_delta(py, name, insert, delete)
324    }

350    #[pyo3(signature = (updates, check_equivalence=false))]
351    pub fn apply_relation_delta_debug(
352        &mut self,
353        py: Python<'_>,
354        updates: &Bound<'_, PyAny>,
355        check_equivalence: bool,
356    ) -> PyResult<PyObject> {
357        let (batch, relation_names) =
358            self.parse_relation_delta_batch("apply_relation_delta_debug", updates)?;
359        let delta_start = Instant::now();
360        let report = self
361            .program
362            .apply_relation_delta_batch_with_session_runtime(
363                self.provider.clone(),
364                &mut self.relation_store,
365                &mut self.evaluation_store,
366                &mut self.session_runtime,
367                batch,
368            )
369            .map_err(types::xlog_err)?;
370        let delta_micros = delta_start.elapsed().as_micros().max(1) as u64;
371        let mut stats = logic_delta_stats_from_report(report);
372        if check_equivalence {
373            let full_start = Instant::now();
374            let (_, full_store) = self
375                .program
376                .evaluate_with_relation_store_and_cache(
377                    self.provider.clone(),
378                    &self.relation_store,
379                    false,
380                )
381                .map_err(types::xlog_err)?;
382            let full_micros = full_start.elapsed().as_micros() as u64;
383            let cached_store = self.evaluation_store.as_ref().ok_or_else(|| {
384                PyRuntimeError::new_err("delta debug missing cached store after delta application")
385            })?;
386            stats.equivalent_to_full_recompute = Some(
387                self.program
388                    .relation_stores_query_equivalent(
389                        self.provider.as_ref(),
390                        &full_store,
391                        cached_store,
392                    )
393                    .map_err(types::xlog_err)?,
394            );
395            let speedup = full_micros as f64 / delta_micros as f64;
396            stats.planner_telemetry.measured_delta_speedup = Some(speedup);
397            if speedup >= 1.0 {
398                stats
399                    .planner_telemetry
400                    .planner_advice
401                    .push(format!("delta path is faster by {speedup:.2}x"));
402            } else {
403                stats.planner_telemetry.planner_advice.push(format!(
404                    "full recompute may be faster; delta measured {speedup:.2}x"
405                ));
406            }
407        }
408        self.last_delta_stats = Some(stats.clone());
409        self.fire_relation_callbacks(py, &relation_names, &stats)?;
410        pack_delta_stats(py, &stats)
411    }

433    pub fn register_relation_callback(
434        &mut self,
435        py: Python<'_>,
436        callback: PyObject,
437    ) -> PyResult<u64> {
438        if !callback.bind(py).is_callable() {
439            return Err(PyValueError::new_err(
440                "register_relation_callback expects a callable",
441            ));
442        }
443        let id = self.next_relation_callback_id;
444        self.next_relation_callback_id = self.next_relation_callback_id.saturating_add(1);
445        self.relation_callbacks
446            .push(RelationChangeCallback { id, callback });
447        Ok(id)
448    }

451        let before = self.relation_callbacks.len();
452        self.relation_callbacks

454        before != self.relation_callbacks.len()

584impl LogicRelationSession {
585    fn parse_relation_delta_batch(
586        &self,
587        method_name: &str,
588        updates: &Bound<'_, PyAny>,
589    ) -> PyResult<(Vec<(String, RelationDelta)>, Vec<String>)> {
590        let seq = updates.downcast::<PySequence>().map_err(|_| {
591            PyValueError::new_err(format!(
592                "{method_name} expects a sequence of update dictionaries"
593            ))
594        })?;
595        let mut batch: Vec<(String, RelationDelta)> = Vec::with_capacity(seq.len()? as usize);
596        let mut relation_names: Vec<String> = Vec::new();
597        for item in seq.try_iter()? {
598            let item = item?;
599            let dict = item.downcast::<PyDict>().map_err(|_| {
600                PyValueError::new_err(format!("{method_name} updates must be dictionaries"))
601            })?;
602            let name_obj = dict.get_item("name")?.ok_or_else(|| {
603                PyValueError::new_err(format!("{method_name} update missing 'name'"))
604            })?;
605            let name: String = name_obj.extract()?;
606            let insert = optional_delta_columns(dict, "insert_columns")
607                .map(|columns| self.relation_delta_buffer(&name, &columns))
608                .transpose()?;
609            let delete = optional_delta_columns(dict, "delete_columns")
610                .map(|columns| self.relation_delta_buffer(&name, &columns))
611                .transpose()?;
612            if insert.is_none() && delete.is_none() {
613                return Err(PyValueError::new_err(format!(
614                    "{method_name} updates require insert_columns, delete_columns, or both"
615                )));
616            }
617            relation_names.push(name.clone());
618            batch.push((name, RelationDelta::new(insert, delete)));
619        }
620        Ok((batch, relation_names))
621    }
622
623    fn relation_delta_buffer(
624        &self,
625        name: &str,
626        dlpack_columns: &Bound<'_, PyAny>,
627    ) -> PyResult<xlog_cuda::CudaBuffer> {
628        if name.starts_with("__") {
629            return Err(PyValueError::new_err(format!(
630                "Relation {} is internal and cannot be updated in a persistent session",
631                name
632            )));
633        }
634        let schema = self.program.schema(name).ok_or_else(|| {
635            PyValueError::new_err(format!(
636                "Unknown relation {} (not present in compiled schemas)",
637                name
638            ))
639        })?;
640        let tensors = collect_dlpack_columns(
641            dlpack_columns,
642            &format!(
643                "Relation {} delta must be a sequence of DLPack columns",
644                name
645            ),
646        )?;
647        self.provider
648            .from_dlpack_tensors_with_schema(schema.clone(), tensors)
649            .map_err(types::xlog_err)
650    }

652    fn apply_single_relation_delta(
653        &mut self,
654        py: Python<'_>,
655        name: String,
656        insert: Option<xlog_cuda::CudaBuffer>,
657        delete: Option<xlog_cuda::CudaBuffer>,
658    ) -> PyResult<PyObject> {
659        let relation_names = vec![name.clone()];
660        let mut deltas = HashMap::new();
661        deltas.insert(name, RelationDelta::new(insert, delete));
662        let report = self
663            .program
664            .apply_relation_deltas_with_session_runtime(
665                self.provider.clone(),
666                &mut self.relation_store,
667                &mut self.evaluation_store,
668                &mut self.session_runtime,
669                deltas,
670            )
671            .map_err(types::xlog_err)?;
672        let stats = LogicDeltaStats {
673            input_delta_count: report.input_delta_count,
674            changed_relations: report.changed_relations,
675            changed_relation_names: report.changed_relation_names,
676            insert_rows: report.insert_rows,
677            delete_rows: report.delete_rows,
678            has_deletes: report.has_deletes,
679            affected_sccs: report.affected_sccs,
680            recomputed_sccs: report.recomputed_sccs,
681            incremental_sccs: report.incremental_sccs,
682            coalesced_insert_rows: report.coalesced_insert_rows,
683            coalesced_delete_rows: report.coalesced_delete_rows,
684            canceled_rows: report.canceled_rows,
685            equivalent_to_full_recompute: None,
686            planner_telemetry: report.planner_telemetry,
687            debug_trace: report.debug_trace,
688        };
689        self.last_delta_stats = Some(stats.clone());
690        self.fire_relation_callbacks(py, &relation_names, &stats)?;
691        pack_delta_stats(py, &stats)
692    }
693
694    fn fire_relation_callbacks(
695        &mut self,
696        py: Python<'_>,
697        relation_names: &[String],
698        stats: &LogicDeltaStats,
699    ) -> PyResult<()> {
700        if self.relation_callbacks.is_empty() || stats.changed_relations == 0 {
701            return Ok(());
702        }
703
704        let mut seen = HashSet::new();
705        let mut events: Vec<(String, u64)> = Vec::new();
706        for relation in relation_names {
707            if seen.insert(relation.clone()) {
708                let generation = self
709                    .relation_generations
710                    .entry(relation.clone())
711                    .and_modify(|current| *current = current.saturating_add(1))
712                    .or_insert(1);
713                events.push((relation.clone(), *generation));
714            }
715        }
716
717        for (relation, generation) in events {
718            let payload = relation_callback_payload(py, &relation, generation, stats)?;
719            for registered in &self.relation_callbacks {
720                registered.callback.call1(py, (payload.clone_ref(py),))?;
721            }
722        }
723
724        Ok(())
725    }
726}

869fn collect_dlpack_columns(
870    obj: &Bound<'_, PyAny>,
871    type_error_message: &str,
872) -> PyResult<Vec<DlpackManagedTensor>> {
873    let seq = obj
874        .downcast::<PySequence>()
875        .map_err(|_| PyValueError::new_err(type_error_message.to_string()))?;
876
877    let mut tensors: Vec<DlpackManagedTensor> = Vec::with_capacity(seq.len()?);
878    for item in seq.try_iter()? {
879        let item = item?;
880        tensors.push(dlpack_from_py(&item)?);
881    }
882    Ok(tensors)
883}

885fn export_buffer_columns(
886    py: Python<'_>,
887    provider: &Arc<xlog_cuda::CudaKernelProvider>,
888    buffer: xlog_cuda::CudaBuffer,
889) -> PyResult<Vec<PyObject>> {
890    let arity = buffer.arity();
891    let table = provider.to_dlpack_table(buffer);
892    let mut tensors: Vec<PyObject> = Vec::with_capacity(arity);
893    for col_idx in 0..arity {
894        let tensor = table.column(col_idx).map_err(types::xlog_err)?;
895        tensors.push(dlpack_capsule_from_tensor(py, tensor)?);
896    }
897    Ok(tensors)
898}

900fn pack_logic_result_with_provider(
901    py: Python<'_>,
902    provider: &Arc<xlog_cuda::CudaKernelProvider>,
903    result: gpu_logic::LogicEvalResult,
904) -> PyResult<LogicEvalResult> {
905    let mut queries: Vec<Py<LogicQueryResult>> = Vec::with_capacity(result.queries.len());
906
907    for q in result.queries {
908        let num_rows = provider
909            .validated_logical_row_count(&q.buffer)
910            .map_err(types::xlog_err)?;
911        let is_true = q.columns.is_empty() && num_rows > 0;
912        let tensors = if q.columns.is_empty() {
913            Vec::new()
914        } else {
915            export_buffer_columns(py, provider, q.buffer)?
916        };
917
918        queries.push(Py::new(
919            py,
920            LogicQueryResult {
921                relation_name: q.relation_name,
922                columns: q.columns,
923                sort_labels: q.sort_labels,
924                tensors,
925                num_rows,
926                is_true,
927            },
928        )?);
929    }
930
931    Ok(LogicEvalResult { queries })
