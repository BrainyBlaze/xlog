17/// Export one (struct) Arrow C Device array from one or more DLPack columns (zero-copy).
18///
19/// EXPERIMENTAL: Requires building `pyxlog` with `--features arrow-device-import`.
20#[cfg(feature = "arrow-device-import")]
21#[pyfunction]
22#[pyo3(signature = (dlpack_columns, device=0, memory_mb=32768))]
23pub(crate) fn export_arrow_device(
24    py: Python<'_>,
25    dlpack_columns: &Bound<'_, PyAny>,
26    device: usize,
27    memory_mb: u64,
28) -> PyResult<PyObject> {
29    if memory_mb == 0 {
30        return Err(PyValueError::new_err("memory_mb must be > 0"));
31    }
32
33    let mut config = GpuConfig::default();
34    config.device_ordinal = device;
35    config.memory_bytes = memory_mb * 1024 * 1024;
36    let provider = provider_from_config(config).map_err(types::xlog_err)?;
37
38    let tensors = match dlpack_columns.downcast::<PySequence>() {
39        Ok(seq) => {
40            let mut out = Vec::with_capacity(seq.len()?);
41            for item in seq.try_iter()? {
42                out.push(dlpack_from_py(&item?)?);
43            }
44            out
45        }
46        Err(_) => vec![dlpack_from_py(dlpack_columns)?],
47    };
48
49    let buffer = provider
50        .from_dlpack_tensors(tensors)
51        .map_err(types::xlog_err)?;
52    let device_array = provider
53        .to_arrow_device_record_batch(buffer)
54        .map_err(types::xlog_err)?;
55    arrow_device_capsule_from_device_array(py, device_array)
56}

58/// Import an Arrow C Device array (device-resident) into DLPack columns (zero-copy).
59///
60/// EXPERIMENTAL: Requires building `pyxlog` with `--features arrow-device-import`.
61#[cfg(feature = "arrow-device-import")]
62#[pyfunction]
63#[pyo3(signature = (device_array, device=0, memory_mb=32768))]
64pub(crate) fn import_arrow_device(
65    py: Python<'_>,
66    device_array: &Bound<'_, PyAny>,
67    device: usize,
68    memory_mb: u64,
69) -> PyResult<(Vec<PyObject>, Vec<String>, usize)> {
70    if memory_mb == 0 {
71        return Err(PyValueError::new_err("memory_mb must be > 0"));
72    }
73
74    let mut config = GpuConfig::default();
75    config.device_ordinal = device;
76    config.memory_bytes = memory_mb * 1024 * 1024;
77    let provider = provider_from_config(config).map_err(types::xlog_err)?;
78
79    let device_array = arrow_device_from_py(device_array)?;
80
81    let buffer = provider
82        .from_arrow_device_record_batch(device_array)
83        .map_err(types::xlog_err)?;
84
85    let names: Vec<String> = buffer
86        .schema
87        .columns
88        .iter()
89        .map(|(n, _)| n.clone())
90        .collect();
91    let num_rows = provider
92        .validated_logical_row_count(&buffer)
93        .map_err(types::xlog_err)?;
94    let arity = buffer.arity();
95
96    let table = provider.to_dlpack_table(buffer);
97    let mut tensors: Vec<PyObject> = Vec::with_capacity(arity);
98    for col_idx in 0..arity {
99        let tensor = table.column(col_idx).map_err(types::xlog_err)?;
100        tensors.push(dlpack_capsule_from_tensor(py, tensor)?);
101    }
102
103    Ok((tensors, names, num_rows))
104}

106#[pyfunction]
107pub(crate) fn dlpack_roundtrip(
108    py: Python<'_>,
109    tensor: &Bound<'_, PyAny>,
110    device: usize,
111    memory_mb: u64,
112) -> PyResult<PyObject> {
113    if memory_mb == 0 {
114        return Err(PyValueError::new_err("memory_mb must be > 0"));
115    }
116    let mut config = GpuConfig::default();
117    config.device_ordinal = device;
118    config.memory_bytes = memory_mb * 1024 * 1024;
119    let provider = provider_from_config(config).map_err(types::xlog_err)?;
120    let managed = dlpack_from_py(tensor)?;
121    let buffer = provider
122        .from_dlpack_tensors(vec![managed])
123        .map_err(types::xlog_err)?;
124    let out = provider
125        .to_dlpack_table(buffer)
126        .column(0)
127        .map_err(types::xlog_err)?;
128    dlpack_capsule_from_tensor(py, out)
129}
