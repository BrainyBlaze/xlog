1//! Pre-kernel request classification — pure host-side helpers.
2//!
3//! Separates the numeric early-return logic (empty candidates, zero positives)
4//! from CUDA-dependent buffer validation so the dead-end cases can be unit
5//! tested without a GPU. Behaviorally mirrors the early returns in the
6//! Python reference (`crates/pyxlog/python/pyxlog/ilp/exact_induce.py`).
7
8use crate::types::ExactInductionResult;
9
10/// Request metadata extracted after buffer-level validation passes.
11/// Pure host-side; contains no CUDA types.
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13pub(crate) struct RequestMetadata {
14    pub candidate_count: u32,
15    pub positive_count: u32,
16    pub negative_count: u32,
17    pub k_per_topology: u32,
18}
19
20/// Outcome of classifying a request against trivial dead-ends.
21#[derive(Debug, Clone, PartialEq, Eq)]
22pub(crate) enum PreKernelOutcome {
23    /// Return this result directly — skip the scoring kernel.
24    TrivialEmpty(ExactInductionResult),
25    /// Proceed to the scoring kernel with this metadata.
26    Proceed(RequestMetadata),
27}
28
29/// Classify against trivial dead-ends, matching the Python reference's early
30/// returns:
31///   * no candidates   → all-zero `ExactInductionResult`
32///   * no positives    → result with zero positives + retained neg/candidate counts
33///   * otherwise       → proceed with the extracted metadata
34pub(crate) fn classify_request(meta: RequestMetadata) -> PreKernelOutcome {
35    if meta.candidate_count == 0 {
36        return PreKernelOutcome::TrivialEmpty(ExactInductionResult::default());
37    }
38    if meta.positive_count == 0 {
39        return PreKernelOutcome::TrivialEmpty(ExactInductionResult {
40            candidates: Vec::new(),
41            total_scored: 0,
42            candidate_count: meta.candidate_count,
43            positive_count: 0,
44            negative_count: meta.negative_count,
45        });
46    }
47    PreKernelOutcome::Proceed(meta)
48}

50#[cfg(test)]
51mod tests {
52    use super::*;
53
54    #[test]
55    fn classify_empty_candidates_returns_default_result() {
56        let meta = RequestMetadata {
57            candidate_count: 0,
58            positive_count: 10,
59            negative_count: 5,
60            k_per_topology: 2,
61        };
62        match classify_request(meta) {
63            PreKernelOutcome::TrivialEmpty(r) => {
64                // All-zero Default — pos/neg counts are discarded when
65                // there are no candidates, matching the Python reference.
66                assert!(r.candidates.is_empty());
67                assert_eq!(r.total_scored, 0);
68                assert_eq!(r.candidate_count, 0);
69                assert_eq!(r.positive_count, 0);
70                assert_eq!(r.negative_count, 0);
71            }
72            PreKernelOutcome::Proceed(_) => panic!("expected TrivialEmpty"),
73        }
74    }
75
76    #[test]
77    fn classify_zero_positives_preserves_neg_and_candidate_counts() {
78        let meta = RequestMetadata {
79            candidate_count: 3,
80            positive_count: 0,
81            negative_count: 7,
82            k_per_topology: 2,
83        };
84        match classify_request(meta) {
85            PreKernelOutcome::TrivialEmpty(r) => {
86                // When positives are zero the Python ref still reports the
87                // candidate and negative counts so callers can log "we saw
88                // facts but had nothing to score."
89                assert!(r.candidates.is_empty());
90                assert_eq!(r.total_scored, 0);
91                assert_eq!(r.candidate_count, 3);
92                assert_eq!(r.positive_count, 0);
93                assert_eq!(r.negative_count, 7);
94            }
95            PreKernelOutcome::Proceed(_) => panic!("expected TrivialEmpty"),
96        }
97    }
98
99    #[test]
100    fn classify_normal_case_returns_proceed_with_metadata() {
101        let meta = RequestMetadata {
102            candidate_count: 4,
103            positive_count: 12,
104            negative_count: 2,
105            k_per_topology: 2,
106        };
107        match classify_request(meta) {
108            PreKernelOutcome::Proceed(m) => assert_eq!(m, meta),
109            PreKernelOutcome::TrivialEmpty(_) => panic!("expected Proceed"),
110        }
111    }

127    #[test]
128    fn classify_empty_candidates_takes_precedence_over_zero_positives() {
129        // Both zero — candidates check fires first, giving all-zero result
130        // (the Python ref also returns the all-zero default here).
131        let meta = RequestMetadata {
132            candidate_count: 0,
133            positive_count: 0,
134            negative_count: 3,
135            k_per_topology: 2,

140                assert_eq!(r.negative_count, 0); // discarded, all-zero default
