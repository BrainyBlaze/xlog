1//! Result aggregation, statistics, and the **CPU oracle/debug** evaluation path.
2//!
3//! The statistics helpers (`binomial_estimate`, `normal_quantile`, …) are used by
4//! both the GPU-native and host paths. The host relation-store evaluation
5//! (`evaluate_program_inplace`, `evidence_satisfied`, `atom_holds`, …) backs only
6//! [`crate::mc::McProgram::evaluate_cpu`] — a seed-matched CPU oracle for
7//! validating GPU device counts. It is never part of the GPU-native hot loop and
8//! never zero-host/GPU-native release evidence.

42#[cfg(feature = "host-io")]
43impl McProgram {
44    pub(super) fn apply_sample_facts(
45        &self,
46        store: &mut HashMap<String, Relation>,
47        bits: &[u8],
48    ) -> Result<()> {
49        for pf in &self.prob_facts {
50            if bits.get(pf.var_idx).copied().unwrap_or(0) == 0 {
51                continue;
52            }
53            store
54                .entry(pf.atom.predicate.clone())
55                .or_default()
56                .insert_tuple(pf.atom.args.clone());
57        }
58
59        for ad in &self.annotated_disjunctions {
60            let mut selected: Option<usize> = None;
61            for (idx, &var_idx) in ad.decision_vars.iter().enumerate() {
62                if bits.get(var_idx).copied().unwrap_or(0) != 0 {
63                    selected = Some(idx);
64                    break;
65                }
66            }
67
68            let outcome = match selected {
69                Some(i) => Some(i),
70                None => {
71                    if ad.has_none {
72                        None
73                    } else {
74                        Some(ad.choices.len().saturating_sub(1))
75                    }
76                }
77            };
78
79            if let Some(outcome_idx) = outcome {
80                let atom = ad.choices.get(outcome_idx).ok_or_else(|| {
81                    XlogError::Compilation(
82                        "Annotated disjunction outcome index out of range".to_string(),
83                    )
84                })?;
85                store
86                    .entry(atom.predicate.clone())
87                    .or_default()
88                    .insert_tuple(atom.args.clone());
89            }
90        }
91
92        Ok(())
93    }
94}

96#[cfg(feature = "host-io")]
97pub(super) fn build_scc_plans(program: &Program) -> Result<Vec<SccPlan>> {
98    let graph = build_dependency_graph(program);
99    let sccs = find_sccs_for_lowering(&graph);
100
101    let mut rules_by_head: HashMap<String, Vec<Rule>> = HashMap::new();
102    for rule in program.proper_rules() {
103        rules_by_head
104            .entry(rule.head.predicate.clone())
105            .or_default()
106            .push(rule.clone());
107    }
108
109    let mut plans: Vec<SccPlan> = Vec::new();
110    for scc in sccs {
111        let mut scc_rules: Vec<Rule> = Vec::new();
112        for pred in &scc {
113            if let Some(rules) = rules_by_head.get(pred) {
114                scc_rules.extend(rules.iter().cloned());
115            }
116        }
117        if scc_rules.is_empty() {
118            continue;
119        }
120
121        let predicate_set: HashSet<String> = scc.iter().cloned().collect();
122
123        let nonmonotone = scc_rules.iter().any(|rule| {
124            rule.body.iter().any(|lit| match lit {
125                BodyLiteral::Negated(atom) => predicate_set.contains(&atom.predicate),
126                BodyLiteral::Positive(atom) if rule.has_aggregation() => {
127                    predicate_set.contains(&atom.predicate)
128                }
129                _ => false,
130            })
131        });
132
133        let kind = if nonmonotone {
134            SccKind::NonMonotone
135        } else if is_recursive_scc(&scc, &scc_rules) {
136            SccKind::MonotoneRecursive
137        } else {
138            SccKind::MonotoneNonRecursive
139        };
140
141        plans.push(SccPlan {
142            predicates: scc,
143            rules: scc_rules,
144            kind,
145        });
146    }
147
148    Ok(plans)
149}

151#[cfg(feature = "host-io")]
152fn is_recursive_scc(scc: &[String], rules: &[Rule]) -> bool {
153    if scc.len() > 1 {
154        return true;
155    }
156    let Some(only) = scc.first() else {
157        return false;
158    };
159    for rule in rules {
160        for lit in &rule.body {
161            if let BodyLiteral::Positive(atom) = lit {
162                if &atom.predicate == only {
163                    return true;
164                }
165            }
166        }
167    }
168    false
169}

171#[cfg(feature = "host-io")]
172pub(super) fn evaluate_program_inplace(
173    scc_plans: &[SccPlan],
174    store: &mut HashMap<String, Relation>,
175    max_nonmonotone_iterations: usize,
176) -> Result<EvalStats> {
177    let mut stats = EvalStats::default();
178
179    for plan in scc_plans {
180        match plan.kind {
181            SccKind::MonotoneNonRecursive => {
182                for rule in &plan.rules {
183                    let derived = eval_rule(rule, store, &HashMap::new(), None)?;
184                    let rel = store.entry(rule.head.predicate.clone()).or_default();
185                    for tuple in derived {
186                        rel.insert_tuple(tuple);
187                    }
188                }
189            }
190            SccKind::MonotoneRecursive => {
191                eval_monotone_recursive_scc(&plan.predicates, &plan.rules, store)?;
192            }
193            SccKind::NonMonotone => {
194                stats.nonmonotone_sccs += 1;
195                let (cycle, hit_limit) = eval_nonmonotone_scc(
196                    &plan.predicates,
197                    &plan.rules,
198                    store,
199                    max_nonmonotone_iterations,
200                )?;
201                if cycle {
202                    stats.nonmonotone_cycles += 1;
203                }
204                if hit_limit {
205                    stats.nonmonotone_iteration_limit_hits += 1;
206                }
207            }
208        }
209    }
210
211    Ok(stats)
212}

214#[cfg(feature = "host-io")]
215fn eval_monotone_recursive_scc(
216    scc: &[String],
217    rules: &[Rule],
218    store: &mut HashMap<String, Relation>,
219) -> Result<()> {
220    const MAX_ITERS: usize = 1024;
221
222    let scc_set: HashSet<&str> = scc.iter().map(|s| s.as_str()).collect();
223
224    let mut full: HashMap<String, Relation> = HashMap::new();
225    for pred in scc {
226        let rel = store.get(pred).cloned().unwrap_or_default();
227        full.insert(pred.clone(), rel);
228    }
229
230    let mut delta: HashMap<String, Relation> = HashMap::new();
231    for rule in rules {
232        let derived = eval_rule(rule, store, &full, None)?;
233        if derived.is_empty() {
234            continue;
235        }
236        let head = rule.head.predicate.clone();
237        let delta_rel = delta.entry(head.clone()).or_default();
238        let full_rel = full.entry(head).or_default();
239        for tuple in derived {
240            if full_rel.tuples.insert(tuple.clone()) {
241                delta_rel.insert_tuple(tuple);
242            }
243        }
244    }
245
246    let mut reached_fixpoint = false;
247    for _ in 0..MAX_ITERS {
248        let any_delta = delta.values().any(|r| !r.is_empty());
249        if !any_delta {
250            reached_fixpoint = true;
251            break;
252        }
253
254        let full_prev = full.clone();
255        let delta_prev = delta.clone();
256        delta.clear();
257
258        for rule in rules {
259            let body_indices: Vec<usize> = rule
260                .body
261                .iter()
262                .enumerate()
263                .filter_map(|(i, lit)| match lit {
264                    BodyLiteral::Positive(atom) if scc_set.contains(atom.predicate.as_str()) => {
265                        let non_empty = delta_prev
266                            .get(&atom.predicate)
267                            .map(|r| !r.is_empty())
268                            .unwrap_or(false);
269                        non_empty.then_some(i)
270                    }
271                    _ => None,
272                })
273                .collect();
274            if body_indices.is_empty() {
275                continue;
276            }
277
278            let mut derived_all: HashSet<Vec<Value>> = HashSet::new();
279            for idx in body_indices {
280                let derived = eval_rule(rule, store, &full_prev, Some((idx, &delta_prev)))?;
281                derived_all.extend(derived);
282            }
283            if derived_all.is_empty() {
284                continue;
285            }
286
287            let head = rule.head.predicate.clone();
288            let delta_rel = delta.entry(head.clone()).or_default();
289            let full_rel = full.entry(head).or_default();
290            for tuple in derived_all {
291                if full_rel.tuples.insert(tuple.clone()) {
292                    delta_rel.insert_tuple(tuple);
293                }
294            }
295        }
296    }
297
298    if !reached_fixpoint {
299        return Err(XlogError::Execution(format!(
300            "Monotone SCC fixpoint iteration limit ({}) exceeded for SCC {:?}",
301            MAX_ITERS, scc
302        )));
303    }
304
305    for (pred, rel) in full {
306        store.insert(pred, rel);
307    }
308    Ok(())
309}

311#[cfg(feature = "host-io")]
312fn eval_nonmonotone_scc(
313    scc: &[String],
314    rules: &[Rule],
315    store: &mut HashMap<String, Relation>,
316    max_iters: usize,
317) -> Result<(bool, bool)> {
318    let mut base: HashMap<String, Relation> = HashMap::new();
319    for pred in scc {
320        base.insert(pred.clone(), store.get(pred).cloned().unwrap_or_default());
321    }
322
323    let mut history: Vec<HashMap<String, Relation>> = Vec::new();
324    history.push(base.clone());
325
326    let mut seen: HashMap<u64, Vec<usize>> = HashMap::new();
327    seen.entry(hash_scc_state(&history[0])).or_default().push(0);
328
329    for _iter in 0..max_iters {
330        let current = history.last().expect("history non-empty").clone();
331
332        let mut next: HashMap<String, Relation> = base.clone();
333
334        for rule in rules {
335            let derived = eval_rule(rule, store, &current, None)?;
336            let rel = next.entry(rule.head.predicate.clone()).or_default();
337            for tuple in derived {
338                rel.insert_tuple(tuple);
339            }
340        }
341
342        if next == current {
343            for (pred, rel) in next {
344                store.insert(pred, rel);
345            }
346            return Ok((false, false));
347        }
348
349        let h = hash_scc_state(&next);
350        if let Some(candidates) = seen.get(&h) {
351            for &idx in candidates {
352                if history.get(idx) == Some(&next) {
353                    // Cycle detected: history[idx..] repeats.
354                    let cycle_states = &history[idx..];
355                    let final_state = intersect_states(cycle_states);
356                    for (pred, rel) in final_state {
357                        store.insert(pred, rel);
358                    }
359                    return Ok((true, false));
360                }
361            }
362        }
363
364        let next_index = history.len();
365        history.push(next);
366        seen.entry(h).or_default().push(next_index);
367    }
368
369    // Iteration budget exhausted: fall back to a conservative invariant set.
370    let final_state = intersect_states(&history);
371    for (pred, rel) in final_state {
372        store.insert(pred, rel);
373    }
374    Ok((false, true))
375}

377#[cfg(feature = "host-io")]
378fn intersect_states(states: &[HashMap<String, Relation>]) -> HashMap<String, Relation> {
379    let mut out: HashMap<String, Relation> = HashMap::new();
380    let Some(first) = states.first() else {
381        return out;
382    };
383
384    for (pred, rel0) in first {
385        let mut intersection: HashSet<Vec<Value>> = rel0.tuples.clone();
386        for state in &states[1..] {
387            if let Some(rel) = state.get(pred) {
388                intersection.retain(|t| rel.tuples.contains(t));
389            } else {
390                intersection.clear();
391                break;
392            }
393        }
394        out.insert(
395            pred.clone(),
396            Relation {
397                tuples: intersection,
398            },
399        );
400    }
401
402    out
403}

405#[cfg(feature = "host-io")]
406fn hash_scc_state(state: &HashMap<String, Relation>) -> u64 {
407    use std::collections::hash_map::DefaultHasher;
408    let mut hasher = DefaultHasher::new();
409
410    let mut preds: Vec<&String> = state.keys().collect();
411    preds.sort();
412    for pred in preds {
413        pred.hash(&mut hasher);
414        if let Some(rel) = state.get(pred) {
415            let mut tuple_hashes: Vec<u64> = Vec::with_capacity(rel.tuples.len());
416            for tuple in &rel.tuples {
417                let mut th = DefaultHasher::new();
418                Hash::hash(tuple, &mut th);
419                tuple_hashes.push(th.finish());
420            }
421            tuple_hashes.sort_unstable();
422            tuple_hashes.hash(&mut hasher);
423        }
424    }
425
426    hasher.finish()
427}

429/// Evaluate a single rule and produce derived head tuples.
430///
431/// `full_scc` is a per-SCC snapshot (used for SCC predicates), and `delta_scc` optionally provides
432/// a delta relation for a specific body literal index (semi-naive).
433#[cfg(feature = "host-io")]
434fn eval_rule(
435    rule: &Rule,
436    global: &HashMap<String, Relation>,
437    full_scc: &HashMap<String, Relation>,
438    delta_scc: Option<(usize, &HashMap<String, Relation>)>,
439) -> Result<Vec<Vec<Value>>> {
440    let mut states: Vec<HashMap<String, Value>> = vec![HashMap::new()];
441
442    for (idx, lit) in rule.body.iter().enumerate() {
443        let mut next_states: Vec<HashMap<String, Value>> = Vec::new();
444        match lit {
445            BodyLiteral::Positive(atom) => {
446                let rel = select_relation(atom, idx, global, full_scc, delta_scc)?;
447                for binding in states {
448                    for tuple in &rel.tuples {
449                        let mut binding2 = binding.clone();
450                        if unify_atom(atom, tuple, &mut binding2)? {
451                            next_states.push(binding2);
452                        }
453                    }
454                }
455            }
456            BodyLiteral::Negated(atom) => {
457                let rel = select_relation(atom, idx, global, full_scc, delta_scc)?;
458                for binding in states {
459                    if negated_atom_holds(atom, rel, &binding)? {
460                        next_states.push(binding);
461                    }
462                }
463            }
464            BodyLiteral::Epistemic(lit) => {
465                return Err(XlogError::UnsupportedEpistemicConstruct {
466                    construct: "host MC evaluation".to_string(),
467                    context: format!("{:?} {}({})", lit.op, lit.atom.predicate, lit.atom.arity()),
468                });
469            }
470            BodyLiteral::Comparison(cmp) => {
471                for binding in states {
472                    if eval_comparison(cmp.op, &cmp.left, &cmp.right, &binding)? {
473                        next_states.push(binding);
474                    }
475                }
476            }
477            BodyLiteral::IsExpr(is_expr) => {
478                for mut binding in states {
479                    if binding.contains_key(&is_expr.target) {
480                        return Err(XlogError::Compilation(format!(
481                            "Is-expression target {} is already bound",
482                            is_expr.target
483                        )));
484                    }
485                    let v = eval_arith_expr(&is_expr.expr, &binding)?;
486                    binding.insert(is_expr.target.clone(), v);
487                    next_states.push(binding);
488                }
489            }
490            BodyLiteral::Univ(_) => {
491                return Err(XlogError::Compilation(
492                    "univ literal was not normalized before host Monte Carlo result materialization"
493                        .to_string(),
494                ));
495            }
496        }
497        states = next_states;
498        if states.is_empty() {
499            break;
500        }
501    }
502
503    if rule.has_aggregation() {
504        eval_aggregate_head(&rule.head, states)
505    } else {
506        let mut out: HashSet<Vec<Value>> = HashSet::new();
507        for binding in states {
508            let head_tuple = materialize_head_non_aggregate(&rule.head, &binding)?;
509            out.insert(head_tuple);
510        }
511        Ok(out.into_iter().collect())
512    }
513}

515#[cfg(feature = "host-io")]
516fn select_relation<'a>(
517    atom: &Atom,
518    body_index: usize,
519    global: &'a HashMap<String, Relation>,
520    full_scc: &'a HashMap<String, Relation>,
521    delta_scc: Option<(usize, &'a HashMap<String, Relation>)>,
522) -> Result<&'a Relation> {
523    if let Some((delta_index, delta_map)) = delta_scc {
524        if delta_index == body_index {
525            return delta_map.get(&atom.predicate).ok_or_else(|| {
526                XlogError::Compilation(format!(
527                    "Missing delta relation for predicate {}",
528                    atom.predicate
529                ))
530            });
531        }
532    }
533    if let Some(rel) = full_scc.get(&atom.predicate) {
534        return Ok(rel);
535    }
536    global
537        .get(&atom.predicate)
538        .ok_or_else(|| XlogError::Compilation(format!("Unknown predicate {}", atom.predicate)))
539}

541#[cfg(feature = "host-io")]
542fn negated_atom_holds(
543    atom: &Atom,
544    rel: &Relation,
545    binding: &HashMap<String, Value>,
546) -> Result<bool> {
547    // Safety: all variables in a negated atom must be bound already (otherwise domain is unknown).
548    for term in &atom.terms {
549        if let Term::Variable(name) = term {
550            if !binding.contains_key(name) {
551                return Err(XlogError::UnsafeVariable(name.clone()));
552            }
553        }
554    }
555
556    for tuple in &rel.tuples {
557        if atom_matches_bound(atom, tuple, binding)? {
558            return Ok(false);
559        }
560    }
561    Ok(true)
562}

564#[cfg(feature = "host-io")]
565fn atom_matches_bound(
566    atom: &Atom,
567    tuple: &[Value],
568    binding: &HashMap<String, Value>,
569) -> Result<bool> {
570    if atom.terms.len() != tuple.len() {
571        return Err(XlogError::Compilation(format!(
572            "Arity mismatch for {}: atom has {}, tuple has {}",
573            atom.predicate,
574            atom.terms.len(),
575            tuple.len()
576        )));
577    }
578    for (term, value) in atom.terms.iter().zip(tuple.iter()) {
579        match term {
580            Term::Variable(name) => {
581                let Some(bound) = binding.get(name) else {
582                    return Err(XlogError::UnsafeVariable(name.clone()));
583                };
584                if bound != value {
585                    return Ok(false);
586                }
587            }
588            Term::Anonymous => {}
589            Term::Integer(_) | Term::Float(_) | Term::String(_) | Term::Symbol(_) => {
590                if &value_from_term(term)? != value {
591                    return Ok(false);
592                }
593            }
594            Term::Aggregate(AggExpr { .. }) => {
595                return Err(XlogError::Compilation(
596                    "Aggregate not allowed in body atom".to_string(),
597                ));
598            }
599            Term::List(_) => return Err(unsupported_mc_result_term_error("matching", "list")),
600            Term::Cons { .. } => return Err(unsupported_mc_result_term_error("matching", "cons")),
601            Term::Compound { .. } => {
602                return Err(unsupported_mc_result_term_error("matching", "compound"));
603            }
604            Term::PredRef(_) => {
605                return Err(unsupported_mc_result_term_error("matching", "predref"))
606            }
607        }
608    }
609    Ok(true)
610}

612#[cfg(feature = "host-io")]
613fn materialize_head_non_aggregate(
614    head: &Atom,
615    binding: &HashMap<String, Value>,
616) -> Result<Vec<Value>> {
617    let mut out = Vec::with_capacity(head.terms.len());
618    for term in &head.terms {
619        match term {
620            Term::Variable(name) => {
621                let v = binding.get(name).ok_or_else(|| {
622                    XlogError::Compilation(format!(
623                        "Unbound head variable {} in {}",
624                        name, head.predicate
625                    ))
626                })?;
627                out.push(v.clone());
628            }
629            Term::Anonymous => {
630                return Err(XlogError::Compilation(format!(
631                    "Anonymous wildcard '_' not allowed in rule head (predicate {})",
632                    head.predicate
633                )));
634            }
635            Term::Integer(_) | Term::Float(_) | Term::String(_) | Term::Symbol(_) => {
636                out.push(value_from_term(term)?);
637            }
638            Term::Aggregate(_) => {
639                return Err(XlogError::Compilation(
640                    "Aggregate term in non-aggregate rule head".to_string(),
641                ));
642            }
643            Term::List(_) => {
644                return Err(unsupported_mc_result_term_error(
645                    "non-aggregate head materialization",
646                    "list",
647                ));
648            }
649            Term::Cons { .. } => {
650                return Err(unsupported_mc_result_term_error(
651                    "non-aggregate head materialization",
652                    "cons",
653                ));
654            }
655            Term::Compound { .. } => {
656                return Err(unsupported_mc_result_term_error(
657                    "non-aggregate head materialization",
658                    "compound",
659                ));
660            }
661            Term::PredRef(_) => {
662                return Err(unsupported_mc_result_term_error(
663                    "non-aggregate head materialization",
664                    "predref",
665                ));
666            }
667        }
668    }
669    Ok(out)
670}

672#[cfg(feature = "host-io")]
673fn eval_aggregate_head(
674    head: &Atom,
675    states: Vec<HashMap<String, Value>>,
676) -> Result<Vec<Vec<Value>>> {
677    // Collect unique group keys (variables) in head order.
678    let mut key_vars: Vec<String> = Vec::new();
679    let mut key_var_to_pos: HashMap<String, usize> = HashMap::new();
680
681    // Collect unique aggregate specs (op, var) in head order.
682    let mut agg_specs: Vec<(AggOp, String)> = Vec::new();
683    let mut agg_to_pos: HashMap<(AggOp, String), usize> = HashMap::new();
684
685    for term in &head.terms {
686        match term {
687            Term::Variable(name) => {
688                if !key_var_to_pos.contains_key(name) {
689                    let pos = key_vars.len();
690                    key_vars.push(name.clone());
691                    key_var_to_pos.insert(name.clone(), pos);
692                }
693            }
694            Term::Aggregate(agg) => {
695                let key = (agg.op, agg.variable.clone());
696                if let std::collections::hash_map::Entry::Vacant(entry) =
697                    agg_to_pos.entry(key.clone())
698                {
699                    let pos = agg_specs.len();
700                    agg_specs.push(key);
701                    entry.insert(pos);
702                }
703            }
704            Term::Anonymous => {
705                return Err(XlogError::Compilation(
706                    "Anonymous wildcard '_' not allowed in aggregate rule head".to_string(),
707                ));
708            }
709            Term::Integer(_) | Term::Float(_) | Term::String(_) | Term::Symbol(_) => {}
710            Term::List(_) => {
711                return Err(unsupported_mc_result_term_error(
712                    "aggregate head planning",
713                    "list",
714                ));
715            }
716            Term::Cons { .. } => {
717                return Err(unsupported_mc_result_term_error(
718                    "aggregate head planning",
719                    "cons",
720                ));
721            }
722            Term::Compound { .. } => {
723                return Err(unsupported_mc_result_term_error(
724                    "aggregate head planning",
725                    "compound",
726                ));
727            }
728            Term::PredRef(_) => {
729                return Err(unsupported_mc_result_term_error(
730                    "aggregate head planning",
731                    "predref",
732                ));
733            }
734        }
735    }
736
737    #[derive(Debug)]
738    struct GroupState {
739        key: Vec<Value>,
740        aggs: Vec<AggState>,
741    }
742
743    let mut groups: HashMap<Vec<Value>, GroupState> = HashMap::new();
744
745    for binding in states {
746        let mut key: Vec<Value> = Vec::with_capacity(key_vars.len());
747        for name in &key_vars {
748            let v = binding
749                .get(name)
750                .ok_or_else(|| XlogError::UnsafeVariable(name.clone()))?;
751            key.push(v.clone());
752        }
753
754        let entry = groups.entry(key.clone()).or_insert_with(|| GroupState {
755            key,
756            aggs: agg_specs.iter().map(|(op, _)| AggState::new(*op)).collect(),
757        });
758
759        for (idx, (op, var)) in agg_specs.iter().enumerate() {
760            let v = binding
761                .get(var)
762                .ok_or_else(|| XlogError::UnsafeVariable(var.clone()))?;
763            entry.aggs[idx].update(*op, v)?;
764        }
765    }
766
767    let mut out: Vec<Vec<Value>> = Vec::with_capacity(groups.len());
768    for group in groups.values() {
769        let mut tuple: Vec<Value> = Vec::with_capacity(head.terms.len());
770        for term in &head.terms {
771            match term {
772                Term::Variable(name) => {
773                    let pos = *key_var_to_pos.get(name).ok_or_else(|| {
774                        XlogError::Compilation(format!(
775                            "Aggregate head variable {} is not a group key",
776                            name
777                        ))
778                    })?;
779                    tuple.push(group.key[pos].clone());
780                }
781                Term::Aggregate(AggExpr { op, variable }) => {
782                    let idx = *agg_to_pos
783                        .get(&(*op, variable.clone()))
784                        .expect("agg_to_pos missing");
785                    tuple.push(group.aggs[idx].finish(*op)?);
786                }
787                Term::Integer(_) | Term::Float(_) | Term::String(_) | Term::Symbol(_) => {
788                    tuple.push(value_from_term(term)?);
789                }
790                Term::Anonymous => unreachable!(),
791                Term::List(_) => {
792                    return Err(unsupported_mc_result_term_error(
793                        "aggregate head materialization",
794                        "list",
795                    ));
796                }
797                Term::Cons { .. } => {
798                    return Err(unsupported_mc_result_term_error(
799                        "aggregate head materialization",
800                        "cons",
801                    ));
802                }
803                Term::Compound { .. } => {
804                    return Err(unsupported_mc_result_term_error(
805                        "aggregate head materialization",
806                        "compound",
807                    ));
808                }
809                Term::PredRef(_) => {
810                    return Err(unsupported_mc_result_term_error(
811                        "aggregate head materialization",
812                        "predref",
813                    ));
814                }
815            }
816        }
817        out.push(tuple);
818    }
819
820    Ok(out)
821}

824pub(super) fn atom_holds(store: &HashMap<String, Relation>, atom: &GroundAtom) -> bool {
825    store
826        .get(&atom.predicate)
827        .map(|rel| rel.contains(&atom.args))
828        .unwrap_or(false)
829}

831#[cfg(feature = "host-io")]
832pub(super) fn evidence_satisfied(
833    store: &HashMap<String, Relation>,
834    evidence: &[(GroundAtom, bool)],
835) -> bool {
836    for (atom, value) in evidence {
837        let holds = atom_holds(store, atom);
838        if holds != *value {
839            return false;
840        }
841    }
842    true
843}

845#[cfg(feature = "host-io")]
846pub(super) fn binomial_estimate(k: usize, n: usize, z: f64) -> (f64, f64, f64, f64) {
847    if n == 0 {
848        return (0.0, 0.0, 0.0, 0.0);
849    }
850    let n_f = n as f64;
851    let p = (k as f64) / n_f;
852    let stderr = (p * (1.0 - p) / n_f).sqrt();
853
854    let z2 = z * z;
855    let denom = 1.0 + z2 / n_f;
856    let center = (p + z2 / (2.0 * n_f)) / denom;
857    let half = z * ((p * (1.0 - p) / n_f) + (z2 / (4.0 * n_f * n_f))).sqrt() / denom;
858
859    let mut lo = center - half;
860    let mut hi = center + half;
861    if lo < 0.0 {
862        lo = 0.0;
863    }
864    if hi > 1.0 {
865        hi = 1.0;
866    }
867    (p, stderr, lo, hi)
868}

870// Acklam's inverse normal CDF approximation.
871#[cfg(feature = "host-io")]
872pub(super) fn normal_quantile(p: f64) -> f64 {
873    if !(0.0 < p && p < 1.0) || p.is_nan() {
874        return f64::NAN;
875    }
876
877    const A: [f64; 6] = [
878        -3.969683028665376e+01,
879        2.209460984245205e+02,
880        -2.759285104469687e+02,
881        1.383_577_518_672_69e2,
882        -3.066479806614716e+01,
883        2.506628277459239e+00,
884    ];
885    const B: [f64; 5] = [
886        -5.447609879822406e+01,
887        1.615858368580409e+02,
888        -1.556989798598866e+02,
889        6.680131188771972e+01,
890        -1.328068155288572e+01,
891    ];
892    const C: [f64; 6] = [
893        -7.784894002430293e-03,
894        -3.223964580411365e-01,
895        -2.400758277161838e+00,
896        -2.549732539343734e+00,
897        4.374664141464968e+00,
898        2.938163982698783e+00,
899    ];
900    const D: [f64; 4] = [
901        7.784695709041462e-03,
902        3.224671290700398e-01,
903        2.445134137142996e+00,
904        3.754408661907416e+00,
905    ];
906
907    const P_LOW: f64 = 0.02425;
908    const P_HIGH: f64 = 1.0 - P_LOW;
909
910    if p < P_LOW {
911        let q = (-2.0 * p.ln()).sqrt();
912        return (((((C[0] * q + C[1]) * q + C[2]) * q + C[3]) * q + C[4]) * q + C[5])
913            / ((((D[0] * q + D[1]) * q + D[2]) * q + D[3]) * q + 1.0);
914    }
915    if p > P_HIGH {
916        let q = (-2.0 * (1.0 - p).ln()).sqrt();
917        return -(((((C[0] * q + C[1]) * q + C[2]) * q + C[3]) * q + C[4]) * q + C[5])
918            / ((((D[0] * q + D[1]) * q + D[2]) * q + D[3]) * q + 1.0);
919    }
920
921    let q = p - 0.5;
922    let r = q * q;
923    (((((A[0] * r + A[1]) * r + A[2]) * r + A[3]) * r + A[4]) * r + A[5]) * q
924        / (((((B[0] * r + B[1]) * r + B[2]) * r + B[3]) * r + B[4]) * r + 1.0)
925}
