1use std::collections::HashMap;
2use xlog_core::RelId;
3
4/// Manages delta relation lifecycle during semi-naive fixpoint iteration.
5///
6/// Each recursive predicate gets a synthetic delta relation (with a unique RelId
7/// and store name). The tracker maps predicate names to their delta identifiers,
8/// tracks which predicates produced new tuples in the current iteration (convergence),
9/// and provides cleanup of synthetic relations on completion.
10///
11/// The tracker does NOT own CudaBuffers — those live in the executor's RelationStore.
12/// It owns the identity mapping and convergence state that was previously scattered
13/// inline across execute_recursive_scc().
14pub(crate) struct DeltaRelationTracker {
15    /// Maps predicate name → (delta RelId, delta store name)
16    entries: HashMap<String, (RelId, String)>,
17    /// Whether any predicate produced new tuples in the current iteration
18    any_changed: bool,
19}
20
21impl DeltaRelationTracker {
22    pub(crate) fn new() -> Self {
23        Self {
24            entries: HashMap::new(),
25            any_changed: false,
26        }
27    }
28
29    /// Register a delta relation for a recursive predicate.
30    pub(crate) fn insert(&mut self, pred: String, rel_id: RelId, store_name: String) {
31        self.entries.insert(pred, (rel_id, store_name));
32    }
33
34    /// Look up the delta (RelId, store_name) for a predicate.
35    pub(crate) fn get(&self, pred: &str) -> Option<&(RelId, String)> {
36        self.entries.get(pred)
37    }
38
39    /// Look up the delta store name for a predicate, returning an error if missing.
40    pub(crate) fn delta_name(&self, pred: &str) -> xlog_core::Result<&str> {
41        self.entries
42            .get(pred)
43            .map(|(_rel_id, name)| name.as_str())
44            .ok_or_else(|| {
45                xlog_core::XlogError::execution_ctx("delta_name", "missing delta relation", &pred)
46            })
47    }
48
49    /// Look up the delta RelId for a predicate, returning an error if missing.
50    pub(crate) fn delta_rel_id(&self, pred: &str) -> xlog_core::Result<RelId> {
51        self.entries
52            .get(pred)
53            .map(|(rel_id, _name)| *rel_id)
54            .ok_or_else(|| {
55                xlog_core::XlogError::execution_ctx("delta_rel_id", "missing delta relation", &pred)
56            })
57    }
58
59    /// Reset convergence state at the start of a new iteration.
60    pub(crate) fn begin_iteration(&mut self) {
61        self.any_changed = false;
62    }
63
64    /// Record that a predicate produced new tuples (non-empty delta).
65    pub(crate) fn mark_changed(&mut self) {
66        self.any_changed = true;
67    }
68
69    /// Returns true if no predicate produced new tuples in this iteration (fixpoint reached).
70    pub(crate) fn is_converged(&self) -> bool {
71        !self.any_changed
72    }
73
74    /// Consume the tracker, yielding owned entries for cleanup.
75    pub(crate) fn into_inner(self) -> HashMap<String, (RelId, String)> {
76        self.entries
77    }
78}
