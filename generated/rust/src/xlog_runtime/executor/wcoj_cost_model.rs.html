15/// Free-Join atom processing-order decision (decline-or-reorder).
16///
17/// The Free Join engine materializes a left-deep prefix; because each probe's
18/// keys must be a leading *column* prefix of its atom, it cannot reorder a
19/// chain to start from a selective tail the way the binary fallback can. On an
20/// adversarial chain this forces a large intermediate even when the result is
21/// tiny (a measured worst case is roughly a 3x peak-memory loss vs binary).
22/// This decision lets the dispatcher rescue or decline that case.
23pub(super) enum FjOrderDecision {
24    /// Keep the default traversal order. Returned whenever the current order
25    /// is already estimated within tolerance of the binary plan (so every
26    /// winning fixture is untouched) or the planner is inactive — i.e. the
27    /// dispatch behaves exactly as it did before the order planner existed
28    /// (fail-OPEN).
29    KeepDefault,
30    /// Dispatch Free Join with this atom processing order (original input
31    /// indices). Prefix-key-joinable by construction.
32    Reorder(Vec<usize>),
33    /// No prefix-key-joinable order is estimated within tolerance of the
34    /// unconstrained binary plan's peak — decline FJ so the binary fallback
35    /// runs (fail-CLOSED).
36    Decline,
37}
38
39/// Free Join is reordered/declined only when the current order's estimated
40/// peak exceeds the binary plan's by more than this ratio (6/5 = 1.2×).
41/// Integer math (`5·a ≤ 6·b` ⇔ `a ≤ 1.2·b`) avoids float comparison. Mirrors
42/// the adversarial-chain peak-memory acceptance test in the integration suite.
43const FJ_PEAK_TOLERANCE_NUM: u128 = 6;
44const FJ_PEAK_TOLERANCE_DEN: u128 = 5;
45
46/// `true` iff `peak ≤ 1.2 × baseline` (within tolerance — FJ is acceptable).
47fn within_peak_tolerance(peak: u64, baseline: u64) -> bool {
48    (peak as u128) * FJ_PEAK_TOLERANCE_DEN <= (baseline as u128) * FJ_PEAK_TOLERANCE_NUM
49}
50
51/// Default selectivity StatsManager applies when no cached selectivity /
52/// column-distinct stats exist (10%). Reused for the statless local estimate
53/// so the planner works for single-rule base-relation joins (where scans
54/// never populate `StatsManager`), consistent with the authoritative path.
55const DEFAULT_JOIN_SELECTIVITY_DEN: u128 = 10;

57pub(super) trait WcojCostModel: Send + Sync {
58    fn should_dispatch_triangle(&self, ctx: &WcojDispatchCtx) -> bool;
59    fn should_dispatch_4cycle(&self, ctx: &WcojDispatchCtx) -> bool;
60
61    /// Fail-OPEN loss-region veto for the general factorized routes
62    /// (aggregate-fused WCOJ over a triangle; Free Join). Returns
63    /// `true` ONLY when the model has full cardinality stats for every
64    /// slot relation AND the largest one is below the WCOJ-worthwhile
65    /// threshold — i.e. the join is provably small, no intermediate can
66    /// blow up, and the binary fallback wins (the measured 1.7–2.0×
67    /// cost-of-generality region for Free Join, and the small-triangle
68    /// region the base triangle cost model already declines).
69    ///
70    /// Deliberately conservative: stats absent for ANY slot, or ANY slot
71    /// large → `false` (no veto). This NEVER vetoes a case with a large
72    /// input (where factorized can win on a large avoided intermediate)
73    /// and NEVER vetoes when stats are unavailable (e.g. recursive deltas
74    /// on early iterations) — so every measured factorized gate win is
75    /// preserved exactly; the veto only removes provably-small losses.
76    fn factorized_loss_veto(&self, ctx: &WcojDispatchCtx) -> bool {
77        let _ = ctx;
78        false
79    }
80
81    /// Free Join order planner (decline-or-reorder). Acts only as a safety
82    /// net: when the current traversal order is already estimated within
83    /// tolerance of the binary plan it returns [`FjOrderDecision::KeepDefault`]
84    /// (every winning fixture untouched); only when the current order is
85    /// estimated to LOSE does it search for a better prefix-key-joinable order,
86    /// reordering to it ([`FjOrderDecision::Reorder`]) or declining to the
87    /// binary fallback ([`FjOrderDecision::Decline`]) if none is competitive.
88    ///
89    /// `atom_vars[i]` = dense join-variable ids per column of input `i` (column
90    /// order preserved — the prefix-key constraint keys on it); `cards[i]` =
91    /// that input's ground-truth row count (from the buffer the dispatcher is
92    /// about to join, NOT `StatsManager` — so it is always available even for
93    /// single-rule base-relation joins); `ctx.slot_rels[i]` = its relation.
94    /// All three index spaces coincide. Per-pair join estimates use
95    /// `StatsManager::estimate_join_cardinality` when stats are populated.
96    ///
97    /// Default: [`FjOrderDecision::KeepDefault`] — only `CardinalityAware`
98    /// plans (so the `SkewClassifier` opt-out also disables the reorder).
99    fn plan_free_join_order(
100        &self,
101        ctx: &WcojDispatchCtx,
102        atom_vars: &[Vec<usize>],
103        cards: &[u64],
104    ) -> FjOrderDecision {
105        let _ = (ctx, atom_vars, cards);
106        FjOrderDecision::KeepDefault
107    }
108}

126pub(super) struct CardinalityAwareCostModel {
127    min_binary_intermediate: u64,
128    large_binary_intermediate: u64,
129}

131impl Default for CardinalityAwareCostModel {
132    fn default() -> Self {
133        Self {
134            min_binary_intermediate: MIN_CARDINALITY_BINARY_INTERMEDIATE,
135            large_binary_intermediate: LARGE_CARDINALITY_BINARY_INTERMEDIATE,
136        }
137    }
138}

140impl CardinalityAwareCostModel {
141    fn populated_cards(&self, ctx: &WcojDispatchCtx) -> Option<Vec<u64>> {
142        ctx.slot_rels
143            .iter()
144            .map(|r| {
145                ctx.stats
146                    .get_relation_stats(*r)
147                    .map(|s| s.cardinality)
148                    .filter(|c| *c > 0)
149            })
150            .collect()
151    }
152
153    fn decide_from_cardinality(&self, binary_est: u64) -> bool {
154        binary_est >= self.large_binary_intermediate || binary_est >= self.min_binary_intermediate
155    }
156}

158impl WcojCostModel for CardinalityAwareCostModel {
159    fn should_dispatch_triangle(&self, ctx: &WcojDispatchCtx) -> bool {
160        debug_assert_eq!(
161            ctx.slot_rels.len(),
162            3,
163            "triangle ctx must carry exactly 3 slot relations"
164        );
165        if self.populated_cards(ctx).is_none() {
166            return false;
167        }
168        let binary_est =
169            ctx.stats
170                .estimate_join_cardinality(ctx.slot_rels[0], ctx.slot_rels[1], &[1], &[0]);
171        self.decide_from_cardinality(binary_est)
172    }
173
174    fn should_dispatch_4cycle(&self, ctx: &WcojDispatchCtx) -> bool {
175        debug_assert_eq!(
176            ctx.slot_rels.len(),
177            4,
178            "4-cycle ctx must carry exactly 4 slot relations"
179        );
180        if self.populated_cards(ctx).is_none() {
181            return false;
182        }
183        let binary_est =
184            ctx.stats
185                .estimate_join_cardinality(ctx.slot_rels[0], ctx.slot_rels[1], &[1], &[0]);
186        self.decide_from_cardinality(binary_est)
187    }
188
189    fn factorized_loss_veto(&self, ctx: &WcojDispatchCtx) -> bool {
190        // Fail-open: need every slot's cardinality to make any claim.
191        let cards = match self.populated_cards(ctx) {
192            Some(c) => c,
193            None => return false,
194        };
195        // Veto only when the LARGEST input is below the WCOJ-worthwhile
196        // threshold: then every join intermediate is bounded small, the
197        // binary plan is cheap, and the factorized route's overhead is
198        // not justified. Any large input → no veto (factorized may win
199        // on the avoided large intermediate).
200        cards.iter().copied().max().unwrap_or(0) < self.min_binary_intermediate
201    }
202
203    fn plan_free_join_order(
204        &self,
205        ctx: &WcojDispatchCtx,
206        atom_vars: &[Vec<usize>],
207        cards: &[u64],
208    ) -> FjOrderDecision {
209        let n = atom_vars.len();
210        // Defensive: index spaces must coincide; <3 atoms never reach here.
211        if n < 3 || cards.len() != n || ctx.slot_rels.len() != n {
212            return FjOrderDecision::KeepDefault;
213        }
214        // Binary fallback's freedom: best estimated peak over ANY connected
215        // order (no prefix-key constraint). The yardstick both the current
216        // and any candidate FJ order are judged against.
217        let binary_peak = match best_greedy_peak(ctx, atom_vars, cards, false) {
218            Some((_, p)) => p,
219            // No connected order at all (degenerate) → don't intervene.
220            None => return FjOrderDecision::KeepDefault,
221        };
222        // Safety-net gate: if the CURRENT traversal order is already within
223        // tolerance, keep it — guarantees every winning fixture (whose
224        // current order is good) is byte-for-byte untouched.
225        let traversal: Vec<usize> = (0..n).collect();
226        match eval_order_peak(ctx, atom_vars, cards, &traversal, true) {
227            Some(default_peak) => {
228                // Absolute floor: a small intermediate is cheap regardless of
229                // order — never intervene (the loss veto's small-join
230                // territory; FJ fires exactly as before). Only LARGE
231                // intermediates can carry the order-loss the planner removes.
232                // Also avoids spurious declines from ratio noise on tiny cards.
233                if default_peak < self.min_binary_intermediate
234                    || within_peak_tolerance(default_peak, binary_peak)
235                {
236                    return FjOrderDecision::KeepDefault;
237                }
238            }
239            None => {
240                // Traversal order is not prefix-key-joinable (FJ already
241                // declines to binary). Only attempt a rescue when the join is
242                // large enough to matter; otherwise keep the original behavior.
243                if binary_peak < self.min_binary_intermediate {
244                    return FjOrderDecision::KeepDefault;
245                }
246            }
247        }
248        // Current order loses (or is not prefix-key-valid) on a large join.
249        // Try the best prefix-key-joinable order; reorder to it if competitive,
250        // else decline to the binary fallback.
251        match best_greedy_peak(ctx, atom_vars, cards, true) {
252            Some((order, fj_peak)) if within_peak_tolerance(fj_peak, binary_peak) => {
253                FjOrderDecision::Reorder(order)
254            }
255            _ => FjOrderDecision::Decline,
256        }
257    }
258}

267// ---------------------------------------------------------------------------
268// Free Join order-planning helpers (decline-or-reorder).
269//
270// A left-deep prefix model: starting from a leader relation, atoms are added
271// one at a time; the running intermediate size is multiplied by the per-tuple
272// fan-out of each added atom (independence assumption), and the planner tracks
273// the peak running size. `constrained = true` enforces FJ's rule that an
274// atom's already-bound vars form a leading COLUMN prefix (so it can be probed
275// as a key); `constrained = false` is the binary fallback's freedom to join on
276// any shared key. The constrained/unconstrained peak gap is exactly the
277// order-loss the adversarial-chain test measures.
278// ---------------------------------------------------------------------------
279
280/// Best (lowest estimated peak) order over all leaders. Returns `(order,
281/// peak)`, or `None` if no complete connected (and, when constrained,
282/// prefix-key-joinable) order exists from any leader.
283fn best_greedy_peak(
284    ctx: &WcojDispatchCtx,
285    atom_vars: &[Vec<usize>],
286    cards: &[u64],
287    constrained: bool,
288) -> Option<(Vec<usize>, u64)> {
289    let n = atom_vars.len();
290    let mut best: Option<(Vec<usize>, u64)> = None;
291    for leader in 0..n {
292        if let Some((order, peak)) = greedy_from_leader(ctx, atom_vars, cards, constrained, leader)
293        {
294            if best.as_ref().map_or(true, |(_, bp)| peak < *bp) {
295                best = Some((order, peak));
296            }
297        }
298    }
299    best
300}
301
302/// Greedy left-deep order from a fixed leader: repeatedly append the
303/// connected (and, when constrained, prefix-key-joinable) not-yet-placed atom
304/// that minimizes the running intermediate estimate. Returns `(order, peak)`
305/// or `None` if some atom can never be appended under the constraints.
306fn greedy_from_leader(
307    ctx: &WcojDispatchCtx,
308    atom_vars: &[Vec<usize>],
309    cards: &[u64],
310    constrained: bool,
311    leader: usize,
312) -> Option<(Vec<usize>, u64)> {
313    let n = atom_vars.len();
314    let mut order = Vec::with_capacity(n);
315    let mut placed = vec![false; n];
316    let mut bound: std::collections::HashSet<usize> = std::collections::HashSet::new();
317    order.push(leader);
318    placed[leader] = true;
319    bound.extend(atom_vars[leader].iter().copied());
320    let mut running = cards[leader].max(1);
321    let mut peak = running;
322    for _ in 1..n {
323        let mut choice: Option<(usize, u64)> = None;
324        for a in 0..n {
325            if placed[a] || !is_addable(&atom_vars[a], &bound, constrained) {
326                continue;
327            }
328            let new_running = estimate_join_onto_prefix(ctx, atom_vars, cards, &order, running, a);
329            if choice.as_ref().map_or(true, |(_, c)| new_running < *c) {
330                choice = Some((a, new_running));
331            }
332        }
333        let (a, new_running) = choice?;
334        order.push(a);
335        placed[a] = true;
336        bound.extend(atom_vars[a].iter().copied());
337        running = new_running;
338        peak = peak.max(running);
339    }
340    Some((order, peak))
341}

343/// Estimated peak of a SPECIFIC processing order (used to score the current
344/// traversal order). Returns `None` if the order is not valid under the
345/// constraints (e.g. the traversal order is not prefix-key-joinable).
346fn eval_order_peak(
347    ctx: &WcojDispatchCtx,
348    atom_vars: &[Vec<usize>],
349    cards: &[u64],
350    order: &[usize],
351    constrained: bool,
352) -> Option<u64> {
353    let mut bound: std::collections::HashSet<usize> = std::collections::HashSet::new();
354    let mut running = 0u64;
355    let mut peak = 0u64;
356    for (step, &a) in order.iter().enumerate() {
357        if step == 0 {
358            running = cards[a].max(1);
359        } else {
360            if !is_addable(&atom_vars[a], &bound, constrained) {
361                return None;
362            }
363            running = estimate_join_onto_prefix(ctx, atom_vars, cards, &order[..step], running, a);
364        }
365        bound.extend(atom_vars[a].iter().copied());
366        peak = peak.max(running);
367    }
368    Some(peak)
369}

371/// Whether atom `a` (its per-column var ids) can extend a prefix that has
372/// already bound `bound`. Always requires connectivity (a shares ≥1 bound
373/// var); when `constrained`, the bound vars must additionally be a leading
374/// COLUMN prefix of `a` (FJ's probe-key rule).
375fn is_addable(
376    a_vars: &[usize],
377    bound: &std::collections::HashSet<usize>,
378    constrained: bool,
379) -> bool {
380    if !a_vars.iter().any(|v| bound.contains(v)) {
381        return false; // disconnected — never a Cartesian step
382    }
383    if constrained {
384        let split = a_vars.iter().take_while(|v| bound.contains(v)).count();
385        // need ≥1 bound key prefix, and no bound var AFTER an unbound one.
386        if split == 0 || a_vars[split..].iter().any(|v| bound.contains(v)) {
387            return false;
388        }
389    }
390    true
391}

393/// Estimated intermediate size after joining atom `a` onto the current
394/// left-deep prefix, via the most selective already-placed partner. Uses the
395/// independence model `running × (pairwise(p,a) / card(p))`.
396fn estimate_join_onto_prefix(
397    ctx: &WcojDispatchCtx,
398    atom_vars: &[Vec<usize>],
399    cards: &[u64],
400    placed: &[usize],
401    running: u64,
402    a: usize,
403) -> u64 {
404    let mut best_new = u64::MAX;
405    for &p in placed {
406        let (p_keys, a_keys) = shared_keys(&atom_vars[p], &atom_vars[a]);
407        if p_keys.is_empty() {
408            continue;
409        }
410        let pairwise = pairwise_estimate(ctx, cards, p, a, &p_keys, &a_keys);
411        let fan_out = pairwise as f64 / (cards[p].max(1)) as f64;
412        let new_running = ((running as f64) * fan_out).ceil().max(1.0);
413        let new_running = if new_running >= u64::MAX as f64 {
414            u64::MAX
415        } else {
416            new_running as u64
417        };
418        best_new = best_new.min(new_running);
419    }
420    // Caller guarantees `a` shares a bound var with some placed atom.
421    best_new.max(1)
422}

424/// Column positions of the join variables shared between `p` and `a`.
425fn shared_keys(p_vars: &[usize], a_vars: &[usize]) -> (Vec<usize>, Vec<usize>) {
426    let mut p_keys = Vec::new();
427    let mut a_keys = Vec::new();
428    for (pi, pv) in p_vars.iter().enumerate() {
429        if let Some(ai) = a_vars.iter().position(|av| av == pv) {
430            p_keys.push(pi);
431            a_keys.push(ai);
432        }
433    }
434    (p_keys, a_keys)
435}

437/// Pairwise join-cardinality estimate. Prefers the authoritative
438/// `StatsManager::estimate_join_cardinality` over the ACTUAL key columns when
439/// both relations have populated stats (cached selectivity / column distinct
440/// estimates then refine it); otherwise applies the same
441/// default model (10% selectivity) over the ground-truth buffer cardinalities,
442/// so statless single-rule joins (where scans never populate StatsManager) are
443/// still planned.
444fn pairwise_estimate(
445    ctx: &WcojDispatchCtx,
446    cards: &[u64],
447    p: usize,
448    a: usize,
449    p_keys: &[usize],
450    a_keys: &[usize],
451) -> u64 {
452    let stats_present = ctx
453        .stats
454        .get_relation_stats(ctx.slot_rels[p])
455        .map_or(false, |s| s.cardinality > 0)
456        && ctx
457            .stats
458            .get_relation_stats(ctx.slot_rels[a])
459            .map_or(false, |s| s.cardinality > 0);
460    if stats_present {
461        ctx.stats
462            .estimate_join_cardinality(ctx.slot_rels[p], ctx.slot_rels[a], p_keys, a_keys)
463    } else {
464        let prod = (cards[p] as u128) * (cards[a] as u128) / DEFAULT_JOIN_SELECTIVITY_DEN;
465        prod.min(u64::MAX as u128).max(1) as u64
466    }
467}

712    fn slots(n: usize) -> Vec<RelId> {
713        (0..n as u32).map(RelId).collect()
714    }

716    /// Adversarial chain q(A,E):-e1(A,B),e2(B,C),e3(C,D),e4(D,E): the ONLY
717    /// prefix-key-joinable order is left-to-right (which materializes the N²
718    /// prefix), while the binary plan can start from the 1-row tail. The
719    /// planner must DECLINE.
720    #[test]
721    fn order_planner_declines_unreorderable_blowup_chain() {
722        let cards = [100u64, 100, 10_000, 1];
723        let stats = stats_with_cards(&cards); // populated, but no cached selectivity
724        let rels = slots(4);
725        let ctx = nway_ctx(&stats, &rels);
726        // e1=[A,B] e2=[B,C] e3=[C,D] e4=[D,E] (vars A=0..E=4)
727        let atom_vars = vec![vec![0, 1], vec![1, 2], vec![2, 3], vec![3, 4]];
728        let model = CardinalityAwareCostModel::default();
729        assert!(
730            matches!(
731                model.plan_free_join_order(&ctx, &atom_vars, &cards),
732                FjOrderDecision::Decline
733            ),
734            "FJ-forced chain order blows up vs the tail-first binary plan → decline"
735        );
736    }

780    /// Alternating-cardinality 4-cycle: the traversal order starts on a big
781    /// relation (bad prefix), but a rotation starting on a 1-row relation is
782    /// both prefix-key-joinable AND competitive — the planner must REORDER to
783    /// a non-identity valid permutation.
784    #[test]
785    fn order_planner_reorders_alternating_cycle() {
786        let cards = [10_000u64, 1, 10_000, 1];
787        let stats = stats_with_cards(&cards);
788        let rels = slots(4);
789        let ctx = nway_ctx(&stats, &rels);
790        // r1=[A,B] r2=[B,C] r3=[C,D] r4=[D,A] (vars A=0,B=1,C=2,D=3)
791        let atom_vars = vec![vec![0, 1], vec![1, 2], vec![2, 3], vec![3, 0]];
792        let model = CardinalityAwareCostModel::default();
793        match model.plan_free_join_order(&ctx, &atom_vars, &cards) {
794            FjOrderDecision::Reorder(order) => {
795                let mut sorted = order.clone();
796                sorted.sort_unstable();
797                assert_eq!(sorted, vec![0, 1, 2, 3], "order must be a permutation");
798                assert_ne!(order, vec![0, 1, 2, 3], "must differ from traversal order");
799                assert!(
800                    eval_order_peak(&ctx, &atom_vars, &cards, &order, true).is_some(),
801                    "reordered plan must be prefix-key-joinable"
802                );
803            }
804            other => panic!("expected Reorder, got {:?}", DecisionDbg(&other)),
805        }
806    }

824    // Minimal Debug for panic messages (FjOrderDecision is not Debug).
825    struct DecisionDbg<'a>(&'a FjOrderDecision);
826    impl std::fmt::Debug for DecisionDbg<'_> {
827        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
828            match self.0 {
829                FjOrderDecision::KeepDefault => write!(f, "KeepDefault"),
830                FjOrderDecision::Reorder(o) => write!(f, "Reorder({o:?})"),
831                FjOrderDecision::Decline => write!(f, "Decline"),
832            }
833        }
834    }
